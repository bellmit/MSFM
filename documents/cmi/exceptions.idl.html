module exceptions
{
    typedef long ErrorCode;

    typedef short SeverityLevel;

    struct ExceptionDetails {
        string message;
        string dateTime;
        exceptions::SeverityLevel severity;
        exceptions::ErrorCode error;
    };

    exception AlreadyExistsException {
        exceptions::ExceptionDetails details;
    };

    exception AuthorizationException {
        exceptions::ExceptionDetails details;
    };

    exception AuthenticationException {
        exceptions::ExceptionDetails details;
    };

    exception CommunicationException {
        exceptions::ExceptionDetails details;
    };

    exception DataValidationException {
        exceptions::ExceptionDetails details;
    };

    exception NotFoundException {
        exceptions::ExceptionDetails details;
    };

    exception NotAcceptedException {
        exceptions::ExceptionDetails details;
    };

    exception NotSupportedException {
        exceptions::ExceptionDetails details;
    };

    exception SystemException {
        exceptions::ExceptionDetails details;
    };
    exception TransactionFailedException {
        exceptions::ExceptionDetails details;
    };

};
