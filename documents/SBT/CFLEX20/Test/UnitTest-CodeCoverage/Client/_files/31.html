485  /**
486   * Initiates the CAS logoff sequence. The method sets a flag that indicates a
487   * logoff is pending. If a logon is in progress, this method will return without
488   * executing the logoff sequence. When the logon completes, the thread executing
489   * the logon sequence will then execute the logoff sequence. If no logon is
490   * currently is progress, the thread calling this method will execute the cas
491   * logoff sequence.
492   */
493  public void initiateCasLogoff() {
494    if(mCasAccessStateStruct.mCasLogoffPending) {
495      return;
496    }
497 
498    synchronized(mCasAccessStateStruct) {
499      if (Log.isDebugOn()) {
500      Log.debug("CasAccessStateStruct is synchronized.......");
501      }
502 
503      // A logon is currently in progress so set the pending logoff flag to true.
504      // This should be set regardless of the logon state so that any subsequent
505      // logoffs/logons will be rejected while a logoff is pending.
506      mCasAccessStateStruct.mCasLogoffPending = true;
507 
508      if (mCasAccessStateStruct.mCasLogonComplete) {
509                        // The CAS logon is already complete so the thread calling this
510                        // method needs to execute the logoff sequence.
511 
512                        // Run logoff in its own thread to allow any callbacks to ruturn on
513                        // the thread that invokes initiateCasLogoff().
514                        new Thread() {
515                                public void run() {
516                                        logoff();
517                                        mFixCasSession.cleanup();
518                                }
519                        }.start();
520 
521                }
522        }
523    if (Log.isDebugOn()) {
524    Log.debug("CasAccessStateStruct is un-synchronized.......");
525    }
526  }
