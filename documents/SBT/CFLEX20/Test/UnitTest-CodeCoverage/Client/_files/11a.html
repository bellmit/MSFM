499                if (Log.isDebugOn())

502                            + legdetails.LegSecurityType[leg]);
503                }
504            }
505            else if (tag.equals(UserDefinedTagConstants.LEG_SYMBOL))
506            {
507                legdetails.LegSymbol[leg] = tags.nextToken();
508                if (Log.isDebugOn())
509                {
510                    Log.debug("mapLegDetails -> LegSymbol " + legdetails.LegSymbol[leg]
511                            + " for leg " + leg);
512                }
513            }
514            else if (tag.equals(UserDefinedTagConstants.LEG_MATURITY_MONTH_YEAR))
515            {
516                // ------------------------------------------------------------- //
517                // Need to accept both tag 610, tag 611 -
518                // but do not overlay tag 611 data with tag 610 data
519                // so test if legdetails.LegMaturityMonthYear[leg] is null
520                // ------------------------------------------------------------- //
521                if (!(legdetails.LegMaturityMonthYear[leg] == null)
522                        && !(legdetails.LegMaturityMonthYear[leg].trim() == ""))
523                {
524                    if (Log.isDebugOn())
525                    {
526                        Log.debug("mapLegDetails -> tag 610, tag 611 both found - tag 610 ignored");
527                    }
528                }
529                else
530                {
531                    legdetails.LegMaturityMonthYear[leg] = tags.nextToken();
532                    if (Log.isDebugOn())
533                    {
534                        Log.debug("mapLegDetails->legdetails.LegMaturityMonthYear(610)[" + leg
535                                + "]:" + legdetails.LegMaturityMonthYear[leg]);
536                    }
537                }
538            }
539            else if (tag.equals(UserDefinedTagConstants.LEG_MATURITY_DATE))
540            {
541                if (!(legdetails.LegMaturityMonthYear[leg] == null)
542                        && !(legdetails.LegMaturityMonthYear[leg].trim() == ""))
543                {
544                    if (Log.isDebugOn())
545                    {
546                        Log

549                }

551                // 611 should be YYYYMMDD
552                // need to put YYYYMM in legDetails.LegMaturityMonthYear[leg]
553                // need to put DD in legDetails.LegMaturityDay[leg]
554                // ------------------------------------------------------------- //
555                String tag611 = tags.nextToken();
556                if (Log.isDebugOn())
557                {
558                    Log.debug("tag611:" + tag611);
559                }

561                {
562                    legdetails.LegMaturityMonthYear[leg] = tag611.substring(0, 6);
563                    legdetails.LegMaturityDay[leg] = tag611.substring(6, 8);

565                    {
566                        Log.debug("mapLegDetails->legdetails.LegMaturityMonthYear(611)[" + leg
567                                + "]:" + legdetails.LegMaturityMonthYear[leg]);

569                                + legdetails.LegMaturityDay[leg]);
570                    }

572                catch (IndexOutOfBoundsException iob)

575                            "Tag 611, LegMaturityDate illegally formatted", 0);
576                }
577            }
578            
579            else if (tag.equals(UserDefinedTagConstants.LEG_STRIKE_PRICE))
