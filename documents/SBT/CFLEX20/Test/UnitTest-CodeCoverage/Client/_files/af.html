631        /**
632         *  This is a sophisticated little routine that compares a list of required MDEntryTypes
633         *  with the list received from a user in a MarketDataRequest message.
634         *
635         *  The routine does this by comparing each MDEntryType from the request against the
636         *  list of required entry types. Each time one of the request entry types is found the
637         *  corresponding element in the requiredEntryTypes is deleted.
638         *
639         *  WARNING: The contents of the requiredEntryTypes will be destroyed by this routine - so
640         *  the caller must make sure that the requiredEntryTypes can be recreated for subsequent
641         *  calls.
642         *
643         * @param  requiredEntryTypes String [] The entries we expect to find. Order is not important.
644         * @param  req MarketDataRequest The object that may have the specified entries.
645         * @return  boolean true if all of the candiates (and only the candidates) are found as MDEntryTypes
646         *
647         * NOR: Made the method public and static so it could be re-used by other market data types.
648         *      Probably should have been a separate class function or "reified" object.
649         */
650        public static boolean validEntries( final String[] requiredEntryTypes, final MarketDataRequest req )
651        {
652            // Go ahead and make sure the request has the required number of entries
653            // - no sense matching if we no there is the wrong number
654            if ( requiredEntryTypes.length != req.NoMDEntryTypes )
655            {
656                return false;
657            }
658 
659        // For each MDEntryType - try to find a match in the array of RequiredEntryTypes
660        int matchedEntries = 0;
661        for( int i = 0; i < req.NoMDEntryTypes; i++ )
662        {
663            // Just a simple linear search - as there are very few items to compare
664            for( int j = 0; j < requiredEntryTypes.length; j++ )
665            {
666                if(requiredEntryTypes[j] != null && req.MDEntryType[i].equals( requiredEntryTypes[j] ))
667                {
668                    matchedEntries++;
669                    requiredEntryTypes[j] = null;   // Here we null out the required entry type if we found it
670                    break;                                     // in the request - this will ensure no duplicate entries
671                }
672            }
673        }
674 
675        // If we have matched all requests to required entries then we have made it
676        if ( matchedEntries == req.NoMDEntryTypes )
677        {
678            return true;
679        }
680        return false;
681        }
