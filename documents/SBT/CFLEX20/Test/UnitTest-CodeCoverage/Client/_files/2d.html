460 * If the line is read successfully then the callback object is called to process the data.
461 */
462public void processScript()
463{
464    boolean clearComments = true;
465 
466    // Setup the pre-processor (first time thru there is no variable's file).
467    useVariablesFile( variablesFileName );
468         
469    // Get the sleep time between data sends.
470    setSleepTime();
471    Date initTime = new Date();
472    initMillSecs = initTime.getTime();
473    // Clear the comments
474    clearComments();
475    // Replay the script nbrOfIterations times.
476    for (int i = 0; i < nbrOfIterations; i++)
477    {
478        openFile();
479        String tempData, data = null;
480 
481        try
482        {
483            // Read till we reach end of file.
484            while ( !exiting && (data = file.readLine()) != null )
485            {
486 
487                  tempData = data.trim();
488 
489                if (tempData.startsWith(OUTPUT_COMMENT)) // Display comment.
490                {
491                    // Hold onto the display comment in an internal collection.
492                    // The comments are cleared before processing the
493                    // next comment that occurs after the first interrupt.
494                    tempData = tempData.substring(OUTPUT_COMMENT.length());
495                    tempData = preProcess(tempData);
496                    addComment( clearComments, tempData);
497                    clearComments = false;
498                    System.out.println(tempData);
499                }
500                else if (tempData.startsWith(SLEEP)) // Wait for input from user.
501                {
502                    long waitTime = Integer.parseInt(tempData.substring(SLEEP.length()).trim());
503                    System.out.println("Sleeping for :" + waitTime);
504                    Thread.sleep(waitTime);
505                }
506                else if (tempData.startsWith(INTERRUPT)) // Wait for input from user.
507                {
508                    // Clear the display comments first when we receive the next
509                    // display comment.
510                    clearComments = true;
511                    interrupt( tempData );
512                }
513                else if (tempData.startsWith(INCLUDE)) // Open the new variables file.
514                {
515                        variablesFileName = tempData.substring(INCLUDE.length()).trim();
516                    appendVariablesFile( variablesFileName );
517                }
518                else if (tempData.startsWith(SET_VARIABLE_NO_VALIDATION))
519                {
520                  final String productSet = tempData.substring(SET_VARIABLE_NO_VALIDATION.length()).trim();
521                  setValueNoValidation(productSet);
522 
523                }
524                else if (tempData.startsWith(SET_VARIABLE)) // Set the variable with a value.
525                {
526                    final String variableSet = tempData.substring(SET_VARIABLE.length()).trim();
527                    setValue( variableSet );
528                }
529                else if (tempData.startsWith(REPEAT_SECTION_ITERATIONS)) // Set the number of repeat section iterations
530                {
531                    int equalsSign = tempData.indexOf('=');
532                    String tempIterations = tempData.substring(equalsSign + 1).trim();
533                    nbrOfRepeatSectionIterations = Integer.parseInt(tempIterations);
534                }
535                else if (tempData.startsWith(START_REPEAT_SECTION)) // Set the mark at the beginning of the repeating seciton
536                {
537                    file.mark(bufferSize);
538                    repeatSectionMarked = true;
539                }
540                else if (tempData.startsWith(GOTO_REPEAT_SECTION)) // Goto the start of the repeating section in the file
541                {
542                    if(--nbrOfRepeatSectionIterations > 0)
543                    {
544                        file.reset();
545                    }
546                }
547                else if (tempData.startsWith(COMMENT) || (tempData.length() == 0)) // Comments
548                {
549                    // Ignore REGULAR comments.
550                }
551                else // This is a data line.
552                {
553                    // Pre process the message if needed and then call the callback object
554                    // to process the message.
555                    if (debugMode)
556                    {
557                        System.out.println(new Date()
558                    + ": Line Nbr(" + file.getLineNumber() + "), PROCESSED MESSAGE: " + preProcess(data));
559                    }
560                    else
561                    {
562                        final String messageText = preProcess(data);
563                        callback.processMessage( messageText.getBytes(), file.getLineNumber());
564                    }
565 
566                    sleep( sleepTime );
567                    if ( incrementalMsgRate > 0)
568                    {
569                      checkAndSetMsgRate(incrementalMsgRate,incrementalMsgRateTimeInterval);
570                    }
571                }
572            } // End while read script file completely.
573 
574            displayEndOfIteration( i );
575        }
576        catch( Exception e )
577        {
578            displayError(e);
579            System.exit(1);
580        }
581    }
582 
583    displayEndOfScript( );
584}
