506    protected void processStrategyOrder(FixCasSession fixCasSession,
507                                        OrderStruct anOrder,
508                                        LegOrderEntryStructV2[] legOrdersV2,
509                                        Order orderMessage,
510                                        ProductStruct productStruct)
511            throws SystemException, CommunicationException
512    {
513        try
514        {
515            if (Log.isDebugOn())
516            {
517                com.cboe.domain.util.ReflectiveStructBuilder.printStruct(anOrder,"processStrategyOrder");
518            }
519 
520            fixCasSession.getCas().getOrderEntry().acceptStrategyOrderV7(anOrder, legOrdersV2,productStruct);
521        }
522        catch(AlreadyExistsException e)
523        {
524            Log.exception(e.details.message, e);
525            throw ExceptionBuilder.systemException(e.details.message, 0);
526        }
527        catch(DataValidationException exception)
528        {
529            Log.exception(exception.details.message,exception);
530            switch(exception.details.error)
531            {
532            case DataValidationCodes.INVALID_SESSION:
533                rejectOrder(fixCasSession,orderMessage, anOrder,
534                     FixConstants.OrdRejReason.BROKER_OR_EXCHANGE_OPTION, exception.details.message);
535                break;
536            case DataValidationCodes.DUPLICATE_ID:
537                rejectOrder(fixCasSession,orderMessage, anOrder,
538                     FixConstants.OrdRejReason.DUPLICATE_ORDER, exception.details.message);
539                break;
540            case DataValidationCodes.ORDER_REJECTED_ON_RSS:
541                  rejectOrder(fixCasSession,orderMessage, anOrder,
542                     FixConstants.OrdRejReason.BROKER_OR_EXCHANGE_OPTION, exception.details.message);
543                break;
544            default:
545                int retry = fixCasSession.getSpreadInterval();
546                    if (retry > 0)             // Only retry when the interval is greater than 0
547                    {
548                        try {
549                            Log.information(new StringBuilder(130)
550                                    .append("DV Exception received on strategy order  ").append(orderMessage.ClOrdID)
551                                    .append(" classKey ").append(anOrder.classKey)
552                                    .append(" productKey ").append(anOrder.productKey)
553                                    .append(" sleeping for ").append(retry).append(" ms ").toString());
554                            Thread.sleep(retry);
555                            Log.information(new StringBuilder(40)
556                                    .append("Retrying strategy order").append(orderMessage.ClOrdID).toString());
557                            fixCasSession.getCas().getOrderEntry().acceptStrategyOrderV7(anOrder, legOrdersV2, productStruct);
558                        } catch (InterruptedException e) {
559                            e.printStackTrace();
560                            Log.exception(e);
561                        } catch (TransactionFailedException e) {
562                            Log.exception(e.details.message, e);
563                            rejectOrder(fixCasSession,orderMessage, anOrder,
564                                 FixConstants.OrdRejReason.BROKER_OR_EXCHANGE_OPTION, "Message Dispatch Failed:TFE");
565                        } catch (DataValidationException e) {
566                            Log.exception(e.details.message, e);
567                            rejectOrder(fixCasSession,orderMessage, anOrder,
568                                 FixConstants.OrdRejReason.BROKER_OR_EXCHANGE_OPTION, e.details.message);
569                        } catch (AlreadyExistsException e) {

571                            throw ExceptionBuilder.systemException(e.details.message, 0);
572                        } catch (NotAcceptedException e) {

591                        if (Log.isDebugOn())
