383        // 0. Check to see if the user has suppressed the message(s) we'd generate
384        QuoteStatusMapper qsm = new QuoteStatusMapper(fill);
385        String qOrdStatus = qsm.getFixOrdStatus();
386        if(fixCasSession.isSuppressedOrderStatusValue(qOrdStatus)) {
387            Log.alarm(new StringBuilder(75)
388                    .append(" Execution report suppressed for User: ").append(fixCasSession.getConnectionID())
389                    .append(" for OrderStatus: ").append(qOrdStatus).toString());
390            return;
391        }
392 
393        // 1. Attempt to Roll Up the Units
394        ContraPartyStruct[] compressedContras = summarizeContras(fill);
395 
396        // 2. Log after rolling up units
397        Log.information(new StringBuilder(110)
398                .append("FIX Contra Compression for Quote2: [ QuoteKey: ").append(quoteKey)
399                .append(" ProductKey: ").append(product.productKeys.productKey)
400                .append(" UserID: ").append(getFixCasSession().getConnectionID()).append(" ]").toString());
401        Log.information(new StringBuilder(70)
402                .append(" - pre-summary ContraCount: ").append(fill.contraParties.length)
403                .append(" - post-summary ContraCount: ").append(compressedContras.length).toString());
404 
405        // 3. Possibly send out multiple Reports
406        int messagesSent = 0;
407        int messagesToSend = compressedContras.length / contrasPerFillReport;
408        messagesToSend += ((compressedContras.length % contrasPerFillReport) == 0) ? 0 : 1;
409        int begIndx = 0;
410        int endIndx;
411        if ((compressedContras.length < contrasPerFillReport)) { 
412            endIndx = compressedContras.length; 
413        } else { 
414            endIndx = contrasPerFillReport; 
415        }
416        int contraCountThisMessage;
417 
418        do {
419            // setup for this message
420            contraCountThisMessage = endIndx - begIndx;
421            int quantityThisMessage = 0;
422            int fromIndx = begIndx;
423            ContraPartyStruct[] contrasThisMessage = new ContraPartyStruct[contraCountThisMessage];
424 
425            // transfer the contras for this message
426            for ( int i = 0 ; i <  contraCountThisMessage ; i++ ) {
427                contrasThisMessage[i] = compressedContras[fromIndx];
428                quantityThisMessage += contrasThisMessage[i].quantity;
429                fromIndx++;
430            } 
431            fill.contraParties = contrasThisMessage;
432 
433            // map the ExecutionReport message, and adjust fields as needed
434            ExecutionReport execRept = QuoteExecutionReportMapper.map(fill, product, quoteKey, statusChange, 
435                    getFixCasSession(), queueDepth);
436            execRept.LastShares = quantityThisMessage;
437            if (0 != messagesSent) {
438                execRept.ExecID = new StringBuilder(30)
439                    .append(execRept.ExecID).append(".").append(messagesSent).toString();
440            }
441 
442            // send the message
443            if (getFixCasSession().isFirmUser()){
444                execRept.header.SenderSubID = fill.userId;
445            }
446            getFixCasSession().sendFixMessage( execRept );
447 
448            // setup for the next message (possibly) 
449            messagesSent++;
450            begIndx += contrasPerFillReport;
451            endIndx += contrasPerFillReport;
452            if (endIndx > compressedContras.length) {
453                endIndx = compressedContras.length;
454            }
455            // Log after each message sent
456            Log.information(new StringBuilder(90)
457                    .append("FIX Contra Compression[Q:").append(quoteKey)
458                    .append(" P:").append(product.productKeys.productKey)
459                    .append(" U:").append(getFixCasSession().getConnectionID())
460                    .append(" - sent msg# ").append(messagesSent)
461                    .append(" of ").append(messagesToSend).toString());
462 
463        } while (messagesToSend > messagesSent);
464    }
