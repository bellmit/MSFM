878     * @param firstLeg
879     * @param leg
880     * @param noOfLegs
881     * @return
882     * @throws DataValidationException
883     * @throws SystemException
884     */
885    protected boolean mapLegDetails(FixLegDetailStruct legdetails, StringTokenizer tags,
886            boolean firstLeg, int leg, int noOfLegs) throws DataValidationException,
887            SystemException
888    {
889        boolean isSecurityType = false;
890        while (tags.hasMoreTokens())
891        {
892            String tag = tags.nextToken();
893            if (tag.equals(UserDefinedTagConstants.LEG_SECURITY_TYPE))
894            {
895                if (!(firstLeg))
896                {
897                    leg = leg + 1;
898                    isSecurityType = true;
899                    if (leg >= noOfLegs)
900                    {
901                        throw ExceptionBuilder.dataValidationException(new StringBuilder(45)
902                                .append("Leg count found exceeds Leg count requested").toString(),
903                                0);
904                    }
905                }
906                legdetails.LegSecurityType[leg] = tags.nextToken();
907                if (Log.isDebugOn())
908                {
909                    Log.debug("mapLegDetails -> LegSecuritytype for leg " + leg + " is "
910                            + legdetails.LegSecurityType[leg]);
911                }
912            }
913            else if (tag.equals(UserDefinedTagConstants.LEG_SYMBOL))
914            {
915                legdetails.LegSymbol[leg] = tags.nextToken();
