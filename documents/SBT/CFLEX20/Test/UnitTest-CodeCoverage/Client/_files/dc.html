504                                OrderDetailStruct origOrder) throws AuthorizationException, SystemException, NotFoundException, CommunicationException, DataValidationException {
505        OrderDetailStruct orderDetails = null;
506        String message = null;
507        orderDetails = origOrder;
508        ProductKeysStruct productKeysStruct = mapOrderProductKeys(orderDetails.orderStruct);
509 
510        try {
511            CancelToExecutionReportMapper mapper =
512                    new CancelToExecutionReportMapper(fixCasSession, cancelRequestMessage, orderDetails);
513            mapper.execRept.ExecType = FixConstants.ExecType.PENDING_CANCEL;
514            mapper.execRept.OrdStatus = FixConstants.OrdStatus.PENDING_CANCEL;
515            fixCasSession.getCas().getOrderEntry().acceptOrderCancelRequest(cancelRequest, productKeysStruct);
516 
517            if (!(fixCasSession.getEngineMode().equals(FixConstants.SbtValues.EngineMode.TT))) {
518                mapper.sendExecutionReport();
519            }
520        }catch (TransactionFailedException ex) {
521            Log.exception(ex.details.message, ex);
522            throw ExceptionBuilder.systemException(ex.getMessage(), 0);
523        }catch (NotFoundException ex) {
524            //
525            // Build and return a CancelReject because we couldn't find the order
526            //
527            Log.exception(ex.details.message, ex); // always log exceptions -Magic
528 
529            // Can't look up order status because it's unknown!
530            ExceptionResponseFactory factory = ExceptionResponseFactory.getInstance();
531            OrderCancelReject cancelReject = factory.createException(ex, cancelRequestMessage);
532            fixCasSession.sendFixMessage(cancelReject);
533        }catch (DataValidationException exception) {
534            Log.exception(exception.details.message, exception); // always log exceptions -Magic
535            CancelToCancelRejectMapper rejectMapper = null;
536 
537            switch (exception.details.error) {
538                case DataValidationCodes.NO_WORKING_ORDER:
539                    // NO_WORKING_ORDER: found in cache (but filled or cancelled)
540                case DataValidationCodes.INVALID_QUANTITY:
541                {

550                default:

552                    if (Log.isDebugOn()) {

557                }
558            }
559 
560            OrderCancelReject cancelReject;
561            try {

564            } catch (Exception e) {

577            StringBuilder nae = new StringBuilder(e.details.message.length()+75);

581                    .append(" ")
582                    .append(e.details.message).toString());
583            MessageObject response = ExceptionResponseFactory.getInstance().createException(e, cancelRequestMessage);
584            response.header.TargetCompID = cancelRequestMessage.header.SenderCompID;
585            if(response == null)
