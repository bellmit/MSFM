475    static String cleanText(String text) {

477            if (text == null) {
478                    return "";
479            }
480 
481            StringBuilder sb = new StringBuilder(text);
482            // Reverse iteration
483            for (int i=sb.length()-1; i >= 0; i--) {
484                    char c = sb.charAt(i);
485 
486                    if (Character.isISOControl(c)) {
487                            // If it's a CR, delete it, assuming it's part of CR-LF pair.
488                            if (c == '\r') {
489                                    sb.deleteCharAt(i);
490                            } else {
491                                    // Replace any other control character with a blank
492                                    sb.setCharAt(i, ' ');
493                            }
494                    }
495            }
496            return sb.toString();
497    }
