361     * Maps the LoginSessionType from a FIX Logon message.
362     * @return  LoginSessionType.SECONDARY if user requests a SECONDARY login otherwise
363     *          LoginSessionType.PRIMARY (default is PRIMARY)
364     * @throws  DataValidationException if SenderSubID tag is improperly formatted or contains an invalid value.
365     */
366    protected short getUserLoginType(final LogonMsg logonMsg) throws DataValidationException {
367        // The SenderSubID field should be in the format: USERNAME:PASSWORD:USERLOGINTYPE
368        StringTokenizer tokenizer = new StringTokenizer(logonMsg.header.SenderSubID, ":");
369        int tokenCount = tokenizer.countTokens();
370        if (Log.isDebugOn()) {
371            Log.debug("getUserLoginType: token count = " + tokenCount);
372        }
373 
374        // If there are only two tokens in the string the LoginSessionType is not
375        // specified. Default value is PRIMARY.
376        if(tokenCount <= 2) {
377            return LoginSessionTypes.PRIMARY;
378        }
379 
380        // Create an array of the tokens
381        String tokens[] = new String[tokenCount];
382        for(int i = 0; i < tokenCount; i++) {
383            tokens[i] = tokenizer.nextToken();
384        }
385 
386        if(tokens[2] != null && tokens[2].equalsIgnoreCase(PRIMARY)) {
387            return LoginSessionTypes.PRIMARY;
388        }
389        else if(tokens[2] != null && tokens[2].equalsIgnoreCase(SECONDARY)) {
390            return LoginSessionTypes.SECONDARY;
391        }
392        else {
393            // The SenderSubID is improperly formatted or the ORDER PUBLISH flag is invalid.
394            throw ExceptionBuilder.dataValidationException("Improperly formatted SenderSubID or invalid UserLoginType: "
395                                                            + logonMsg.header.SenderSubID, 0);
396        }
397    }
