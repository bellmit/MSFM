402        if (order == null) {
403                throw new StrategyNotFoundException("Order is null");
404        }

412 
413    protected BrokerStrategy findOrderStrategy(Order order, boolean useTargetState, short contingencyType)
414            throws StrategyNotFoundException
415    {
416        short state = getStateCode(order.getProductKey(), useTargetState);
417    
418    boolean nbboProtection = order.needsNBBOProtection();
419    boolean useMarketOpenStrategy = useMarketOpenStrategy(order, useTargetState);
420 
421    if(order.isLinkageOrder())
422    {
423        return findLinkageOrderStrategy(state, order.getOrderOriginType());
424    }
425    
426    if(order.getOrderOriginType() == InternalOrderOriginTypes.CROSS_PRODUCT_LEG_SYNTHETIC_ORDER)
427    {
428        return this.findCrossProductLegOrderStrategy(state, nbboProtection);
429    }
430    
431    BrokerStrategy targetStrategy;
432        if (contingencyType == ContingencyTypes.NONE ||
433            contingencyType == ContingencyTypes.RESERVE || 
434            contingencyType == ContingencyTypes.NOTHELD || contingencyType == ContingencyTypes.DO_NOT_ROUTE) {
435                if (order.getPrice().isMarketPrice()) {
436                        targetStrategy = findMarketOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
437                }
438                else {
439                        targetStrategy = findLimitOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
440                }
441        }
442        else if (contingencyType == ContingencyTypes.IOC || contingencyType == ContingencyTypes.FOK) {
443                targetStrategy = findTimedOrderStrategy(state, nbboProtection, order.generatesNBBOFlash());
444        }
445    else if (contingencyType == ContingencyTypes.AON) {
446        // AON orders are treated the same as a regular limit order
447        targetStrategy = findLimitOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
448    }
449    else if (contingencyType == ContingencyTypes.OPG) {
450        targetStrategy = findOPGOrderStrategy(state, useMarketOpenStrategy);
451    }
452    else if (order.isCrossOrder()) {
453        targetStrategy = findCrossOrderStrategy(state);
454    }
455    else if (contingencyType == ContingencyTypes.STP || contingencyType == ContingencyTypes.STP_LOSS) {
456        // Inactive stop orders need to have stop price met to be made active, active stop orders
457        // have already had the contingency met and can be processed as if they don't have a
458        // contingency.
459        if (order.isWaiting()) {
460            targetStrategy = findStopOrderStrategy(state, true /* market order */, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
461        }
462        else {
463                        targetStrategy = findMarketOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
464                }
465    }
466    else if (contingencyType == ContingencyTypes.STP_LIMIT) {
467        // Inactive stop orders need to have stop price met to be made active, active stop orders
468        // have already had the contingency met and can be processed as if they don't have a
469        // contingency.
470        if (order.isWaiting()) {
471            targetStrategy = findStopOrderStrategy(state, false /* not a market order */, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
472        }
473                else {
474                        targetStrategy = findLimitOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
475                }
476    }
477    else if (contingencyType == ContingencyTypes.INTERMARKET_SWEEP) {
478        targetStrategy = findSweepOrderStrategy(state);
479    }else if(contingencyType == ContingencyTypes.AUCTION_RESPONSE && isFlexOrder(order.getProductKey(),order.getProductType())){
480        targetStrategy =  findLimitOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
481    }
482    else
483    {
484        throw new StrategyNotFoundException("Order strategy not found, unsupported contingency code of " + contingencyType);
485    }
486        return targetStrategy;
487}
