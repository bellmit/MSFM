375    private final void setQtyBreakdownForPreviousTradable()
376    {
377        if (remoteFills == null || remoteFills.length == 0 || localTRs == null || localTRs.length == 0 || localTRs[nextLegIndex]==null )
378        {
379                // If the stock leg or the next options leg has not traded, should not set the breakdown
380                // quantity for the current options leg.
381                return;
382        }
383        else
384        {
385                int qtyBreakdownLength  = 0;
386                int tradeReportPointer =0;
387                // If this is the first time we are filling the trade for this leg, number of trade reports
388                // is equal to number of trade reports of the next leg.
389                if(localTRs[curLegIndex] == null)
390                {
391                        qtyBreakdownLength = localTRs[nextLegIndex].size();
392                }
393                // if there are already trade reports in the current leg, then the number of new trade reports will
394                // be the difference between the trade report quantites of next lg and current leg.
395                else
396                {
397                        qtyBreakdownLength = localTRs[nextLegIndex].size() - localTRs[curLegIndex].size();
398                        tradeReportPointer = localTRs[curLegIndex].size();
399                }
400 
401            curLegTradable.qtyBreakDown = new int[qtyBreakdownLength];
402 
403            for (int i=0; i<qtyBreakdownLength; i++)
404            {
405                curLegTradable.qtyBreakDown[i] = localTRs[nextLegIndex].get(tradeReportPointer).getQuantity() / strategyInfo.spreadParams.legs[nextLegIndex].legRatioQuantity;
406                ++tradeReportPointer;
407            }
408     }
409    }
410    //  TODO: if we have more local legs in the future,
411    //      we might need to translate fill reports into trade reports
412    //      we also need a new method handleNextLegFillReport()
413    //      which handles remoteFills[nextLegIndex] directly
414    //      assuming it's populated properly by sub sequence calls
415    private final void handleLastLegFillReports(FilledReportStruct[] curFill)
416    throws DataValidationException, NoTradeException
417    {
418        int legTradedQty = 0;
419 
420        if (remoteFills[nextLegIndex] == null)
421        {
422            remoteFills[nextLegIndex] = new ArrayList<FilledReportStruct>();
423        }
424 
425        if (curFill == null || curFill.length == 0)
426        {
427        }
428        else
429        {
430 
431            curLegTradable.qtyBreakDown = new int[curFill.length];
432 
433            for (int i = 0; i < curFill.length; i++)
434            {
435                ((OrderImpl)getOrder()).addFillReportToList(curFill[i]);
436                legTradedQty += curFill[i].tradedQuantity;
437                curLegTradable.qtyBreakDown[i] = curFill[i].tradedQuantity
438                        / strategyInfo.spreadParams.legs[nextLegIndex].legRatioQuantity;
439                createLegTradable(nextLegIndex, curFill[i].tradedQuantity);
440                remoteFills[nextLegIndex].add(curFill[i]);
441            }
442        }
443        validateNextLegTradedQty(legTradedQty);
444

446                / strategyInfo.spreadParams.legs[nextLegIndex].legRatioQuantity;
447    }
