356         * @return Object[] received on processing queue.
357         */
358        public Object[] getNextQueueEntry()
359        {
360            Object[] nextEntry = null;
361            try
362            {
363                nextEntry = myQueue.dequeueFully(myDequeueBlockSize, myBlockWaitTime, Queue.INFINITE_TIMEOUT);
364            }
365            catch (QueueException e)
366            {
367                Log.alarm("Received exception on market data queue in thread " + this.getName()
368                        + " " + e);
369                try
370                {
371                    Thread.currentThread().sleep(100);
372                }
373                catch (InterruptedException ex)
374                {
375                 // avoid spinning wildly on a constant dequeue failure.
376                }
377            }
378            return nextEntry;
379        }
380 
381        /**
382         * Processes events from market data queue.
383         */
384        public void run()
385        {
386            Object[] nextEntry;
387            QueueEntry entry;
388            Exception dequeueEx = null;
389            if (myQueue != null)
390            {
391                myQueue.reportDequeueThreadRunning();
392            }
393            try
394            {
395                while (true)
396                {
397                    nextEntry = getNextQueueEntry();
398                    if (nextEntry != null && nextEntry.length > 0)
399                    { // queue will return null entry on time out.
400                        try
401                        {
402                            Transaction.startTransaction();
403                            for (int i = 0; i < nextEntry.length; i++)
404                            {
405                                entry = (QueueEntry) nextEntry[i];
406                                entry.execute();
407                            }
408                            Transaction.commit();
409                        }
410                        catch (Exception e)
411                        {
412                            Log.exception("Received exception on market data queue in thread when processing in run() "
413                                    + this.getName() + " " + e, e);
414                            Transaction.rollback();
415                            dequeueEx = e;
416                        }
417                    }
418                }
419            }
420            catch (Exception e)
421            {
422                dequeueEx = e;
423            }
424            finally
425            {
426                synchronized (this)// notify thread waiting for this thread to stop
427                {
428                    notifyAll();
429                }
430                if (myQueue != null)
431                {
432                    myQueue.reportDequeueThreadExited(dequeueEx);
433                }
434            }
435        }
436        // end of MarketDataReportProcessorThread inner class
437    }
