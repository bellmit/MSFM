328        private void addToExchangeList(ExchangeEnumForCodec p_exchange, int p_vol,
329                        IntFieldCodec p_exchangeBits,
330                        SequenceFieldCodec p_volumes)
331        {
332            final int bits = p_exchangeBits.getValue();
333//            if(p_exchange.isBitSet(bits))
334//                throw new IllegalArgumentException("exch/vol "+p_exchange+"/"+p_vol+" already in array");
335        if(!p_exchange.isBitSet(bits))
336        {
337            p_exchangeBits.setValue(p_exchange.setBit(bits));
338        }
339        
340        final int bitsLower = bits & (p_exchange.setBit(0)-1);
341        final int volAryIdx = Integer.bitCount(bitsLower);
342        final int numEls = p_volumes.getNumElements();
343        for (int i=numEls; i > volAryIdx; i--)
344        {
345            final int other = ((CompressedIntFieldCodec)p_volumes.getElement(i-1)).getValue();
346            ((CompressedIntFieldCodec)p_volumes.getElement(i)).setValue(other);
347        }
348        ((CompressedIntFieldCodec)p_volumes.getElement(volAryIdx)).setValue(p_vol);
349        }
