680    private void overideBillingTypeCode(ParticipantItem buyer,ParticipantItem seller,Price tradePrice)
681    {
682         Tradable buyerTradable = buyer.getParticipantTradable();
683         Tradable sellerTradable = seller.getParticipantTradable();
684         char buyerBillingCode = buyer.getBillingType();
685         char sellerBillingCode = seller.getBillingType();
686         
687         if ( buyerTradable instanceof SpreadTradableImpl)
688         {
689             buyerTradable=((SpreadTradableImpl)buyerTradable).getTradable();
690         }
691         
692         if ( sellerTradable instanceof SpreadTradableImpl)
693         {
694             sellerTradable=((SpreadTradableImpl)sellerTradable).getTradable();
695         }
696         
697         if (buyerBillingCode == BillingTypeIndicators.FLASH ||buyerBillingCode == BillingTypeIndicators.FLASH_PRICE_IMP)
698         {
699             if (isPriceImprovement(buyer.getParticipantTradable(), tradePrice))
700             {
701                 buyer.setBillingType(BillingTypeIndicators.FLASH_PRICE_IMP);
702                 seller.setBillingType(BillingTypeIndicators.FLASH_RESPONSE_PRICE_IMP);
703             }
704             else
705             {
706                 buyer.setBillingType(BillingTypeIndicators.FLASH);
707                 seller.setBillingType(BillingTypeIndicators.FLASH_RESPONSE);
708             }
709         }
710         else if (sellerBillingCode == BillingTypeIndicators.FLASH ||sellerBillingCode == BillingTypeIndicators.FLASH_PRICE_IMP)
711         {
712             if (isPriceImprovement(seller.getParticipantTradable(), tradePrice))

715                 buyer.setBillingType(BillingTypeIndicators.FLASH_RESPONSE_PRICE_IMP);
716             }
717             else

719                 seller.setBillingType(BillingTypeIndicators.FLASH);
720                 buyer.setBillingType(BillingTypeIndicators.FLASH_RESPONSE);
721             }
722         }
723         else if (sellerBillingCode == BillingTypeIndicators.CROSS ||sellerBillingCode == BillingTypeIndicators.CROSS_PRICE_IMP)

737             }
738         }
