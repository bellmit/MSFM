576        
577    BusinessDay currentDay = getBusinessDayHome().findCurrent();
578 
579    // don't set timers if day isn't started.
580    if (currentDay.getState() != BusinessDayStates.STARTED) {
581        return;
582    }
583 
584    if (productListener == null) {
585        productListener = new ProductTimeListener();
586    }
587 
588    // remove current timers in case times or flags were changed
589    currentDay.removeListener(productListener, BusinessDayListener.PRODUCT_PRE_OPEN_TIMER_TYPE);
590    currentDay.removeListener(productListener, BusinessDayListener.PRODUCT_OPEN_TIMER_TYPE);
591    currentDay.removeListener(productListener, BusinessDayListener.PRODUCT_CLOSE_TIMER_TYPE);
592    
593         if (BCRolloutTradingPropertyConsumerHomImpl.find().isAllBCRollout(getSession().getSessionName()))
594        {
595                Log.information("Product state timer for :"+elementName+" is set on BC");
596                return;
597        }
598 
599         Log.information("Product state timer for :"+elementName+" is set on GLOBAL");
600    
601    if (autoPreOpenProducts()) {
602        currentDay.registerListener(productListener, BusinessDayListener.PRODUCT_PRE_OPEN_TIMER_TYPE, getProductPreOpenTime());
603    }
604    if (autoOpenProducts()) {
605        currentDay.registerListener(productListener, BusinessDayListener.PRODUCT_OPEN_TIMER_TYPE, getProductOpenTime());
606    }
607    if (autoCloseProducts()) {
608        currentDay.registerListener(productListener, BusinessDayListener.PRODUCT_CLOSE_TIMER_TYPE, getProductCloseTime());
609    }
