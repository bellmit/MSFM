843 * 2. verify if reporting classes with curSym and newSym exist and they should have
844 *    the same product class.
845 */
846private void validateWhenSymbolChange(String curSym, String newSym, short prodType)
847throws DataValidationException
848{

850    ReportingClass newRptClass;

853    }

856        msg = msg + "The reporting class to be adjusted does not exists. ";
857        msg = msg + "CurrnetSym:NewSym = " + curSym + ":" + newSym;
858        Log.alarm(msg);
859        throw ExceptionBuilder.dataValidationException(msg, DataValidationCodes.INVALID_REPORTING_CLASS);
860    }
861    try {
862        newRptClass = getReportingClassHome().findBySymbol(newSym,prodType);
863        if (curRptClass.getProductClass().getClassKey() != newRptClass.getProductClass().getClassKey()){
864            String msg = "ReportingClassAjustmentImpl >>> Can not make adjustment. ";
865            msg = msg + "Current and new reporting classes are not in the same product class: ";
