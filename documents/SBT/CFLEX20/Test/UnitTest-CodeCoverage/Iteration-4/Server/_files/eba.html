411            {
412               legFill = remoteLegFills[i].get(fillLevel);
413               legQty[i][0] = legFill.tradedQuantity;
414               legPrice[i][0] = PriceFactory.create(legFill.price);
415               oneFill.put(legTradables[i].get(fillLevel), new FilledReportStruct[] {legFill});
416               getCrossProductTradeRelationHome().addRemoteLegTradeRelation(spreadTR, legFill);
417            }
418            else
419            {
420                legTR = (TradeReportImpl)localLegTRs[i].get(fillLevel);
421                legQty[i][0] = legTR.getQuantity();
422                legPrice[i][0] = legTR.getPrice();
423                //TODO: bad coding::child.setParent is not the same as parent.addChild()
424                ((TradeReportImpl)spreadTR).addChildReport(legTR);
425            }
426            if (levelQty==-1)
427            {
428                levelQty = legQty[i][0] / strategyInfo.spreadParams.legs[i].legRatioQuantity;
429            }
430        }
431        Price levelPrice = DerivedQuoteUtil.calculateSpreadPrice(strategyInfo, legQty, legPrice, levelQty, spreadSide);
432        orderSide.setTradePrice(levelPrice);
433        orderSide.addElement(new ParticipantItemImpl(getOrder(), levelQty));
434        return levelQty;
435    }
