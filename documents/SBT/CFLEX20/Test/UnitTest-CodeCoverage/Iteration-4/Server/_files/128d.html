440    /** 
441     * 
442     *         1. end AON Complex Auction trade  
443     *  beginPrice: trade begining price for each step. 
444     *  endPrice: trade ending price for each step 
445     *      
446     *  @author Cognizant Technology Solutions.    
447     */ 
448     protected static int endAONComplexAuctionWithTrade(AuctionInternal auction, Tradable incomingTradable, Price tradePrice ) 
449    { 
450         Order firmOrder = auction.getFirmMatchOrder(); 
451         Side auctionSide = auction.getSide(); 
452         Side responseSide = auctionSide.getOtherSide(); 
453         Price firmOrderPrice = firmOrder.getPrice(); 
454         int auctionQuantity = auction.getTotalAuctionQuantity(); 
455         TradableList auctionResponsesBetterPrice = new TradableListImpl(); 
456         TradableList auctionResponsesEqualPrice = new TradableListImpl(); 
457         TradableList betterPriceTradables = new TradableListImpl(); 
458         TradableList equalPriceTradables = new TradableListImpl(); 
459         int auctionRespEqualQty = 0; 
460         int spreadOrderEqualPriceQty = 0; 
461         int spreadOrderBetterPriceQty = 0; 
462         int auctionRespBetterQty = 0; 
463         int betterResp = 0; 
464         int equalResp = 0; 
465         int spreadEqualResp = 0; 
466         int spreadBetterResp = 0; 
467         TradableList spreadOrderEqualPriceTradables = new TradableListImpl(); 
468         TradableList spreadOrderBetterPriceTradables = new TradableListImpl(); 
469         TradableList mktOrderTradables = new TradableListImpl(); 
470         int equalPriceTradableQuantity = 0; 
471         int result = AuctionProcessorFlexSpreadAIMImpl.NO_AON_TRADE; 
472         TradableList responseSideTradables = new TradableListImpl();
473         TradableList spreadOrderTradables = new TradableListImpl(); 
474         try 
475         { 
476             
477             responseSideTradables = nTradeServiceImpl.getAONAuctionResponses( auction.getAuctionBook(), tradePrice); 
478             if (responseSideTradables != null) {
479                 for (int i = 0; i < responseSideTradables.size(); i++) 
480                 { 
481                     Tradable tradable = responseSideTradables.get(i); 
482                     if ((tradable.isAuctionResponse()) 
483                             && (tradable.getPrice().equals(getBetterPrice(responseSide, firmOrderPrice, 
484                                     tradable.getPrice())))) 
485                     { 
486                         if ( tradable.getPrice().equals(firmOrderPrice)) { 
487                             auctionResponsesEqualPrice.add(equalResp, tradable); 
488                             auctionRespEqualQty = auctionRespEqualQty + tradable.getQuantityAllowed(); 
489                             ++equalResp; 
490                         } 
491                         else { 
492                             auctionResponsesBetterPrice.add(betterResp, tradable); 
493                             auctionRespBetterQty = auctionRespBetterQty + tradable.getQuantityAllowed(); 
494                             ++betterResp; 
495                         } 
496                     } 
497                 } 
498             }
499             /* spread order tradables */ 
500             
501             nTradeServiceImpl.addAONTradablesFromOrderBook(auction.getAuctionBook().getOrderBook(), responseSide, tradePrice, spreadOrderTradables);
502             if (incomingTradable != null) {
503                 spreadOrderTradables.add(incomingTradable);
504             }
505             if (spreadOrderTradables != null) {
506                 for (int i = 0; i < spreadOrderTradables.size(); i++) 
507                 { 
508                     Tradable tradable = spreadOrderTradables.get(i); 
509                     if (tradable.getPrice().equals(getBetterPrice(responseSide, firmOrderPrice,tradable.getPrice()))) 
510                     { 
511                         if ( tradable.getPrice().equals(firmOrderPrice)) { 
512                             spreadOrderEqualPriceTradables.add(spreadEqualResp, tradable); 
513                             spreadOrderEqualPriceQty = spreadOrderEqualPriceQty + tradable.getQuantityAllowed(); 
514                             ++spreadEqualResp; 
515                         } 
516                         else { 
517                             spreadOrderBetterPriceTradables.add(spreadBetterResp, tradable); 
518                             spreadOrderBetterPriceQty = spreadOrderBetterPriceQty + tradable.getQuantityAllowed(); 
519                             ++spreadBetterResp; 
520                         } 
521                     } 
522                 }
523             }
524             getMarketableOrders(spreadOrderTradables, auction, mktOrderTradables);
525 
526             if (spreadOrderTradables.isEmpty() && responseSideTradables.isEmpty()) {
527                result = AuctionProcessorFlexSpreadAIMImpl.AON_PO_MO_TRADE; 
528             }
529 
530             /* with Orders at better price than Matched order*/ 
531             if (auctionRespBetterQty + spreadOrderBetterPriceQty >= auctionQuantity) { 
532                 /* Trade primary order with Auction responses, quotes and Opposite Orders 
533                  * Matched Order gets cancelled 
534                  */  
535                 result = AuctionProcessorFlexSpreadAIMImpl.AON_MO_CANCEL; 
536              }    
537      
538             /* customer orders with MKT price */ 
539             if (hasCustomerOrders(mktOrderTradables)) { 
540                 if (getResponseTradableQuantity(mktOrderTradables, auctionQuantity) >= auctionQuantity) { 
541                     /* Trade primary with opposite orders 
542                      * Matched order gets cancelled 
543                      */ 
544                     result = AuctionProcessorFlexSpreadAIMImpl.AON_MO_CANCEL; 
545                 } 
546                 else { 
547                     /* primary and Matched order gets cancelled 
548                      * opposite orders either trade with book @mkt price and remaining quantity is routed away 
549                      */ 
550                    // No MO or PO 
551                    result = AuctionProcessorFlexSpreadAIMImpl.NO_AON_TRADE;
552                 } 
553             } 
554              
555             /* only Auction Responses */ 
556             if (betterPriceTradables.size() == 0 && equalPriceTradables.size() == 0 && 
557                     spreadOrderBetterPriceTradables.size() == 0 && spreadOrderEqualPriceTradables.size() == 0)
558             {
559                  if (!auctionResponsesBetterPrice.isEmpty() && auctionRespBetterQty >= auctionQuantity)
560                  {
561                    /* auction response price is better and enough quantity */
562                    result = AuctionProcessorFlexSpreadAIMImpl.AON_MO_CANCEL;
563                  }
564                  else if (!auctionResponsesBetterPrice.isEmpty() || !auctionResponsesEqualPrice.isEmpty())
565                  {
566                      /* auction response price is equal or better than matched price and quantity is not sufficient  
567                       * */ 
568                      /* Trade Primary Order with Matched Order  
569                         * and Auction Responses get cancelled */ 
570                      result = AuctionProcessorFlexSpreadAIMImpl.AON_PO_MO_TRADE; 
571                  }
572                  else
573                  {
574                      result = AuctionProcessorFlexSpreadAIMImpl.AON_PO_MO_TRADE; 
575                  }
576            } 
577            /* with non Customer Orders at same price as Matched order*/ 
578             // if ((betterPriceTradables.size() == 0 && spreadOrderBetterPriceTradables.size() == 0 && auctionResponsesBetterPrice.isEmpty()
579              if ((auctionRespBetterQty + spreadOrderBetterPriceQty < auctionQuantity)
580                    && (equalPriceTradables.size() != 0 || spreadOrderEqualPriceTradables.size() != 0))
581              
582              { 
583                 if  ((equalPriceTradables.size() > 0 && !hasCustomerOrders(equalPriceTradables)) ||
584                         (spreadOrderEqualPriceTradables.size() > 0 && !hasCustomerOrders(spreadOrderEqualPriceTradables))) 
585                 { 
586                      /* Trade Primary Order with Matched Order  
587                       * Auction responses get cancelled */ 
588                     result = AuctionProcessorFlexSpreadAIMImpl.AON_PO_MO_TRADE; 
589                 } 
590                 else /* Customer orders at same price*/
591                 { 
592                      /*check for non AON customer Orders at same price */ 
593                     if ((equalPriceTradables.size() > 0 && !hasAONCustomerOrders(equalPriceTradables)) || 
594                             (spreadOrderEqualPriceTradables.size() > 0 && !hasAONCustomerOrders(spreadOrderEqualPriceTradables)))
595                     { 
596                          /* check for quantity */ 
597                         if (auctionRespEqualQty + equalPriceTradableQuantity + spreadOrderEqualPriceQty >= auctionQuantity)
598                         { 
599                                //Trade Primary Order with Auction responses, quotes and Opposite Orders 
600                                // Matched Order gets cancelled 
601                             result = AuctionProcessorFlexSpreadAIMImpl.AON_MO_CANCEL; 
602                          } 
603                         else /* not enough quantity */ 
604                         { 
605                                  // no trade cancel all */ 
606                              // No MO or PO 
607                              result = AuctionProcessorFlexSpreadAIMImpl.NO_AON_TRADE;
608                         } 
609                     } 
610                 } 
611            } 
612             
613            /* AON customer Orders */ 
614            if ((equalPriceTradables.size() > 0 && hasAONCustomerOrders(equalPriceTradables) ) || 
615                    (spreadOrderEqualPriceTradables.size() > 0 && hasAONCustomerOrders(spreadOrderEqualPriceTradables)) || 
616                    (betterPriceTradables.size() > 0 && hasAONCustomerOrders(betterPriceTradables)) || 
617                    (spreadOrderBetterPriceTradables.size() > 0 && hasAONCustomerOrders(spreadOrderBetterPriceTradables)))  
618            { 
619              if ((spreadOrderBetterPriceQty == auctionQuantity) || 
620                       (equalPriceTradableQuantity == auctionQuantity) ||
621                       (spreadOrderEqualPriceQty == auctionQuantity))  
622              { 
623                  /* Trade primary order with opposite orders 
624                   * Matched order gets cancelled. 
625                   */ 
626                  result = AuctionProcessorFlexSpreadAIMImpl.AON_MO_CANCEL; 
627              } 
628              else 
629              { 
630                  /* Trade primary order with Matched order  
631                   * and opposite orders will be either booked  
632                   * or traded with the book. */
633                  result = AuctionProcessorFlexSpreadAIMImpl.AON_PO_MO_TRADE; 
634              }   
635            } 
636     
637            return result; 
638         } 
639         catch (Exception e2) 
640         { 
641             //any other condition will consider as an error condition. Log system high alarm. But continue 
642             //go on to expire the auction. 
643             Log.exception("AONComplexAuctionHelper>>endAONComplexAuctionWithTrade. No trade is done due to the following error.", e2); 
644             return 0;
645         }
