713/**
714 * If scale is < 1 then 256 is assumed.  Read integerDigits chars and interpret
715 * them as the integer portion of hte price value.  Read a single character:
716 * if it's a space then read the remainder as a fraction, applying the scale
717 * value to it.  If the character's a '.' then interpret the remainder as a
718 * decimal portion.  Any other character value will cause a ParseException.
719 * Any other non-digital characters will cause a ParseException.
720 * 
721 * @author Steven Sinclair
722 * @return com.cboe.util.Price
723 * @param integerDigits int
724 * @param remainderDigits int
725 * @param scale int
726 */
727public Price readStandardPrice(int integerDigits, int remainderDigits, long scale)
728        throws IOException, ParseException
729{
730        // Since the integer portion can be "M" or "MKT" (instead of a numeric string),
731        // we need to recognize this case before proceeding.
732        //
733        long highestDigit = -1;
734        boolean neg = false;
735        if (integerDigits > 0)
736        {
737                char firstChar = (char)readByte();
738                if (firstChar == 'M')
739                {
740                        readString(FIXED_LENGTH, integerDigits + remainderDigits);
741                        return PriceFactory.create(Price.MARKET_STRING);
742                }
743                --integerDigits;
744                if (Character.isDigit(firstChar) && firstChar != '0')
745                {
746                        highestDigit = firstChar - '0';
747                }
748                neg = (firstChar == '-');
749        }
750        
751        long price = readLongAsString(integerDigits);
752        if (neg)
753        {
754                price *= -1;
755        }
756        else if (highestDigit > 0)
757        {
758                long priceDigits = Math.abs(price);
759                while (priceDigits > 0)
760                {
761                        highestDigit *= 10;
762                        priceDigits  /= 10;
763                }
764                price += highestDigit;
765        }
766        switch ((char)readByte())
767        {
768                case '.':
769                        return PriceFactory.create(price + readDecimal(remainderDigits));
770                case ' ': return PriceFactory.create(price + readLongAsString(remainderDigits)/(double)scale);
771                default:  throw new ParseException("Error reading price: expected '.' or ' '.", getPos());
772        }
773}
774        /**
775         *  Read a string based on the given policy, which may be paramterized by the int policyValue.
776         *  @param policy  Any of the following are acceptable policies:
777         *                <lu>
778         *          <li> <b>NULL_TERM</b> the string is terminated by a null character ('\0').  policyValue is ignored.
779         *                        <li> <b>FIXED_LENGTH</b> the string length is assumed to be exactly the value of polivyValue.
780         *          <li> <b>FIXED_LENGTH_OR_NULL</b> like fixced length, but will stop rreading if '\0' is encountered.
781         *                        <li> <b>TEXTUALLY_ENCODED_LENGTH</b> the next 'policyValue' characters are expected to be a numeric string,
782         *                                        which will be parsed to determine the number of characters to read to make up the string.
783         *                        <li> <b>SHORT_ENCODED_LENGTH</b> the next two bytes are assumed to be an unsigned short value representing the string length.
784         *                        <li> <b>BYTE_ENCODED_LENGTH</b> the next byte is treated as an unsigned 8-bit integer representing the string length.
785         *      </lu>
786         *        @param policyValue the integer value associated with the selected policy.  It's value is defined in the list accompanying
787         *                the <code>policy</code> parameter.
788         *  @param buf The StringBuilder to which the string will be appended
789         */
790        public String readString(int policy, int policyValue)
791                throws IOException
792        {
793                int len = -1;
794                switch (policy)
795                {
796                        case NULL_TERM:
797                        {
798                                StringBuilder buf = new StringBuilder();
799                                while (true)
800                                {
801                                        char ch = (char)readByte();
802                                        if (ch == '\0')
803                                                break;
804                                        buf.append(ch);
805                                }
806                                return buf.toString();
807                        }
808                        case TEXTUALLY_ENCODED_LENGTH:
809                        {
810                                byte[] bytes = bytePool.borrowArray(policyValue);
811                                try
812                                {
813                                        if (in.read(bytes) < policyValue)
814                                                throw new IOException("Couldn't read " + policyValue + " bytes to determine string length. (pos " + getPos() + ")");
815                                        try
816                                        { 
817                                                len = Integer.parseInt(new String(bytes));
818                                        }
819                                        catch (NumberFormatException ex)
820                                        {
821                                                throw new IOException("Couldn't read string length value: " + ex + " (pos " + getPos() + ")");
822                                        }
823                                }
824                                finally
825                                {
826                                        bytePool.recycleArray(bytes);
827                                }
828                                break;
829                        }
830                        case FIXED_LENGTH:
831                                len = policyValue;
832                                break;
833                        case SHORT_ENCODED_LENGTH:
834                                len = this.readUnsignedShort();
835                                break;
836                        case BYTE_ENCODED_LENGTH:
837                                len = this.readUnsignedByte();
838                                break;
839                        default:
840                                throw new IllegalArgumentException("Invalid string policy " + policy + " at pos " + getPos());
841                }
842 
843                byte[] bytes = bytePool.borrowArray(len);
844                try
845                {
846                        this.readFully(bytes);
847                        if ((policy & NULL_TERM) != 0)
848                        {
849                                for (int i=0; i < len; ++i)
850                                {
851                                        if (bytes[i] == '\0')
852                                                return new String(bytes, 0, i);
853                                }
854                        }
855                        return new String(bytes);
856                }
857                finally
858                {
859                        bytePool.recycleArray(bytes);
860                }
861        }
862/**
863 * 
864 * @author Steven Sinclair
865 * @return long
866 */
867public long readUnsignedInt() throws IOException
868{
869        long ch1 = read();
870        long ch2 = read();
871        long ch3 = read();
872        long ch4 = read();
873        if ((ch1 | ch2 | ch3 | ch4) < 0)
874             throw new EOFException();
875        return ((ch1 << 24) + (ch2 << 16) + (ch3 << 8) + (ch4 << 0));
876}
