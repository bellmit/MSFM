424     * Processes a cancel for an order. Depending upon the order's internal state the cancel may be done
425     * immediately or postponed till the Auction expires.
426     *
427     * @param anOrder       order to be cancelled
428     * @param quantity      quantity to cancel out of the order
429     * @exception           SystemException
430     *
431     */
432    public boolean processCancel(Order order, int quantity, String userAssignedCancelId) throws SystemException, DataValidationException
433    {
434       boolean processFlag = false;
435 
436       if (Log.isDebugOn())
437       {
438           Log.debug("AuctionProcessor.processCancel is called orderId="
439                   + StructToString.toString(order.getOrderId()));
440       }
441       
442       // Instead of checking the status of the order, check the flashed quantity
443       
444       if ( order.getOddLotFlashedQuantity() == 0  )
445       {    
446           return processFlag;
447       }
448 
449       AuctionInternal auction = getAuctionService().findAuction(order.getProductKey().intValue());
450 
451       if ( !isAuctionGoingOn(auction, order.getProductKey()) )
452       {
453           return processFlag;
454       }
455 
456       Order auctionedOrder = auction.getAuctionedOrder();
457       
458       int remainingFlash = 0;
459       int auctionedQuantity = auctionedOrder.getOddLotFlashedQuantity();
460       
461       
462       // if the amount to cancel is odd, I should deal only with the flash quantity
463       if ( LotCalculator.isOddLot(quantity, getTradingClass(order.getProductClassKey()).getTradingPropertyLotSize()) )
464       {   
465           remainingFlash = auctionedQuantity - quantity;
466       }
467       else
468       {
469           // take out first the amount not flashed 
470           int availableQuantity = auctionedOrder.getQuantityAllowed() - quantity;
471           // if its negative, then it should affect the auction.
472           remainingFlash = auctionedQuantity + availableQuantity;
473       }
474       
475       if ( remainingFlash  >=  auctionedQuantity )
476       {
477           // nothing happens and the auction can continue
478           return processFlag;
479       }
480       
481       processFlag = true;
482       
483       if (  remainingFlash  >  0 )
484       { 
485           // update the odd lot flashed quantity
486           auctionedOrder.setOddLotFlashedQuantity(remainingFlash);
487       }
488       else
489       {   
490           // the remaining portion must be 0 or less, expire the auction
491           processAuctionExpire(auction, AuctionTerminateReasons.AUCTIONED_ORDER_CANCEL, null);
492       }
493       return processFlag;
494    }
