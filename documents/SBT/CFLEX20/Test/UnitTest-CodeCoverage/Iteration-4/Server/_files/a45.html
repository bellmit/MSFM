370    /**
371     * If p_short is true, adds an exception without the stack trace (as a single line).
372     * The "short version" includes the short exception type, the message, and the
373     * top of the stack.  
374     * If p_short is true and getCause() is non-null, then the cause will be included.
375     * <p>
376     * For example, logging the runtime thrown by <code>new MyClass().someMethod()</code> where 
377     * <blockquote><code>
378     *      package my.pkg;
379     *      public class MyClass { public void someMethod() 
380     *          { 
381     *              try { new ArrayList<String>().get(0); }
382     *              catch (Exception ex) { throw new RuntimeException("blah", ex); }
383     *          }
384     * </code></blockquote>
385     * would produce the output:
386     * <blockquote><code>
387     *      "RuntimeException:blah:MyClass.someMethod(5)[CAUSED BY ArrayOutOfBoundsException::]
388     * </code></blockquote>
389     * 
390     * @param p_throwable
391     * @param p_short
392     * @return
393     */
394    public MsgBuilder add(Throwable p_throwable, boolean p_short)
395    {
396        if (p_short)
397        {
398            addShortName(p_throwable.getClass()).add(':').add(p_throwable.getMessage());
399            StackTraceElement[] stack = p_throwable.getStackTrace();
400            if (stack.length > 0)
401            {
402                add(':').addShortName(stack[0].getClassName()).add('.');
403                add(stack[0].getMethodName()).add('(');
404                if (stack[0].isNativeMethod())
405                {
406                    add("native");
407                }
408                else if (stack[0].getLineNumber() < 0)
409                {
410                    add("no src");
411                }
412                else
413                {
414                    add(stack[0].getLineNumber());
415                }
416                add(')');
417            }
418            if (p_throwable.getCause() != null)
419            {
420                add("[CAUSED BY ").add(p_throwable.getCause(), p_short).add(']');
421            }
422            return this;
423        }
424        else
425        {
426            return add(p_throwable);
427        }
428    }

441    /**
442     * A recursive routing to log the format exceptions
443     * 
444     * @param p_linePrefix
445     * @param p_throwable
446     * @param p_details
447     * @return
448     */
449    private MsgBuilder formatException(String p_linePrefix, Throwable p_throwable, ExceptionDetails p_details)
450    {
451        // Generate the stack trace:
452        //
453        newLine();
454        add(p_linePrefix).add("==== BEGIN EXCEPTION ====").newLine();
455        add(p_linePrefix).add(p_throwable.getClass().getName(), p_throwable.getMessage()).newLine();
456        if (p_details != null)
457        {
458            add(p_details).newLine();
459        }
460        final StackTraceElement[] stackTrace =  p_throwable.getStackTrace(); 
461        for (int i = 0; i < stackTrace.length; i++)
462        {
463            add(p_linePrefix).add(stackTrace[i]).newLine();
464        }
465        
466        // Recurse into the originating exceptions:
467        //
468        Throwable causedBy = p_throwable.getCause();
469        if (causedBy != null)
470        {
471            newLine().add(p_linePrefix).add("CAUSED BY:").newLine();
472            formatException(p_linePrefix+"    ", causedBy, null);
473        }
474        newLine().add(p_linePrefix).add("===== END EXCEPTION =====").newLine();
475        return this;
476    }
