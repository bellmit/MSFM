718/**
719 * Make sure that this addition/modification won't violate any scheduling rules (overlapping trades of
720 * the same products).
721 *
722 * We don't need to validate againt custom sessions since a custom session will override any competing
723 * overlapping templates.
724 *
725 * We don't validate against inactive templates, but when we attempt to update a session to be active,
726 * we will validate it at that point to make sure that it doesn't produce a conflict.
727 *
728 * If the template being tested has a false isActive state, then we won't check for product class conflicts.
729 *
730 * @param template com.cboe.businessServices.tradingSessionService.sessionManagement.TradingSessionElementTemplate
731 * @exception com.cboe.exceptions.DataValidationException The exception description.
732 */
733public void validateTemplate(TradingSessionElementTemplate template) throws DataValidationException
734{
735        // If template inactive then just return.
736        if (!template.isActive())
737        {
738                return;
739        }
740        TemplateClass[] newClasses = template.getClasses();
741        Enumeration overlapping = getOverlappingTemplates(template.getProductOpenTime(), template.getProductCloseTime());
742        while (overlapping.hasMoreElements())
743        {
744                TradingSessionElementTemplate otherTemplate = (TradingSessionElementTemplate)overlapping.nextElement();
745                if (!otherTemplate.isActive() || (otherTemplate.getTemplateName().equals(template.getTemplateName())))
746                {
747                        continue; // if we can skip this template comparison, then move on to the next one.
748                }        
749                for (int i=0; i < newClasses.length; ++i)
750                {
751            try {
752                if (getProductClassHome().findByKey(newClasses[i].getClassKey()).getProductType() == ProductTypes.STRATEGY) {
753                    continue;  // allows strateyg to be traded in different sections at the same time.
754                }                
755                otherTemplate.findClass(newClasses[i].getClassKey());
756                                throw ExceptionBuilder.dataValidationException("Invalid time range for template \"" + template.getTemplateName() + "\": class key " + newClasses[i].getClassKey() + " already being traded in this time period on template " + otherTemplate.getTemplateName() + ".", DataValidationCodes.INVALID_TIME);
757            }
758            catch (NotFoundException e) {
759                // class not found in overlapping template - its a good thing
760                        }
761                }
762        }
763}
