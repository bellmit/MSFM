305    /**
306     * Applies a supplemental hash function to a given hashCode, which defends
307     * against poor quality hash functions. This is critical because HashMap
308     * uses power-of-two length hash tables, that otherwise encounter collisions
309     * for hashCodes that do not differ in lower bits. Note: Null keys always
310     * map to hash 0, thus index 0.
311     */
312    static int hash(int h)
313    {
314        // This function ensures that hashCodes that differ only by
315        // constant multiples at each bit position have a bounded
316        // number of collisions (approximately 8 at default load factor).
317        h ^= (h >>> 20) ^ (h >>> 12);
318        return h ^ (h >>> 7) ^ (h >>> 4);
319    }

467     * capacity. This method is called automatically when the number of keys in
468     * this map reaches its threshold.

471     * map, but sets threshold to Integer.MAX_VALUE. This has the effect of

532     * currently in the specified map.
533     * 
534     * @param m
535     *            mappings to be stored in this map
536     * @throws NullPointerException
537     *             if the specified map is null
538     */
539    public void putAll(IntMap<? extends V> m)
540    {
541        int numKeysToBeAdded = m.size();
542        if (numKeysToBeAdded == 0)
543            return;
544 
545        /*
546         * Expand the map if the map if the number of mappings to be added is
547         * greater than or equal to threshold. This is conservative; the obvious
548         * condition is (m.size() + size) >= threshold, but this condition could
549         * result in a map with twice the appropriate capacity, if the keys to
550         * be added overlap with the keys already in this map. By using the
551         * conservative calculation, we subject ourself to at most one extra
552         * resize.
553         */
554        if (numKeysToBeAdded > threshold)
555        {
556            int targetCapacity = (int) (numKeysToBeAdded / loadFactor + 1);
557            if (targetCapacity > MAXIMUM_CAPACITY)
558                targetCapacity = MAXIMUM_CAPACITY;
559            int newCapacity = table.length;
560            while (newCapacity < targetCapacity)
561                newCapacity <<= 1;
562            if (newCapacity > table.length)
563                resize(newCapacity);
564        }

566        for (Iterator<? extends IntMapEntry<? extends V>> i = m.entrySet().iterator(); i.hasNext();)
567        {
568            IntMapEntry<? extends V> e = i.next();
569            put(e.getKey(), e.getValue());
570        }
571    }
