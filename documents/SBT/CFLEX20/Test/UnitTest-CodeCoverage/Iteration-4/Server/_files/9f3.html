415    public void validateCrossOrder(OrderHandlingStruct order1, OrderHandlingStruct order2) throws DataValidationException, AuthorizationException, CommunicationException, SystemException
416    {
417        // product key match
418        if (order1.productKey != order2.productKey)
419        {
420            String msg = "cross order productKey does not match: " + order1.productKey +" != " + order2.productKey;
421            if (Log.isDebugOn())
422            {
423                Log.debug(this, msg);
424            }
425            throw ExceptionBuilder.dataValidationException(msg, DataValidationCodes.INVALID_PRODUCT);
426        }
427 
428        // price match
429        Price price1 = PriceFactory.create(order1.price);
430        Price price2 = PriceFactory.create(order2.price);
431        if (!price1.equals(price2))
432        {
433            String msg = "cross order price does not match: " + price1 + " != " + price2;
434            if (Log.isDebugOn())
435            {
436                Log.debug(this, msg);
437            }
438            throw ExceptionBuilder.dataValidationException(msg, DataValidationCodes.INVALID_PRICE);
439        }
440 
441        // contingency match
442        short contingency1 = order1.contingency.type;
443        short contingency2 = order2.contingency.type;
444 
445        if (((contingency1 != ContingencyTypes.AUTOLINK_CROSS) || (contingency2 != ContingencyTypes.AUTOLINK_CROSS_MATCH)) &&
446            ((contingency1 != ContingencyTypes.AUTOLINK_CROSS_MATCH) || (contingency2 != ContingencyTypes.AUTOLINK_CROSS)) &&
447            (!(contingency1 == ContingencyTypes.MIDPOINT_CROSS || contingency1 == ContingencyTypes.CROSS || 
448               contingency1 == ContingencyTypes.TIED_CROSS || contingency1 == ContingencyTypes.CROSS_WITHIN ||
449               contingency1 == ContingencyTypes.TIED_CROSS_WITHIN ||
450               contingency1 == ContingencyTypes.BID_PEG_CROSS ||
451               contingency1 == ContingencyTypes.OFFER_PEG_CROSS ||
452               contingency1 == ContingencyTypes.CASH_CROSS ||
453               contingency1 == ContingencyTypes.NEXT_DAY_CROSS ||
454               contingency1 == ContingencyTypes.TWO_DAY_CROSS ||
455               contingency1 == ContingencyTypes.TIED_CROSS_SWEEP) || (contingency1 != contingency2)))
456        {
457            String msg = "cross order contingency type is not valid: " + contingency1 + "/" + contingency2;
458            if (Log.isDebugOn())
459            {
460                Log.debug(this, msg);
461            }
462            throw ExceptionBuilder.dataValidationException(msg, DataValidationCodes.INVALID_CONTINGENCY_TYPE);
463        }
464 
465        // quantity match
466        if (order1.originalQuantity != order2.originalQuantity)
467        {
468            String msg = "cross order quantity does not match: " + order1.originalQuantity + " != " + order2.originalQuantity;
469            if (Log.isDebugOn())
470            {
471                Log.debug(this, msg);
472            }
473            throw ExceptionBuilder.dataValidationException(msg, DataValidationCodes.INVALID_QUANTITY);
474        }
475 
476        // odd/mixed lot size not allowed for autolink cross
477        if (contingency1 == ContingencyTypes.AUTOLINK_CROSS || contingency1 == ContingencyTypes.AUTOLINK_CROSS_MATCH)
478        {
479            validateOrderLotSize(order1);
480        }
481 
482        // cross order sides must match
483        if (((order1.side == Sides.BUY) && (order2.side != Sides.SELL &&
484                                            order2.side != Sides.SELL_SHORT &&
485                                            order2.side != Sides.SELL_SHORT_EXEMPT)) ||
486            ((order1.side == Sides.SELL ||
487              order1.side == Sides.SELL_SHORT ||
488              order1.side == Sides.SELL_SHORT_EXEMPT) && (order2.side != Sides.BUY)))
489        {
490            String msg = "cross order side does not match: " + order1.side + "/" + order2.side;
491            if (Log.isDebugOn())
492            {
493                Log.debug(this, msg);
494            }
495            throw ExceptionBuilder.dataValidationException(msg, DataValidationCodes.INVALID_SIDE);
496        }
497        
498        //Validate contingencyPrice are equal for PEG_CROSS
499        validateCrossOrderContingencyPrice(order1, order2);
500    }
