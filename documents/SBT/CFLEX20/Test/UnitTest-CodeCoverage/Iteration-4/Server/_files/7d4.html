431     * as
432     * 
433     * <pre>
434     *       # (listOfHomes is optional, if order of homes is important)
435     *       # (listOfHomes does not need to be complete)
436     *       listOfHomes = {homesName1},{homeName2},...{homeNameN}
437     *       home.{homeName1} = {homeClassName1}

442     *       {homeName1}.{prop2} = {value2}
443     *       #...
444     * </pre>
445     * 
446     * @param homeProperties
447     */
448    public void addHomes(Properties homeProperties)
449    {
450        List listOfHomes = Arrays.asList(homeProperties.getProperty("listOfHomes", "").split(
451                "\\w+,\\w+"));
452        Properties homeImpls = new Properties();
453        Iterator propIter = homeProperties.keySet().iterator();
454        while (propIter.hasNext())
455        {
456            String key = (String) propIter.next();
457            if (key.startsWith("home."))
458            {
459                String homeName = key.substring("home.".length());
460                try
461                {
462                    Class implClass = Class.forName(homeProperties.getProperty(key));
463                    homeImpls.put(homeName, implClass);
464                }
465                catch (ClassNotFoundException ex)
466                {
467                    throw new RuntimeException("Impl for home " + homeName + ": " + ex);
468                }
469            }
470        }
471 
472        // for each home in listOfHomes:
473        for (int i = 0; i < listOfHomes.size(); i++)
474        {
475            String homeName = (String) listOfHomes.get(i);
476            if (homeName.length() == 0)
477            {
478                continue;
479            }
480            if (!homeImpls.containsKey(homeName))
481            {
482                throw new RuntimeException("home." + homeName + " property not found "
483                        + "for home in listOfHomes");
484            }
485            addHome(homeName, (Class) homeImpls.get(homeName), homeProperties, homeName + ".", true);
486        }
487 
488        // for each home NOT in listOfHomes:
489        Iterator nameIter = homeImpls.keySet().iterator();
490        while (nameIter.hasNext())
491        {
492            String homeName = (String) nameIter.next();
493            if (!listOfHomes.contains(homeName))
494            {
495                addHome(homeName, (Class) homeImpls.get(homeName), homeProperties, homeName + ".",
496                        true);
497            }
498        }
