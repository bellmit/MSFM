587    private boolean canTradeAtTSB(DerivedQuote currentDQ,AuctionInternal auction,OrderBook spreadOrderBook,Price auctionSideBSM,Tradable incomingTradable)
588    {
589        Order auctionedOrder = auction.getAuctionedOrder();
590        Order firmMatchOrder = auction.getFirmMatchOrder();
591        Side auctionSide =auctionedOrder.getSide();
592        Side responseSide = auctionSide.getOtherSide();
593        Price auctionedSideDSM = null;
594        if (auctionSide.isAsDefinedSide() || auctionSide.isBuySide())
595        {
596            auctionedSideDSM = currentDQ.getBid().getPrice();
597        }
598        else
599        {
600            auctionedSideDSM = currentDQ.getAsk().getPrice();
601        }
602 
603        if (!auctionedOrder.treatedLikeCustomer())
604        {
605            return false;
606        }
607 
608        Price bestLimitPrice = spreadOrderBook.getBestLimitPrice(auctionSide);
609        if (!bestLimitPrice.isValuedPrice() || auctionSide.isFirstBetter(auctionedSideDSM, bestLimitPrice))
610        {
611            return false;
612        }
613 
614        if (!auctionSide.isFirstBetter(auctionSideBSM,auctionedSideDSM))
615        {
616            return false;
617        }
618 
619        /*
620         * CASE 2.58
621         * If there is a non customer order TSB and If there is incoming quote or quote like order on the opposite side of the auctioned order
622         * and is marketable against the TSB and auctioned order is customer order then trade at the TSB is allowed.
623         *
624         *  - Customer auctioned order
625         *  - NON CUSTOMER TSB on the same side of auction
626         *  - Auctioned order price is equal or better than non customer TSB.
627         *  - Incoming quote or quote like order is marketable
628         *     ALLOWED TO TRADE AT THE NON CUSTOMER TSB
629         */
630        if (incomingTradable !=null && incomingTradable.treatedLikeQuote())
631        {
632            if (incomingTradable.getSide().isSameSide(responseSide))
633            {
634                Price incomingPrice = incomingTradable.getPrice();
635                if (incomingPrice.isValuedPrice() &&
636                    auctionSide.isFirstBetterOrEqual(auctionedOrder.getPrice(), auctionSideBSM) &&
637                    !hasCustomerTSB(auctionSide,spreadOrderBook,auctionSideBSM) &&
638                    auctionSide.areTwoPricesTradable(auctionSideBSM,incomingPrice))
639                {
640                    return true;
641                }
642            }
643        }
644 
645        /* CASE 2.20
646         * If There No Customer order at the TSB on the same side of auctioned order and auctioned order is customer order price is equal or better
647         * than TSB and firm match order lock or cross the TSB then trade at the TSB is allowed
648         * -Customer auctioned order
649         * -Non customer TSB
650         * -Auctioned order price is equal or better non customer TSB
651         * -Firm match order is tradable
652         */
653        Price firmMatchPrice = firmMatchOrder.getPrice();
654        boolean result=false;
655        if (firmMatchPrice.isValuedPrice() && auctionSide.areTwoPricesTradable(auctionSideBSM,firmMatchPrice))
656        {
657            if (auctionSide.isFirstBetterOrEqual(auctionedOrder.getPrice(), auctionSideBSM))
658            {
659                result = !hasCustomerTSB(auctionSide,spreadOrderBook,auctionSideBSM);
660            }
661        }
662 
663        return result;
664    }
