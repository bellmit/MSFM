823                     }
824                }

829                    // Set the updateInProgress to false 
830                    updateInProgress = false;
831                    return message;
832                }
833            } catch ( CommunicationException ce ) {
834                Log.exception( "CommunicationException: ",ce );
835                message = "FAILURE : CommunicationException, for detail please see the GlobalServer.log";
