360                ? "ok."
361                : "badly.  Stack trace to follow.").append('\n');
362        
363        double elapsedSecs = this.elapsedMillis / 1000.0D;
364        report.append("Actual rate was ").append(fmt(this.numExecutions / elapsedSecs));
365        report.append("/sec for ").append(fmt(elapsedSecs)).append(" seconds ");
366        if (this.elapsedMillis > 6 * 60*60000 /* (6hr) */)
367        {
368            report.append(" (").append(fmt(elapsedSecs / 3600)).append(" hours)");
369        }
370        else if (this.elapsedMillis > 2 * 60000/* 2min */)
371        {
372            report.append(" (").append(fmt(elapsedSecs / 60)).append(" minutes)");
373        }
374        double overallRate = this.numExecutions / elapsedSecs; 
375        report.append(" (").append(fmtSigned(100*(overallRate-ratePerSec)/ratePerSec));
376        report.append("% of target ").append(ratePerSec).append("/sec)\n");
377 
378        report.append("  Total calls    = ").append(this.numExecutions).append('\n');
379        report.append("  Total errors   = ").append(this.numErrors).append('\n');
380        report.append("  Total failures = ").append(this.numFailures).append('\n');
381    }
