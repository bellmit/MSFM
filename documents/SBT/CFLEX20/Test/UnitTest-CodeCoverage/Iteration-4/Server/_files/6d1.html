550        private void startPingThread()
551        {
552                final ControlMessageBuffer pingBuffer = ControlMessageBuffer.getInstance();
553                final ControlMessagePingCodec pingCodec = new ControlMessagePingCodec();
554                final DataBufferBlock block = new DataBufferBlock(64);
555                final int numChannels = mktBuffPubHome.getNumChannels();
556                final int numHistoryChannels = mktHistoryBuffPubHome.getNumChannels();
557 
558                pingTimer.scheduleAtFixedRate(new TimerTask() {
559                        @Override
560                        public void run()
561                        {
562                                boolean ok=false;
563                                if (log().isDebugOn())
564                                {
565                                        log().debug(msg().add("Sending PING to ").add(numChannels).add(" MarketBuffer channels..."));
566                                        log().debug(msg().add("Sending PING to ").add(numHistoryChannels).add(" MarketHistoryBuffer channels..."));
567                                }
568                                try
569                                {
570                                        pingCodec.setTimestamp();
571                                        pingCodec.setServerInstanceId(getInstanceId());
572                                        pingCodec.setStatus(getSessionStatus());
573                                        pingBuffer.write(pingCodec);
574                                        block.write(pingBuffer);
575                                        for (int i = 0; i < numChannels; i++)
576                                        {
577                                                mktBuffPubHome.publishDataToChannel(block, i, i);
578                                        }
579                                        for (int i = 0; i < numHistoryChannels; i++)
580                    {
581                                            mktHistoryBuffPubHome.publishDataToChannel(block, i, i);
582                    }
583                                        block.reset();
584                                        pingBuffer.reset();
585                                        pingCodec.reset();
586                                        ok=true;
587                                }
588                                catch (final Exception e)

592                                finally
593                                {
594                                        if (log().isDebugOn())
