370    private boolean updateOrderEvent(final Order p_order, final OrderSyncData p_oldData, final OrderSyncData p_newData,
371            final OrderCacheFilterActionType p_cacheAction) 
372        throws TransactionFailedException, DataValidationException, NotFoundException, NotAcceptedException, SystemException 
373    {
374        Long orderId = p_newData.getOrderId();
375        String historyResultString = "SUCCESS";
376        boolean removeOrder = false;
377        try {
378            logOrderSyncEvent(OrderSyncEventHistory.UPDATE_EVENT, p_cacheAction, p_newData);
379            if (p_cacheAction == OrderCacheFilterActionType.REMOVE_ORDER) {
380                removeOrder = true;
381                removeOrderFromAllHolders(orderId);
382                if ((p_order != null) && (p_order.getBookedQuantity() > 0)) {
383                    getBrokerService().acceptRemoveFromOrderBookInSlave(p_order, p_order.getBookedQuantity());
384                }
385            }
386            else if (p_cacheAction == OrderCacheFilterActionType.CANCEL_ORDER_HOLDER) {
387                addToHolder(SyncOrderHolderType.CANCEL, p_order);
388            }
389            else if (p_cacheAction == OrderCacheFilterActionType.P_PA_ORDER_HOLDER) {
390                addToHolder(SyncOrderHolderType.P_PA, p_order);
391            }
392            else if (p_cacheAction == OrderCacheFilterActionType.SATISFACTION_ORDER_HOLDER) {
393                addToHolder(SyncOrderHolderType.SATISFACTION, p_order);
394            }
395            else if (p_cacheAction == OrderCacheFilterActionType.ADD_TO_RETURN_TO_PAR_ORDER_HOLDER) {
396                addToHolder(SyncOrderHolderType.RETURN_TO_PAR, p_order);
397            }
398            else if (p_cacheAction == OrderCacheFilterActionType.REMOVE_FROM_RETURN_TO_PAR_ORDER_HOLDER) {
399                removeFromHolder(SyncOrderHolderType.RETURN_TO_PAR, p_order.getUniqueId());
400            }
401            else if (p_cacheAction == OrderCacheFilterActionType.ADD_TO_MOC_ORDER_HOLDER)  {
402                addToHolder(SyncOrderHolderType.MOC, p_order);
403            }
404            else if (p_cacheAction == OrderCacheFilterActionType.REMOVE_FROM_MOC_ORDER_HOLDER)  {
405                removeFromHolder(SyncOrderHolderType.MOC, p_order.getUniqueId());
406                updateOrderEvent(p_order, p_oldData, p_newData, OrderCacheFilterActionType.MKT_ORDER_HOLDER);
407            }
408            else if (p_cacheAction == OrderCacheFilterActionType.ADD_TO_LINKED_AWAY_ORDER) {
409                addToHolder(SyncOrderHolderType.LINKED_AWAY, p_order);
410                if((p_order != null) && (p_order.getPrice().isMarketPrice()))
411                {
412                    removeFromHolder(SyncOrderHolderType.MKT, p_order.getUniqueId());
413                }
414                if((p_order != null) && (TEOrderCacheFilterer.filterAIMwithSweepOrder(p_order)))
415                {
416                    removeFromHolder(SyncOrderHolderType.CANCEL, p_order.getUniqueId());
417                }
418                if ((p_order != null) && (p_order.getBookedQuantity() > 0)) {
419                    getBrokerService().acceptRemoveFromOrderBookInSlave(p_order, p_order.getBookedQuantity());
420                }
421            }
422            else if (p_cacheAction == OrderCacheFilterActionType.REMOVE_FROM_LINKED_AWAY_ORDER) {
423                removeFromHolder(SyncOrderHolderType.LINKED_AWAY, p_order.getUniqueId());
424                if(p_order.getPrice().isMarketPrice())
425                {
426                    processMarketOrder(p_order, p_oldData, p_newData);
427                }
428                else
429                {
430                    if(TEOrderCacheFilterer.filterAIMwithSweepOrder(p_order))
431                    {
432                        if(p_order.getRemainingQuantity() > 0)
433                            addToHolder(SyncOrderHolderType.CANCEL, p_order);
434                    }
435                    else
436                    {
437                        if(!TEOrderCacheFilterer.filterSweepNReturnOrder(p_order) &&
438                                !TEOrderCacheFilterer.filterFishNReturnOrder(p_order))
439                        {
440                            updateOrderEvent(p_order, p_oldData, p_newData, OrderCacheFilterActionType.UPDATE_BOOK);
441                        }
442                    }
443                }
444            }
445            else if (p_cacheAction == OrderCacheFilterActionType.ADD_TO_STP_ORDER_HOLDER)      {
446                addToHolder(SyncOrderHolderType.STP, p_order);
447            }
448            else if (p_cacheAction == OrderCacheFilterActionType.REMOVE_FROM_STP_ORDER_HOLDER)      {
449                removeFromHolder(SyncOrderHolderType.STP, p_order.getUniqueId());
450                if (p_newData.getOrderHandlingStruct().price.type == PriceTypes.MARKET)  {    
451                    updateOrderEvent(p_order, p_oldData, p_newData, OrderCacheFilterActionType.MKT_ORDER_HOLDER);  
452                }
453                else {
454                    updateOrderEvent(p_order, p_oldData, p_newData, OrderCacheFilterActionType.UPDATE_BOOK);
455                }
456            }
457            else if (p_cacheAction == OrderCacheFilterActionType.MKT_ORDER_HOLDER) {
458                processMarketOrder(p_order, p_oldData, p_newData);
459            }
460            else if (p_cacheAction == OrderCacheFilterActionType.BOOKABLE_NOT_YET_BOOKED) {
461                addToHolder(SyncOrderHolderType.BOOKABLE_NOT_YET_BOOKED, p_order);
462            }
463            else if (p_cacheAction == OrderCacheFilterActionType.UPDATE_BOOK) {
464                processUpdateBookCacheAction(p_order, p_newData);
465            }
466            else {
467                historyResultString = "FAILED";
468            }
469        }
470        finally {
471            recordOrderSyncEventHistory(OrderSyncEventHistory.UPDATE_EVENT, p_cacheAction, p_newData, historyResultString);        
472        }
473        return removeOrder;
474    }

495    private void processUpdateBookCacheAction(final Order p_order, final OrderSyncData p_newData)
496        throws DataValidationException, NotAcceptedException, SystemException
497    {
498        int newBookableQty = getRemainingQuantity(p_newData);
499        int currentBookedQty = 0;
500        if (p_order != null) {
501            currentBookedQty = p_order.getBookedQuantity();
502        }
503        if ((newBookableQty > 0) && (currentBookedQty == 0)) {
504            logOrderSyncEvent(OrderSyncEventHistory.UPDATE_EVENT, OrderCacheFilterActionType.UPDATE_BOOK, p_newData, " ADDING TO BOOK");
505            getBrokerService().acceptAddToOrderBookInSlave(p_order, newBookableQty);
506            removeFromHolder(SyncOrderHolderType.BOOKABLE_NOT_YET_BOOKED, p_order.getUniqueId());
507            addToHolder(SyncOrderHolderType.BOOKED, p_order);
508        }
509        else if ((newBookableQty - currentBookedQty) != 0) {
510            logOrderSyncEvent(OrderSyncEventHistory.UPDATE_EVENT, OrderCacheFilterActionType.UPDATE_BOOK, p_newData, " UPDATING BOOK");
511            getBrokerService().acceptUpdateOrderBookInSlave(p_order, newBookableQty - currentBookedQty);
512        }
513        else {
514            logOrderSyncEvent(OrderSyncEventHistory.UPDATE_EVENT, OrderCacheFilterActionType.UPDATE_BOOK, p_newData, " NO CHANGE IN BOOKABLE QTY");
515        }
516    }

692    protected boolean checkOHSAndTEMaster(String message)
693    {
694        boolean connected = false;
695        int retryCounter = 0;
696        boolean ready = false;
697        boolean nowOHSReady = false;
698        while (!ready && retryCounter < MAX_RETRY) {
699            if ((retryCounter > 10) && ((retryCounter % 10) == 0)) {
700                Log.alarm("OrderSyncEventProcessor: "+message+" on OHS failed after " + retryCounter + " attempts.");
701            }
702            try {
703                boolean nowTEReady = FoundationFramework.getInstance().isMaster();
704                if(nowTEReady)
705                {
706                    short pingResult = getOrderMaintenanceServiceHome().find().ping();
707                    nowOHSReady = (pingResult == SUCCESS ? true : false);
708                }
709                ready = (nowOHSReady && nowTEReady);
710            }
711            catch (CommunicationException commException)  {
712                // retry the communication exception
713                // sleep for an allocated time and retry this
714                retryCounter ++;
715                Log.information("OrderSyncEventProcessor: Communication exception between OHS and TE. Will try to "+message+" orders again. " +
716                                " Retry attempt count = " + retryCounter);
717                Log.exception(commException);
718                sleepBeforeRetryOnOHS();
719            }
720            catch (org.omg.CORBA.SystemException e)   {
721                // retry the object not exist exception
722                // this could be due to OHS server is probably still coming up.
723                retryCounter++;
724                Log.information("OrderSyncEventProcessor: OHSServer Not up or CORBA Exception: Will try to "+message+" orders again. " +
725                        " Retry attempt count = " + retryCounter);
726                Log.exception(e);
727                sleepBeforeRetryOnOHS();
728            }
729            catch (Exception e) {
730                // ignore and carry on
731                Log.information("OrderSyncEventProcessor: "+message+" orders Failed on TE call From OHS Exception: " + e);
732            }
733        }
