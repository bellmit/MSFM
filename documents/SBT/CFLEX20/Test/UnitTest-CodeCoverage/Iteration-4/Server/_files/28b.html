434                }
435                else
436                {
437                    if (UserInputMonitorImpl.this.isNeverCancel())
438                    {
439                        Log.information(UserInputMonitorImpl.this,

457                        // a check made in InternalActivityReasons,

459                        // one of the disqualifying reasons for report publishing.
460                        //

462                                ActivityTypes.SYSTEM_CANCEL_QUOTE,
463                                ActivityReasonsOperations.NO_USER_ACTIVITY, true);
464                    }
465                    catch (final Exception e)
466                    {
467                        Log.alarm(UserInputMonitorImpl.this, "Failed to cancel quotes: " + e);
468                        return;

472                        // cancel I-orders with reason and *with* reports

474                                UserInputMonitorImpl.this.getPublishedCancelReason());
475                    }
476                    catch (final Exception e)

483                            UserInputMonitorImpl.this.routeKeys, p_userId,
484                            UserInputMonitorImpl.this.session, allClassKeys(),
485                            UserInputMonitorImpl.this.getPublishedCancelReason(), 
486                                "UIM cancel at " + new Timestamp(now) + " for " + p_userId 
487                                + ": no quote activity since " + p_time);
488                    Log.information(UserInputMonitorImpl.this, "Completed cancelling quotes for "
489                            + p_userId);
490                    
491                    Thread.currentThread().yield(); // attempt some throttling of a 'burst' of cancels.
492                }
493            }
494        };
495 
496        try
497        {
498            this.cancelExecutorPool.execute(cancelJob);
499        }
500        catch (RejectedExecutionException e)
501        {
502            // This should not happen since the executor is using an unbounded queue.
503            Log.exception(UserInputMonitorImpl.this, "Failed to cancel quotes for "+p_userId, e);
504        }
505    }
