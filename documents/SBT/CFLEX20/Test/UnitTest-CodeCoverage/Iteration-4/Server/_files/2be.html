471                }
472                processDataMessage(message, !preconfirmData, out);
473                break;
474            case TPFMessageTypes.DATA:
475                processDataMessage(message, false, out);
476                break;
477            case TPFMessageTypes.DATA_REJECT:
478                logAlarm("Received DATA_REJECT: dumping reject message");
479                message.dumpMessage(System.out);
480                break;
481            case TPFMessageTypes.CONFIRM_RESPONSE:
482                logInfo("Received CONFIRM_RESPONSE");
483            case TPFMessageTypes.HEARTBEAT_REQUEST:
484                logInfo("Received HEARTBEAT_REQUEST: sending HEARTBEAT_RESPONSE");
485                response = responseMessage(TPFMessageTypes.HEARTBEAT_RESPONSE);
486                copyHeaderData(message, response);
487                break;
488            case TPFMessageTypes.HEARTBEAT_RESPONSE:
489                logAlarm("Received HEARTBEAT_RESPONSE: that's strange, since I never send any HB requests: message ignored.");
490                break;
491            default:
492                throw new TPFMessageException("Read unknown message type " + message.getMsgType());
493        }
494        if (response != null)
495        {
496            write(response);
497        }
498        if (stopClient)
499        {
500            stopClient();
501        }
502    }
