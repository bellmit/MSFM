491/**
492 * Update the values for a session.  Note that if a session is being transferred from a closed state to an open one,
493 * then if the stored session is transient it will be automatically promoted to a persistent storage, since we need
494 * to keep track of all open sessions.
495 *
496 * @param TradingSessionElementStruct
497 */
498public synchronized void updateElement(TradingSessionElementStruct struct) throws DataValidationException, NotFoundException
499{
500        // Make sure the session exists.
501        TradingSessionElementImpl oldSession = getElementForKey(struct.elementKey);
502        if (oldSession == null)
503        {
504                throw ExceptionBuilder.notFoundException("Could not find session to update: Session key = " + struct.elementKey, 0);
505        }
506 
507        // Not allowed to change session state.
508        if (struct.elementState != oldSession.getElementState())
509        {
510                throw ExceptionBuilder.dataValidationException("Illegal state change operation, not allowed: Session key = " + struct.elementKey, DataValidationCodes.INVALID_UPDATE_ATTEMPT);
511        }
512 
513        // Not allowed to change session key.
514        if (struct.elementKey != oldSession.getElementKey())
515        {
516                throw ExceptionBuilder.dataValidationException("Illegal update operation, not allowed to change session keys: Session key old/new = " + oldSession.getElementKey() + "/" + struct.elementKey, DataValidationCodes.INVALID_UPDATE_ATTEMPT);
517        }
518 
519        // Update the session
520        oldSession.fromStruct(struct);
521 
522        // Now validate it. If transaction rolls back, the update above will not
523        // take place.
524        validateSession(oldSession);
525}
