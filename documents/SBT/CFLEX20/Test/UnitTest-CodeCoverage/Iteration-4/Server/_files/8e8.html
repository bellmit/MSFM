742    private PriceTradableCheck getTradePrice(  PriceRange priceRange, Price orderPrice, Price salNBBOPrice,
743            OrderBook orderBook, Side orderSide)
744        {
745                boolean isTradable = false;
746                Price tradePrice = NO_PRICE;
747                Price startingPrice = priceRange.getStartingPrice();
748                Price endingPrice = priceRange.getEndingPrice();
749                PriceTradableCheck result = new PriceTradableCheck();
750                
751                if (orderSide.isBuySide())
752                {
753                        tradePrice = selectWorstPrice(endingPrice, orderPrice, orderSide );
754                        isTradable = canTradeAtPrice(priceRange, tradePrice, salNBBOPrice, orderBook, orderSide);
755                        if(!isTradable)  //try the other end of NBBO price
756                        {
757                                tradePrice = selectWorstPrice(startingPrice, orderPrice, orderSide );
758                                isTradable = canTradeAtPrice(priceRange, tradePrice, salNBBOPrice, orderBook, orderSide);
759                        }
760                }
761                else
762                {
763                        tradePrice = selectWorstPrice(startingPrice, orderPrice, orderSide );
764                        isTradable = canTradeAtPrice(priceRange, tradePrice, salNBBOPrice, orderBook, orderSide);
765                        if(!isTradable)  //try the other end of NBBO price
766                        {
767                                tradePrice = selectWorstPrice(endingPrice, orderPrice, orderSide );
768                                isTradable = canTradeAtPrice(priceRange, tradePrice, salNBBOPrice, orderBook, orderSide);
769                        }
770                }
771                
772                result.isTradable = isTradable;
773                result.tradePrice = tradePrice;
774                
775                return result;
776        }
