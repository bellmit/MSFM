868    private static final int     MS_PER_HOUR         = 60 * 60 * 1000;
869    private static final int     MS_PER_MINUTE       = 60 * 1000;
870    private static final int     MS_PER_SECOND       = 1000;
871    private static final int     MS_PER_100TH_SEC    = 10;

874    private static final TimeStruct getTimeStruct(long p_timeInMillis) {
875 
876        /*
877         * The OLD CBOE Direct DateWrapper.convertToTime(long aTime) code was 
878         * pretty spendy. It created 3 extra objects to get a TimeStruct, and 
879         * uses the Java Calendar to figure out the hour, minute and second 
880         * from the timeInMillis. The Calendar is VERY expensive since it does 
881         * lots of modulus and division to figure out leap years from 1970, and 
882         * to figure out year, month, day, day of week, hour, minute, second 
883         * and lots of other stuff we don't need.
884         * 
885         * We will do an optimization which exploits how this method is most used.
886         * 99%+ of all calls to this method will be with timestamps from TODAY.
887         * This means we can do a "clever cheat" which avoids the Calendar.
888         * 
889         * We'll first compute millisSinceMidnight = timeInMillis - todaysMidnightMillis.
890         * If this is >= 0 and < MILLIS_IN_A_DAY, this means millisSinceMidnight
891         * is a relative time since midnight TODAY, and therefore timeInMillis is
892         * a timestamp from TODAY. To compute hr, min, sec and fraction is then
893         * simple; we use simple division and modulus to obtain the values.
894         * If the millisSinceMidnight is < 0 or > MILLIS_IN_A_DAY, this means
895         * it is not from today, so we'll use the old code that uses the Calendar
896         * to get the TimeStruct instead. 
897         * 
898         * It is expected this method won't be called much with times outside 
899         * of today, however, so we'll do a MUCH faster code path for the most likely
900         * case. 
901         */
902        
903        TimeStruct   ts;
904        long         millisSinceMidnight;
905        int          millis, hr, min, sec, frac;
906        
907        millisSinceMidnight = p_timeInMillis - baselineMillis;
908        
909        if (millisSinceMidnight >= 0 && millisSinceMidnight < MS_PER_DAY) {
910            
911            // Use 4 byte integer math; should run faster on 32 bit CPUs
912            millis   = (int)millisSinceMidnight;   
913 
914            // Divide millis by the appropriate divisor to get each value, 
915            // then modulus to remove that value from millis to get the next 
916            // value, etc.  Use subtract and multiply to do modulus; this should 
917            // run faster than "%" modulus which is a form of division.
918            hr       = millis / MS_PER_HOUR;
919            millis   = millis - (hr * MS_PER_HOUR);
920 
921            min      = millis / MS_PER_MINUTE;
922            millis   = millis - (min * MS_PER_MINUTE);
923            
924            sec      = millis / MS_PER_SECOND;
925            millis   = millis - (sec * MS_PER_SECOND);
926            
927            // NOTE: CboeDir DateWrapper TRUNCATES the millis to 100ths by using division!!!
928            //       We do the same so we have identical behavior.
929            frac     = millis / MS_PER_100TH_SEC;    
930            
931            ts = new TimeStruct((byte)hr,(byte)min,(byte)sec,(byte)frac);
932        }
933        else { 
934            // Revert to old way of doing things.
935            if (p_timeInMillis != 0) {
936                DateWrapper wrap = new DateWrapper(p_timeInMillis);
937                ts = wrap.toTimeStruct();
938            }
939            else {
940                ts = StructBuilder.buildTimeStruct();
941            }
942        }
943        return ts;
944    }
