373                        Class type = fields[i].getType();
374                        if (type.isArray())
375                        {
376                                fields[i].set(obj, Array.newInstance(type.getComponentType(), 0));
377                        }
378                        else if (char.class == type)
379                        {
380                                fields[i].setChar(obj, '\0');
381                        }
382                        else if (!type.isPrimitive())
383                        {
384                                fields[i].set(obj, String.class==type ? "" : newStruct(type));
385                        }
386                }
387                return obj;
388        }
389        catch (IllegalAccessException ex)
390        {
391                throw new IllegalArgumentException("Class " + structClass.getName() + " doesn't seem to be an IDL-generated struct.");
392        }
393        catch (InstantiationException ex)
394        {
395                throw new IllegalArgumentException("Class " + structClass.getName() + " doesn't seem to be an IDL-generated struct.");
396        }
397}
398/**
399 * Calls <code>printStruct(Object, String, Writer)</code>, using a new
400 * <code>PrintWriter</code> tied to <code>System.out</code>.  If there's
401 * an IOException generated then it'll be rethrown as an Error.
402 * <p>
403 * Prints out a description of the fields of struct, line-by-line, recursing
404 * non-primitive types, prefixing each line of output with the prefix string.
405 *
406 * @see #printStruct(Object, String, Writer)
407 * @author Steven Sinclair
408 * @param struct java.lang.Object
409 * @param prefix java.lang.String
410 */
411public static void printStruct(Object struct, String prefix)
412{
413        try
414        {
415                PrintWriter writer = new PrintWriter(System.out);
416                printStruct(struct, prefix, writer);
417                writer.flush();
418        }
419        catch (IOException ex)
420        {
421                throw new Error("Unexpected IO error using System.out: " + ex);
422        }
423}
424/**
425 *  Print the values of the public fields of the given <code>struct</code> to 
426 *  <code>writer</code> line-by-line, prefixing each line with the value of <code>prefix</code>. 
427 *  If the object passed in is not a string or array, and if the prefix doesn't end in a '.', 
428 *  then a '.' will be appended to the prefix.  Strings will be double-quote delimited and arrays will
429 *  be iterated.
430 *
431 *  @param struct The object to print.  Any object can be passed in (including arrays),
432 *                                        though only it's public fields will be printed.
433 *  @param prefix The prefix string to write before all output lines generated by this method. (usually the
434 *                instance name of the struct being printed)
435 *  @param writer The writer to write the structure's description to.
436 *  @exeption java.io.IOException thrown if there's a problem using <code>writer</code>.
437 */
438static public void printStruct(Object struct, String prefix, Writer writer)
439        throws IOException
440{
441        try
442        {
443                if (struct == null)
444                {
445                        writer.write(prefix + " = null\n");
446                        return; // finished
447                }
448 
449                Class structType = struct.getClass();
450 
451                if (structType == String.class)
452                {
453                        // A string should be wrapped with double quotes ('"').
454                        //
455                        writer.write(prefix + " = \"");
456                        writer.write(struct.toString());
457                        writer.write("\"\n");
458                        return;        // finished
459                }
460                if (structType.isArray())
461                {
462                        // Handle arrays: print prefix + array index + element value
463                        // If the array type is not primitive, then recurse this method.
464                        //
465                        Object array = struct;
466                        int len = Array.getLength(array);
467                        if (len == 0)
468                        {
469                                writer.write(prefix + " = [empty]\n");
470                        }
471                        Class arrayType = structType.getComponentType();
472                        for (int j=0; j < len; ++j)
473                        {
474                                String pre = prefix + '[' + j + "] = ";
475                                if (arrayType.isPrimitive())
476                                {
477                                        if (arrayType==byte.class || arrayType==short.class ||
478                                            arrayType==int.class || arrayType==long.class)
479                                        {
480                                                writer.write(pre + Array.getLong(array, j));
481                                        }
482                                        else if (arrayType==char.class)
483                                                writer.write(pre + Array.getChar(array, j));
484                                        else if (arrayType==float.class || arrayType==double.class)
485                                                writer.write(pre + Array.getDouble(array, j));
486                                        else if (arrayType==boolean.class)
487                                                writer.write(pre + Array.getBoolean(array, j));
488                                        writer.write('\n');
489                                }
490                                else
491                                {
492                                        printStruct(Array.get(array, j), prefix + "[" + j + "]", writer);
493                                }
494                        }
495                        return;                // finished.
496                }
497 
498                // Process the struct field-by-field
499                //
500                Field[] fields = structType.getFields();
501                if (fields.length == 0)
502                {
503                        writer.write(prefix);
504                        writer.write(" [struct has no public fields ]\n");
505                }
506 
507                if (prefix.length() > 0 && prefix.charAt(prefix.length()-1) != '.')
508                        prefix += '.';
509 
510                for (int i=0; i < fields.length; ++i)
511                {
512                        Class type = fields[i].getType();
513                        if (Modifier.isStatic(fields[i].getModifiers()))
514                        {
515                                continue; // skip static fields
516                        }
517                        String fieldPrefix = prefix + fields[i].getName();
518                        if (type.isPrimitive())
519                        {
520                                // Print primitive types
521                                //
522                                writer.write(fieldPrefix);
523                                if (type==byte.class || type==short.class || type==int.class || type==long.class)
524                                        writer.write(" = "+fields[i].getLong(struct));
525                                else if (type==char.class)
526                                        writer.write(" = "+fields[i].getChar(struct));
527                                else if (type==float.class || type==double.class)
528                                        writer.write(" = "+fields[i].getDouble(struct));
529                                else if (type==boolean.class)
530                                        writer.write(" = "+fields[i].getBoolean(struct));
531                                else
532                                        writer.write(" = "+fields[i].get(struct));
533                                writer.write('\n');
534                        }
535                        else
536                        {
537                                // Recurse non-trivial structures!
538                                //
539                                printStruct(fields[i].get(struct), fieldPrefix, writer);
540                        }
541                }
542        }
543        catch (IllegalAccessException ex)
544        {
545                throw new Error("Unexpected access exception printing structure: " + ex);
546        }
547}
