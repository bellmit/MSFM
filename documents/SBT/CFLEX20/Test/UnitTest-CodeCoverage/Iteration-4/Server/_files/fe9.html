539    /**
540     * Extract the options and non-option arguments from the given list of
541     * command-line arguments. The specified locale is used for parsing options
542     * whose values might be locale-specific.
543     */
544    public final void parse(String[] argv, Locale locale) throws OptionException
545    {
546        ArrayList otherArgs = new ArrayList();
547        int position = 0;
548        this.values = new HashMap(10);
549        while (position < argv.length)
550        {
551            String curArg = argv[position];
552            if (curArg.startsWith("-"))
553            {
554                if (curArg.equals("--"))
555                { // end of options
556                    position += 1;
557                    break;
558                }
559                String valueArg = null;
560                if (curArg.startsWith("--"))
561                { // handle --arg=value
562                    int equalsPos = curArg.indexOf("=");
563                    if (equalsPos != -1)
564                    {
565                        valueArg = curArg.substring(equalsPos + 1);
566                        curArg = curArg.substring(0, equalsPos);
567                    }
568                }
569                Option opt = (Option) this.options.get(curArg);
570                if (opt == null)
571                {
572                    throw new UnknownOptionException(curArg);
573                }
574                Object value = null;
575                if (opt.wantsValue())
576                {
577                    if (valueArg == null)
578                    {
579                        position += 1;
580                        valueArg = null;
581                        if (position < argv.length)
582                        {
583                            valueArg = argv[position];
584                        }
585                    }
586                    value = opt.getValue(valueArg, locale);
587                }
588                else
589                {
590                    value = opt.getValue(null, locale);
591                }
592                this.values.put(opt.longForm(), value);
593                position += 1;
594            }
595            else
596            {
597                break;
598            }
599        }
600        for (; position < argv.length; ++position)
601        {
602            otherArgs.add(argv[position]);
603        }
604        
605        this.remainingArgs = (String[]) otherArgs.toArray(new String[otherArgs.size()]);
606        
607        Iterator grpIter = switchGroups.iterator();
608        while (grpIter.hasNext())
609        {
610            SwitchGroup grp = (SwitchGroup)grpIter.next();
611            grp.validate();
612        }
613    }
