329        throw ExceptionBuilder.dataValidationException("Unable to find product being adjusted", DataValidationCodes.INVALID_PRODUCT);
330    }
331    if (!hasPriceAdjustment(tempProduct))
332    {
333        setAdjustedProduct(tempProduct);
334        if (newAdjustment.newProductSymbol.length() == 0)
335        {
336            // set new symbol to current symbol if it isn't being changed.
337            setNewProductSymbol(tempProduct.getProductName().productSymbol);
338        }
339        else
340        {
341            setNewProductSymbol(newAdjustment.newProductSymbol);
342        }
343        // UD 06/06/05. moved setting of adjustment values to a private method.
344        setAdjustmentValues(newAdjustment);
345        Vector tempClasses = new Vector(newAdjustment.adjustedClasses.length);
346        setAdjustedClassCollection(tempClasses);
347        for (int i = 0; i < newAdjustment.adjustedClasses.length; i++)
348        {
349            addAdjustedClass(newAdjustment.adjustedClasses[i]);
350        }
351        createClassAdjustmentsForFutures();
352    }
353    else
354    {
355        throw ExceptionBuilder.alreadyExistsException("Adjustment already exists for product", AlreadyExistCodes.ADJUSTMENT_ALREADY_EXISTS);
356    }
357}

359/*
360*   UD 06/06/05
361*   set the failed product error details.
362*/
363public void create(PriceAdjustmentStruct newAdjustment, PriceAdjustmentReportingClassResultStruct[] reportingClassAdjustmentsResults) throws AlreadyExistsException, DataValidationException
364{
365        Product tempProduct;
366         try
367          {
368                tempProduct = findProduct(newAdjustment);
369          }
370           catch (NotFoundException e)
371    {
372        String msg = "PriceAdjustmentError: Unable to find product being adjusted" + newAdjustment.productKey;
373        Log.information(msg);
374            throw ExceptionBuilder.dataValidationException(msg, DataValidationCodes.INVALID_PRODUCT);
375    }
376        if (!hasPriceAdjustment(tempProduct))
377        {
378                setAdjustedProduct(tempProduct);
379                if (newAdjustment.newProductSymbol.length() == 0) {
380                        // set new symbol to current symbol if it isn't being changed.
381                        setNewProductSymbol(tempProduct.getProductName().productSymbol);
382                }
383                else {
384                        setNewProductSymbol(newAdjustment.newProductSymbol);
385                }
386        // UD 06/06/05... changes for IPD.
387                setAdjustmentValues(newAdjustment);
388                Vector tempClasses = new Vector(newAdjustment.adjustedClasses.length);
389                setAdjustedClassCollection(tempClasses);
390        int failedReportingClassAdjustments = 0;
391        for (int i = 0; i < newAdjustment.adjustedClasses.length; i++)
392        {
393            ExceptionDetails details = new ExceptionDetails();
394            reportingClassAdjustmentsResults[i] = initializeReportingClassResultStruct(newAdjustment.adjustedClasses[i].items.length, newAdjustment.productKey);
395            if(!validateReportingClassAdjustment(newAdjustment.adjustedClasses[i], details)) {
396                setAdjustmentResult(details, reportingClassAdjustmentsResults[i], newAdjustment.adjustedClasses[i].classKey);
397                failedReportingClassAdjustments++;
398                Log.alarm(reportingClassAdjustmentsResults[i].reportingClassErrorResult.errorMessage);
399                continue;
400            }
401            if(!addAdjustedClass(newAdjustment.adjustedClasses[i], reportingClassAdjustmentsResults[i])) {
402                // increment failed reporting classes count
403                failedReportingClassAdjustments++;
404            }
405        }
406        if(failedReportingClassAdjustments == newAdjustment.adjustedClasses.length) {
407            // If there are errors on all reporting class for this underlying, throw exception.
408            throw ExceptionBuilder.dataValidationException("All reporting class adjustments have exceptions for[" +
409                + newAdjustment.productKey + "], product adjustment not created", DataValidationCodes.INVALID_PRODUCT);
410        }
411        }
412        else
413        {
414        String msg = "PriceAdjustmentError: Adjustment already exists for product" + newAdjustment.productKey;
415        Log.information(msg);
416                throw ExceptionBuilder.alreadyExistsException(msg, AlreadyExistCodes.ADJUSTMENT_ALREADY_EXISTS);
417        }

661/**
662 * Gets the contract multiplier for reporting classes.
663 */
664private int getNewSharesPerContract(int oldShares)
665{
666    int newShares = oldShares;
667    switch (getAdjustmentType())
668    {
669    case PriceAdjustmentTypes.SPLIT:
670        if (getSplitDenominator() != 1)
671        {
672            newShares = (oldShares * getSplitNumerator()) / getSplitDenominator();
673        }
674        break;
675    case PriceAdjustmentTypes.DIVIDEND_CASH:
676        // no affect on shares per contract
677        break;
678    case PriceAdjustmentTypes.DIVIDEND_PERCENT:
679        // convert percentage to decimal
680        newShares = (int) (1.00 + getStockDividend().toDouble() / 100.0) * oldShares;
681        break;
682    case PriceAdjustmentTypes.DIVIDEND_STOCK:
683        // add in new shares
684        newShares = oldShares + (int) getStockDividend().toDouble();
685        break;
686    case PriceAdjustmentTypes.LEAP_ROLLOVER:
687        // no affect on shares per contract
688        break;
689    case PriceAdjustmentTypes.MERGER:
690        // no affect on shares per contract
691        break;
692    case PriceAdjustmentTypes.SYMBOL_CHANGE:
693        // no affect on shares per contract
694        break;
695    case PriceAdjustmentTypes.COMMON_DISTRIBUTION:
696        // no affect on shares per contract
697        break;
698    default:
699        Log.alarm(this, "Unknown adjustment type (" + getAdjustmentType() + ") assuming doesn't affect shares per contract");
700    }
701    return newShares;
702}
