715 */
716protected void recreateRFQs() {
717    Order[] marketOrders = getOrderHome().findActiveMarketOrders();
718    for (int i = 0; i < marketOrders.length; i++) {
719        // find first unprocessed market order
720        if (marketOrders[i] != null) {

722            int rfqQuantity = marketOrders[i].getQuantityAllowed();
723            marketOrders[i] = null;
724            // get total quantity for all market orders for product
725            for (int j = i + 1; j < marketOrders.length; j++) {
726                if(marketOrders[j] != null)
727                {

729                        rfqQuantity += marketOrders[j].getQuantityAllowed();
730                        marketOrders[j] = null;
731                    }
732                }
733            }
734            // create RFQ for product if product is in trading state.
735            try {
736                TradingProduct product = getTradingProductHome().findByKey(productKey);
737                if (product.inTradingState()) {
738                            Broker broker = null;
739                            try
740                            {
741                                    broker = getBroker(productKey, null);
742                            }

746                                    throw dve;

749                }
