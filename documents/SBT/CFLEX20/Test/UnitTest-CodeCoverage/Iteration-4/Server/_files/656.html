382        IntHashMap allocationStrategies = new IntHashMap();
383        AllocationStrategyStructV2 strategyStruct;
384        AllocationStrategy strategy;
385        for (int i = 0; i < strategyStructs.length; i++ ){

390 
391 
392        //If TradeType STRATEGY_AUCTION_TRADE(8) is not found, populate it from the default class.

394        if(allocationStrategies.get(spreadAuctionTradeKey) == null)
395        {
396                if(defaultStrategyStructs == null)
397                {
398                                Log.information("AllocationStrategyFactoryImpl >>>Getting the default class's allocation strategy value for (8) trade type");
399                        defaultStrategyStructs = aTradingClass.getDefaultAllocationStrategiesForClass();
400                        }
401            for (int i = 0; i < defaultStrategyStructs.length; i++ )
402            {
403                    AllocationStrategyStructV2 defaultStrategyStruct = defaultStrategyStructs[i];
404                    if(defaultStrategyStruct.allocationTradeType == AllocationTradeTypes.STRATEGY_AUCTION_TRADE)
405                    {
406                            AllocationStrategy defaultStrategy = createStrategy(defaultStrategyStruct, aTradingClass);
407                            allocationStrategies.put(spreadAuctionTradeKey, defaultStrategy);
408 
409                    }
410 
411            }
412        }
413 
414        // If the TradeType STRATEGY_AUCTION_TRADE(8) is not set for the default too, use the hardcoded value.
415        if(allocationStrategies.get(spreadAuctionTradeKey) == null)
416        {
417                        Log.information("AllocationStrategyFactoryImpl >>> ** AllocationStrategy TradeType SpreadAuctionTrade(8) Not set for Default , Using the hardcoded value 8=1;4;21 ***");
418                        if(defaultStrategyForSpreadAuctionTrade == null)
419                        {
420                                initializeDefaultForSpreadAuctionTradeType();
421                        }
422                allocationStrategies.put(spreadAuctionTradeKey, defaultStrategyForSpreadAuctionTrade);
423        }
424 
425 
426        int regularKey = AllocationTradeTypes.REGULAR;
427        if (allocationStrategies.get(regularKey) == null){
428            allocationStrategies.put(regularKey, defaultStrategy);
429        } // if there is no strategy configured for REGULAR trade type, the default is used.
430        return allocationStrategies;

451    /**
452     * Creates a strategy from the a definition struct.
453     */
454    private AllocationStrategy createStrategy(AllocationStrategyStructV2 strategyStruct, TradingClass aTradingClass) {
455        ArrayList regularElements = new ArrayList();
456        ArrayList contingencyElements = new ArrayList();
457        int key;
458        AllocationStrategyElement aStrategyElement;
459        for (int i = 0; i < strategyStruct.prioritizedStrategyCodes.length; i++) {
460            aStrategyElement = getPriorityStrategyElement(aTradingClass, strategyStruct.prioritizedStrategyCodes[i]);
461            if (aStrategyElement != null ){
462                regularElements.add(aStrategyElement);
463            }
464            else{
465                Log.alarm("AllocationStrategyFactoryImpl >>> Unknown or invalid priority allocation strategy code skipped.  Code = " + strategyStruct.prioritizedStrategyCodes[i]);
466            }
467        }
468        key = strategyStruct.defaultStrategyCode;
469        aStrategyElement = (AllocationStrategyElement) getBaseStrategyMap().get(key);
470        if (aStrategyElement != null ){
471            regularElements.add(aStrategyElement);
472        }
473        else{
474            Log.alarm("Unknown or invalid default strategy code = " + strategyStruct.defaultStrategyCode + " not used.  Price/Time strategy used instead.");
475            regularElements.add(priceTimeStrategy); // default to price/time strategy
476        }
477               
478        // if there is reserved quantity allocation strategy configured, add this before contingency strategy
479        if(this.reservedQuantityStrategies != null)
480        {    
481            regularElements.addAll(this.reservedQuantityStrategies);
482        }
483        
484        contingencyElements.add(contingencyStrategy);
485        regularElements.addAll(contingencyElements); // combine elements to create strategies
486        return new AllocationStrategyImpl(marketOrderStrategy, takeAllStrategy, regularElements);
487    }
