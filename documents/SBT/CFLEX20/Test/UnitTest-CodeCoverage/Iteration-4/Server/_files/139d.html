425        void rehash()
426        {
427            HashEntry<V>[] oldTable = table;
428            int oldCapacity = oldTable.length;
429            if (oldCapacity >= MAXIMUM_CAPACITY)
430                return;
431 
432            /*
433             * Reclassify nodes in each list to new Map. Because we are using
434             * power-of-two expansion, the elements from each bin must either
435             * stay at same index, or move with a power of two offset. We
436             * eliminate unnecessary node creation by catching cases where old
437             * nodes can be reused because their next fields won't change.
438             * Statistically, at the default threshold, only about one-sixth of
439             * them need cloning when a table doubles. The nodes they replace
440             * will be garbage collectable as soon as they are no longer
441             * referenced by any reader thread that may be in the midst of
442             * traversing table right now.
443             */
444 
445            HashEntry<V>[] newTable = HashEntry.newArray(oldCapacity << 1);
446            threshold = (int) (newTable.length * loadFactor);
447            int sizeMask = newTable.length - 1;
448            for (int i = 0; i < oldCapacity; i++)
449            {
450                // We need to guarantee that any existing reads of old Map can
451                // proceed. So we cannot yet null out each bin.
452                HashEntry<V> e = oldTable[i];
453 
454                if (e != null)
455                {
456                    HashEntry<V> next = e.next;
457                    int idx = e.hash & sizeMask;
458 
459                    // Single node on list
460                    if (next == null)
461                        newTable[idx] = e;
462 
463                    else
464                    {
465                        // Reuse trailing consecutive sequence at same slot
466                        HashEntry<V> lastRun = e;
467                        int lastIdx = idx;
468                        for (HashEntry<V> last = next; last != null; last = last.next)
469                        {
470                            int k = last.hash & sizeMask;
471                            if (k != lastIdx)
472                            {
473                                lastIdx = k;
474                                lastRun = last;
475                            }
476                        }
477                        newTable[lastIdx] = lastRun;
478 
479                        // Clone all remaining nodes
480                        for (HashEntry<V> p = e; p != lastRun; p = p.next)
481                        {
482                            int k = p.hash & sizeMask;
483                            HashEntry<V> n = newTable[k];
484                            newTable[k] = new HashEntry<V>(p.key, p.hash, n, p.value);
485                        }
486                    }
487                }
488            }
489            table = newTable;
490        }

493         * Remove; match on key only if value null, else match both.
494         */
495        V remove(int key, int hash, Object value)

556     * Creates a new, empty map with the specified initial capacity, load factor
557     * and concurrency level.
558     * 
559     * @param initialCapacity
560     *            the initial capacity. The implementation performs internal
561     *            sizing to accommodate this many elements.
562     * @param loadFactor
563     *            the load factor threshold, used to control resizing. Resizing
564     *            may be performed when the average number of elements per bin
565     *            exceeds this threshold.
566     * @param concurrencyLevel
567     *            the estimated number of concurrently updating threads. The
568     *            implementation performs internal sizing to try to accommodate
569     *            this many threads.
570     * @throws IllegalArgumentException
571     *             if the initial capacity is negative or the load factor or
572     *             concurrencyLevel are nonpositive.
573     */
574    public ConcurrentIntHashMap(int initialCapacity, float loadFactor, int concurrencyLevel)
575    {
576        if (!(loadFactor > 0) || initialCapacity < 0 || concurrencyLevel <= 0)
577            throw new IllegalArgumentException();
578 
579        if (concurrencyLevel > MAX_SEGMENTS)
580            concurrencyLevel = MAX_SEGMENTS;
581 
582        // Find power-of-two sizes best matching arguments
583        int sshift = 0;
584        int ssize = 1;
585        while (ssize < concurrencyLevel)
586        {
587            ++sshift;
588            ssize <<= 1;
589        }
590        segmentShift = 32 - sshift;
591        segmentMask = ssize - 1;
592        this.segments = Segment.newArray(ssize);
593 
594        if (initialCapacity > MAXIMUM_CAPACITY)
595            initialCapacity = MAXIMUM_CAPACITY;
596        int c = initialCapacity / ssize;
597        if (c * ssize < initialCapacity)
598            ++c;
599        int cap = 1;
600        while (cap < c)
601            cap <<= 1;
602 
603        for (int i = 0; i < this.segments.length; ++i)
604            this.segments[i] = new Segment<V>(cap, loadFactor);
605    }

670     * 
671     * @return <tt>true</tt> if this map contains no key-value mappings
672     */
673    public boolean isEmpty()
674    {
675        final Segment<V>[] segments = this.segments;
676        /*
677         * We keep track of per-segment modCounts to avoid ABA problems in which
678         * an element in one segment was added and in another removed during
679         * traversal, in which case the table was never actually empty at any
680         * point. Note the similar use of modCounts in the size() and
681         * containsValue() methods, which are the only other methods also
682         * susceptible to ABA problems.
683         */
684        int[] mc = new int[segments.length];
685        int mcsum = 0;
686        for (int i = 0; i < segments.length; ++i)
687        {
688            if (segments[i].count != 0)
689                return false;
690            else
691                mcsum += mc[i] = segments[i].modCount;
692        }
693        // If mcsum happens to be zero, then we know we got a snapshot
694        // before any modifications at all were made. This is
695        // probably common enough to bother tracking.
696        if (mcsum != 0)
697        {
698            for (int i = 0; i < segments.length; ++i)
699            {
700                if (segments[i].count != 0 || mc[i] != segments[i].modCount)
701                    return false;
702            }
703        }
704        return true;
705    }

707    /**
708     * Returns the number of key-value mappings in this map. If the map contains
709     * more than <tt>Integer.MAX_VALUE</tt> elements, returns
710     * <tt>Integer.MAX_VALUE</tt>.
711     * 
712     * @return the number of key-value mappings in this map
713     */
714    public int size()
715    {
716        final Segment<V>[] segments = this.segments;
717        long sum = 0;
718        long check = 0;
719        int[] mc = new int[segments.length];
720        // Try a few times to get accurate count. On failure due to
721        // continuous async changes in table, resort to locking.
722        for (int k = 0; k < RETRIES_BEFORE_LOCK; ++k)
723        {
724            check = 0;
725            sum = 0;
726            int mcsum = 0;
727            for (int i = 0; i < segments.length; ++i)
728            {
729                sum += segments[i].count;
730                mcsum += mc[i] = segments[i].modCount;
731            }
732            if (mcsum != 0)
733            {
734                for (int i = 0; i < segments.length; ++i)
735                {
736                    check += segments[i].count;
737                    if (mc[i] != segments[i].modCount)
738                    {
739                        check = -1; // force retry
740                        break;
741                    }
742                }
743            }
744            if (check == sum)
745                break;
746        }
747        if (check != sum)
748        { // Resort to locking all segments
749            sum = 0;
750            for (int i = 0; i < segments.length; ++i)
751                segments[i].lock();
752            for (int i = 0; i < segments.length; ++i)
753                sum += segments[i].count;
754            for (int i = 0; i < segments.length; ++i)
755                segments[i].unlock();
756        }
757        if (sum > Integer.MAX_VALUE)
758            return Integer.MAX_VALUE;
759        else
760            return (int) sum;
761    }
