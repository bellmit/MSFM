469    private int[] getAllQuoters(int[] participants, OrderBook book)
470    {
471        Set<Integer> quoters = Collections.synchronizedSet(new HashSet<Integer>());
472        
473        if (participants != null)
474        {
475            for (int i = 0; i < participants.length; i++)
476            {
477                quoters.add(new Integer(participants[i]));
478            }
479        }
480 
481        Order[] quoteLikeOrders = getQuoteLikeOrders(book);
482        for (int i=0; i<quoteLikeOrders.length; i++)
483        {
484            quoters.add(new Integer(quoteLikeOrders[i].getUserKey()));
485        }
486 
487        int[] allQuoters =new int[quoters.size()];
488        int index = 0;
489        Iterator ite = quoters.iterator();
490        while (ite.hasNext())
491        {
492            allQuoters[index++] = ((Integer)ite.next()).intValue();
493        }
494 
495        return allQuoters;
496    }

498    private Order[] getQuoteLikeOrders(OrderBook book)
499    {
500        Tradable tradable;
501        Set<Order> quoteLikeOrders = Collections.synchronizedSet(new HashSet<Order>());
502 
503        Enumeration enumeration = book.getTradables(SideFactory.getBuySide());
504        while (enumeration.hasMoreElements())
505        {
506            tradable = (Tradable)enumeration.nextElement();
507            if (tradable.isTradable(TradableTypes.ORDER) && tradable.treatedLikeQuote())
508            {
509                quoteLikeOrders.add(((Order)tradable));
510            }
511        }
512 
513        enumeration = book.getTradables(SideFactory.getSellSide());
514        while (enumeration.hasMoreElements())
515        {
516            tradable = (Tradable)enumeration.nextElement();
517            if (tradable.isTradable(TradableTypes.ORDER) && tradable.treatedLikeQuote())
518            {
519                quoteLikeOrders.add(((Order)tradable));
520            }
521        }
522 
523        return (Order[])quoteLikeOrders.toArray(new Order[quoteLikeOrders.size()]);
524    }
