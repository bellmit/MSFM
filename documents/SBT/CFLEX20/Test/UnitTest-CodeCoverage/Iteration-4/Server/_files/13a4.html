550                {

552                    return;
553                
554                try

556                    for (int i = 0; i < productStates.length; i++)
557                    {
558                        if (productStates[i].productState == ProductStates.CLOSED ||
559                            productStates[i].productState == ProductStates.HALTED ||
560                            productStates[i].productState == ProductStates.SUSPENDED)
561                        {
562                                if (Log.isDebugOn())
563                                {
564                                        Log.debug(this, "Product state changed to "+productStates[i].productState+ " cancelling Light Orders for session (" + sessionName + ")");
565                                }
566                            short cancelReason;
567                            switch (productStates[i].productState)
568                            {
569                                case ProductStates.HALTED:    
570                                        cancelReason = InternalActivityReasons.INTERNAL_PROD_HALTED; 
571                                        break;
572                                case ProductStates.SUSPENDED: 
573                                        cancelReason = InternalActivityReasons.INTERNAL_PROD_SUSPENDED; 
574                                        break;
575                                case ProductStates.CLOSED:
576                                default:                      
577                                        cancelReason = InternalActivityReasons.INTERNAL_PROD_CLOSED; 
578                                        break;
579                            }                       
580                            
581                            cancelLightOrdersForProduct(productStates[i].productKeys.productKey, cancelReason, productStates[i].productState);
582                        }
583                    }
584                }
585                catch (Exception e){
586                    Log.exception(this, "Exception while trying to cancel Light Orders on product state change.", e);
587                }
588                }
