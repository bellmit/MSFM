403 
404        SessionClassInfo sessionClassInfo = (SessionClassInfo) (getClassSessionMap().get( new Integer(classKey)));
405        if(sessionClassInfo == null) {

411        currTime = toMillisSinceMidnight(currTime);
412        if(Log.isDebugOn()) Log.debug("Current time is: " + currTime);
413        for(int i = 0; i < sessionClassInfo.namelist.size(); i++){

415                   ((OpenTimeRange)sessionClassInfo.openTimeRange.get(i)).productPreOpenTime +
416              " CloseTime: " + ((OpenTimeRange)sessionClassInfo.openTimeRange.get(i)).productCloseTime);
417           if(currTime >((OpenTimeRange)sessionClassInfo.openTimeRange.get(i)).productPreOpenTime
418              && currTime < ((OpenTimeRange)sessionClassInfo.openTimeRange.get(i)).productCloseTime){
419               sessionName = (String)sessionClassInfo.namelist.get(i);
420               break;
421           }
422        }
423        if(Log.isDebugOn()) Log.debug("Returning session name: " + sessionName);
424        return sessionName;
