460    public static String formatIOSCMessage(String message, boolean shouldDisplayAll)
461    {
462        String[] fields = message.split(";");
463            
464        StringBuffer formattedString = new StringBuffer();
465        
466        Iterator fieldNameKeysItr = StructFormatter.FIELDS_MAP.keySet().iterator();
467        
468        while ( fieldNameKeysItr.hasNext())
469        {
470            Short fieldNameKey;
471            String fieldNameValue;
472            boolean fieldFound = false;
473            int numberOfEmptySpaces = 0;
474 
475            fieldNameKey= (Short) fieldNameKeysItr.next();
476            fieldNameValue= (String) StructFormatter.FIELDS_MAP.get(fieldNameKey);
477            
478            if (!shouldDisplayAll && fieldNameKey.shortValue() >  StructFormatter.DEFAULT_FIELDS_MAX_KEY){continue;}
479            
480            for (int i = 0; i < fields.length; i++ )
481            {
482                try {
483                    String[] fieldValues = fields[i].split("=");
484                    Short fieldValueKey;
485                    String fieldValue;
486                    //If no value , skipp.
487                    if ( fieldValues.length < 1 ){
488                        continue;
489                     }; 
490                    
491                    fieldValueKey = new Short(fieldValues[0]);
492                    fieldValue = fieldValues[1];
493                    
494                    if ( fieldValueKey.shortValue() == fieldNameKey.shortValue() ) 
495                    {
496                        fieldFound = true;
497                        formattedString.append(fieldValue);
498                        numberOfEmptySpaces = fieldNameValue.length() -  fieldValue.length();
499                    }                        
500                } catch (java.lang.IndexOutOfBoundsException e) {
501                    //ignore 
502                } catch (java.lang.NumberFormatException e) {
503                    //ignore 
504                }
505            }
506            if (!fieldFound) 
507            {
508                numberOfEmptySpaces = fieldNameValue.length();
509            }
510            for ( int j = -1; j < numberOfEmptySpaces ; j++)
511            {
512                formattedString.append(" ");
513            }
514            
515        }
516        
517            
518        return formattedString.toString();
519    }
