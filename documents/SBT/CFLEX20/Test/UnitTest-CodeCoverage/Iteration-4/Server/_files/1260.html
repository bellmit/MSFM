464     *  - Cancel Manual Quotes for Orders not clearing like quote and don't change Spread Quantity.
465     *  - For Orders not clearing like quote, change spread quantity and check if hit max auto_ex.
466     * returns false 
467     *  - If not indexhybrid enabled .
468     *  - If indexhybrid enabled and has not hit max AUTO_EX.

471    protected static boolean processIndexHybridOrder(boolean isIndexHybridEnabled, Tradable newTradable, StrategyLegTradeContainer legTradeContainer, 
472            TradingProduct product, DerivedQuote derivedQuote, Price price, int iterationQuantity) throws DataValidationException,SystemException
473    {
474        boolean hitMaxNonCustQty = false;
475        
476        if (!isIndexHybridEnabled || ((Order)newTradable).getContingencyType() == ContingencyTypes.AUCTION_RESPONSE) {
477            return false;
478        }
479        
480        // Cancel ManualQuotes if the incoming spread I-order can trade against the leg market
481        // This requirement does not make sense anymore.. in proecessTradable.checkOrderNeedsToBeReturned() its made sure that
482        // the spread order will be returned back if found to be eligible for a leg trade w/ manual quote present 
483        // so, below code will not get executed ever.. test it, remove it.
484//        if(((Order)newTradable).clearsLikeQuote()) {
485//            if(IndexHybridStrategyHelper.manualQuoteExistsOnTradableLegSide(product, newTradable.getSide())
486//                    && legTradeContainer.getCanTradeWithLegs())
487//            {
488//                Log.information("XXX SHOULD NOT SEE THIS LINE ...");
489//                Log.debug("XXX SHOULD NOT SEE THIS LINE ...");
490//                checkAndCancelManualQuotes(product, newTradable.getSide(), legTradeContainer);
491//            }
492//        }
493        
494        // Change incoming spread qty if tradable is an order and doesn't clear like Quote ...
495        if(newTradable.isTradable(TradableTypes.ORDER) && !((Order)newTradable).clearsLikeQuote()) 
496        {
497            TradingStrategyLeg[] legs = product.getStrategyLegs();
498            int totalAllowedSpreadQuantity = 0;
499            IndexHybridAutoExQtyHolder[] autoExHolders = getAllowedLegQuantityHolders(newTradable, legs, legTradeContainer);
500            HandlingInstruction instruction = ((Order)newTradable).getHandlingInstruction();
501           // check if is PAR T/A or T/B
502            if ( instruction != null && instruction.getTradingRestriction() == TradingRestrictions.PAR_TRADE_ALL && 
503                 !IndexHybridStrategyHelper.manualQuoteExistsOnTradableLegSide(product, newTradable.getSide()))
504            {
505                for (IndexHybridAutoExQtyHolder theAutoExHolder : autoExHolders) 
506                {
507                    int autoExSizeForClass = theAutoExHolder.getAutoExSizeForClass();
508                    int custQty = theAutoExHolder.getCustomerQuantity(); // since MQ exists this is only CUST QTY.
509                    int notCustomerQty = theAutoExHolder.getNonCustomerQuantity();
510                    int finalAutoExQty = 0;
511                    
512                    if(notCustomerQty > autoExSizeForClass)

515                        theAutoExHolder.setAvailableNonCustQtyMoreThanAutoEx(true);
516                    }

520                    }
521                    theAutoExHolder.setMaxAutoExQty(finalAutoExQty);
522                }
523            }
524 
525            // If there is MQ
526            totalAllowedSpreadQuantity = getMaxAllowedQuantity(newTradable, legs, legTradeContainer, autoExHolders, legTradeContainer.getUseContingentBest() ? false : true);
527            if (totalAllowedSpreadQuantity < iterationQuantity) {
528                setMaxAllowedQuantityOnTradable(newTradable, totalAllowedSpreadQuantity,
529                        OrderRoutingReasons.STRATEGY_PARTIALLY_TRADED_AGAINST_AUTO_EX);
530            }
531            hitMaxNonCustQty = hitMaxNonCustomerQuantity(legs, autoExHolders, totalAllowedSpreadQuantity, legTradeContainer.getUseContingentBest() ? false : true);

545    /*
546     * For Leg book with Contingent Orders that were tradable before IH qty change not tradable now
547     * recalculate the DQ and StrategLegTradeContainer, if the changed qty has is tradable against 
548     * contingent market or not. If not tradable, this order has to be rtreturned.
549     * If tradable, then recalculate the tradable spread qty so all prices points are considered with the
550     * incoming tradable spread price. Deduct the previous minimum contingent quantity as or we can end up 
551     * with trading more than autoex for non customer quantity, as the holder returns the totdal available 
552     * quantity till the worst price point but if the spread is not tradable at the previous contignent prices
553     * should be removed from the current total allowed spread quantity
554     * @return true if maxAutoExHit has been hit else return false ...
555     */
556    private static boolean processContingeny(boolean isIndexHybridEnabled, Tradable newTradable, TradingStrategyLeg[] legs,
557            TradingProduct product,  Price price, int iterationQuantity, int totalAllowedSpreadQuantity, StrategyLegTradeContainer legTradeContainer) throws DataValidationException
558    {
559        if(!legTradeContainer.getCanTradeWithLegs()){
560            return true;
561        }
562        else {
563            ((Order) newTradable).getHandlingInstruction().setDefaultMaxTradableQtyForIndexHybrid();
564            IndexHybridAutoExQtyHolder[] autoExHolders = getAllowedLegQuantityHolders(newTradable, legs, legTradeContainer);
565            // If can't trade with Contingent, filter out contingent qty from desired auto ex qty on each leg. 
566            totalAllowedSpreadQuantity = getMaxAllowedQuantity(newTradable, legs, legTradeContainer, autoExHolders, legTradeContainer.getUseContingentBest() ? false : true);
567            int prevMinContingentQuantity = getPrevMinContingentQuantity(autoExHolders);
568            totalAllowedSpreadQuantity = prevMinContingentQuantity > 0 && prevMinContingentQuantity < totalAllowedSpreadQuantity ? 
569                                            totalAllowedSpreadQuantity - prevMinContingentQuantity : 
570                                                totalAllowedSpreadQuantity;
571            if (totalAllowedSpreadQuantity < iterationQuantity) {
572                setMaxAllowedQuantityOnTradable(newTradable, totalAllowedSpreadQuantity,
573                        OrderRoutingReasons.STRATEGY_PARTIALLY_TRADED_AGAINST_AUTO_EX);
574            }
575            return hitMaxNonCustomerQuantity(legs, autoExHolders, totalAllowedSpreadQuantity, true);
576        }
577    }
