715/**

720 */
721public void save() throws PersistenceException
722{

724    productName = ProductStructBuilder.toString(getProductName());

730        productName = productName + ":" + System.currentTimeMillis();

733        super.save();

817private void updateCompositeProductListingStates(short newState)
818{
819    // MUST be done here.  Otherwise NullPointerExceptions will occur.
820    ProductHomeImpl myHome = getProductHomeImpl();
821    myHome.addToContainer(this);
822 
823    ProductComponent[] components = myHome.getProductComponentHome().findByComponent(this);
824 
825    if(Log.isDebugOn())
826    {
827        Log.debug("ProductImpl: Found " + components.length + " product components which contain product key = " +
828            getProductKey());
829    }
830 
831    for(int i = 0; i < components.length; i++)
832    {
833        Product product = components[i].getComposite();
834 
835        if(product != null)
836        {
837            if(Log.isDebugOn())
838            {
839                Log.debug("ProductImpl: Marking composite product with product key " + product.getProductKey() +
840                    " inactive.");
841            }
842 
843            // NOTE: Transaction should be handled outside this method
844            product.setListingState(newState);
845        }
846    }            
847}

849private void updateDerivativeProductClassListingStates(short newState)
850{
851    // MUST be done here.  Otherwise NullPointerExceptions will occur.
852    ProductHomeImpl myHome = getProductHomeImpl();
853    myHome.addToContainer(this);

855    ProductClass[] classes = myHome.getProductClassHome().findByUnderlying(this, true);

868        }
869 
870        // NOTE: Transaction should be handled outside this method
