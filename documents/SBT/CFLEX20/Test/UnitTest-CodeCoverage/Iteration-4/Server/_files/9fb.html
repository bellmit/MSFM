488    /*
489     * All market orders received outside of the regular trading session will be rejected 
490     * unless they have a valid cross contingency type. The checking can be turned off by 
491     * setting the property enableRegularTradingTimeValidation to false.     
492     */
493    protected void validateMarketOrderDuringExtendedTradingHours(OrderHandlingStruct order) throws DataValidationException
494    {
495        if ( order.price.type == PriceTypes.MARKET && !isMarketOrderCrossType(order))
496        {
497            // It is a market order and it does not have any valid cross type contingency.
498            // So, check to see if the current time is outside regular trading hours 
499            // and the extended trading hour validation is enabled
500            TradingClass tradingClass = null;  
501            try
502            {
503                tradingClass = getTradingProductHome().findByKey(order.productKey).getTradingClass();
504                if (tradingClass != null && tradingClass.getEnabledRegularMaketTimeValidation())
505                {
506                    // check if the current time is outside of the regular trading hours
507                    int curTimeInMillisFromMidnight = TimeHelper.getMillisSinceMidnight();
508                    int openTime = tradingClass.getRegularMarketHours().getOpenTimeAsMillisSinceMidnight();
509                    int closeTime = tradingClass.getRegularMarketHours().getCloseTimeAsMillisSinceMidnight();
510                    boolean reject = false;
511                    if (openTime < closeTime)
512                    {
513                        if (curTimeInMillisFromMidnight < openTime || curTimeInMillisFromMidnight > closeTime)
514                        {
515                            reject = true;
516                        }
517                    }
518                    else if (openTime > closeTime)
519                    {
520                        // openTime later than closeTime means session spans over two days. 
521                        if (curTimeInMillisFromMidnight < openTime && curTimeInMillisFromMidnight > closeTime)
522                        {
523                            reject = true;
524                        }
525                    }
526                    if (reject)
527                    {
528                        throw ExceptionBuilder.dataValidationException(
529                                "Invalid order.  Market orders are not accepted outside regular market hours. ",
530                                DataValidationCodes.INVALID_STATE);
531                    }
532                }           
533            }
534            catch( NotFoundException nfe )
535            {
536                Log.alarm("Could not find trading product, skipping extended trading hour validation" );
537            }
538        }
