755    public void processUpdateAllOrdersFromSync()
756    {
757        remoteCache.addListener(this);
758        Map<Long, DistributedCacheValue<OrderSyncData>> allData = remoteCache.getAll();
759        syncLocalCacheWithRemoteCache(allData);
760 
761        boolean success = false;
762        Order order = null;
763        OrderSyncData orderData=null;
764        Set<Long> allOrderIdValues = allData.keySet();
765        Iterator<Long> iter = allOrderIdValues.iterator();
766        Long orderId = 0L;
767        try {
768            Log.information(this, "Retrieved " + allData.size() + " orders from order sync. Updating local order cache.");
769            Transaction.startTransaction();
770            while (iter.hasNext()) {
771                orderId = iter.next();
772                DistributedCacheValue<OrderSyncData> data = allData.get(orderId);
773                orderData = data.getValue();
774                if (orderProcessor.shouldRetainOrderAtStartup(orderData))  {
775                    try {
776                        order = orderHome.create(data.getValue().getOrderHandlingStruct());
777                    }
778                    catch (Exception e) {

780                        Log.exception(this, e);
781                    }
782                    if (order != null) {
783                        updateOrder(allData.get(orderId), order);
784                        String openSession = getSessionNameHelper().getOpenConfiguredSessionName();
785                        if (openSession != null) {
786                            boolean removeOrderFromCache = orderProcessor.updateOrderAtStart(orderData);
787                            if (removeOrderFromCache) {
788                                orderHome.deleteOrderFromCache(order);
789                            }
790                        }
791                    }
792                }
793                else {
794                    localCache.remove(orderId);
795                    this.logOrderSyncEvent("Skippig order with zero qty. ", orderData.getOrderIdStruct(), orderData.getOrderId(), 
796                            orderData.getOrderHandlingStruct().productKey);
797                }
798 
799 
800            }
801            success = Transaction.commit();
802 
803            if (orderData != null)
804            {
805                    Log.information("Finished processinf all orders from remote cache, last order processed:"+orderData );
806            }
807            
808        }
809        catch (Exception e)
810        {
811            Log.alarm(this, "Encountered error while updating order (" + orderId + "): " + e.getMessage());
812            Log.exception(this, e);
813        }
814 
815        finally {
816            if (!success) {
817                Transaction.rollback();
818                localCache.clear();
819                Log.alarm(this, "Encountered error while processing orders from order sync");                
820            }
821            else {
822                Log.information(this, "Processed " + allData.size() + " orders retrieved from order sync.");
