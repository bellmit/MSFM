979/**
980 * Updates this class from a CORBA struct.
981 *
982 * @see ProductClass#update
983 */
984public void update(ClassDefinitionStruct updatedClass) throws DataValidationException
985{
986    try
987    {
988        // Underlying product must be changed through adjustments
989        setSymbol(updatedClass.classSymbol);
990        setPrimaryExchange(updatedClass.primaryExchange);
991        setProductDescription(updatedClass.descriptionName);
992        setListingState(updatedClass.listingState);
993        setDefaultTransactionFeeCode(updatedClass.defaultTransactionFeeCode);
994        setIsTestClass(updatedClass.testClass);
995 
996        //if the underlying is not provided by checking the product key = 0 or product sym = "", not update the underlying
997        if (isUnderlyingRequired(updatedClass.productType) && isUnderlyingProvided(updatedClass.underlyingProduct) )
998        {
999            Product underlyingProduct = findProduct(updatedClass.underlyingProduct.productKeys.productKey, updatedClass.underlyingProduct.productName);
1000            setUnderlyingProduct(underlyingProduct);
