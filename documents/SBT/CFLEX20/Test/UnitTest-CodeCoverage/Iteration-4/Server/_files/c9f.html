517 *  Returns the indices in pattern of occurrences of the given character ch.
518 */
519protected static int[] getPatternPositions(String pattern, char ch)
520{
521    int[] indices = new int[1024]; // much larger than anything we would ever expect.
522    int num = 0;
523    int last = -1;
524    while (true)
525    {
526        int idx = pattern.indexOf(ch, last+1);
527        if (idx < 0)
528        {
529            break;
530        }
531        indices[num++] = idx;
532        last = idx;
533    }
534 
535    int[] result = new int[num];
536    System.arraycopy(indices, 0, result, 0, num);
537    return result;
