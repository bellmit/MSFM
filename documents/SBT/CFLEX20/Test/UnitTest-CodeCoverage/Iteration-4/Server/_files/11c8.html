501/**
502 * Clear a given structure's public fields (recursing references to other structures).
503 * All public data members will be cleared.  If a given field is a primitive type, then
504 * set it to it's default setting (0, 0.0, or false); if it's a string, then the
505 * value "" is used; if it's another class, then the method is recursed to implement
506 * that method.  Note that a mutual dependency can cause infinite recursion (ie, class
507 * A contains a reference to B, and B contains a reference to A).
508 *
509 * @author Steven Sinclair
510 * @return java.lang.Object
511 * @param structClass java.lang.Class
512 */
513public static void clearStruct(Object struct)
514{
515        try
516        {
517                Field[] fields = struct.getClass().getFields();
518                for (int i=0; i < fields.length; ++i)
519                {
520                        Class type = fields[i].getType();
521                        if (String.class == type)
522                        {
523                                fields[i].set(struct, "");
524                                continue;
525                        }
526                        if (type.isArray())
527                        {
528                                fields[i].set(struct, Array.newInstance(type.getComponentType(), 0));
529                                continue;
530                        }
531                        if (!type.isPrimitive())
532                        {
533                                fields[i].set(struct, newStruct(type));
534                                continue;
535                        }
536                        if (int.class == type)
537                        {
538                                fields[i].setInt(struct, 0);
539                                continue;
540                        }
541                        if (char.class == type)
542                        {
543                                fields[i].setChar(struct, '\0');
544                                continue;
545                        }
546                        if (short.class == type)
547                        {
548                                fields[i].setShort(struct, (short)0);
549                                continue;
550                        }
551                        if (long.class == type)
552                        {
553                                fields[i].setLong(struct, 0L);
554                                continue;
555                        }
556                        if (boolean.class == type)
557                        {
558                                fields[i].setBoolean(struct, false);
559                                continue;
560                        }
561                        if (double.class == type)
562                        {
563                                fields[i].setDouble(struct, 0.0D);
564                                continue;
565                        }
566                        if (byte.class == type)
567                        {
568                                fields[i].setByte(struct, (byte)0);
569                                continue;
570                        }
571                        if (float.class == type)
572                        {
573                                fields[i].setFloat(struct, 0.0F);
574                                continue;
575                        }
576                }
577        }
578        catch (IllegalAccessException ex)
579        {
580                throw new IllegalArgumentException("Class " + struct.getClass().getName() + " doesn't seem to be an IDL-generated struct.");
581        }

645/**

647 * CORBA structures.  All public data members will be initialized.  If a given field
648 * is a primitive type, then set it to it's default setting (0, 0.0, or false); if
649 * it's a string, then the value "" is used; if it's another class, then the method

663{
664        StructCreator creator = (StructCreator)getPrototypes().get(structClass);
665        if (creator != null)
666        {
667                return (T)creator.create();

671        {

675                {
676                        initField(fields[i], obj);

679        }
680        catch (IllegalAccessException ex)
681        {
682                throw new IllegalArgumentException("Class " + structClass.getName() + " doesn't seem to be an IDL-generated struct.");
683        }
684        catch (InstantiationException ex)
685        {
686                throw new IllegalArgumentException("Class " + structClass.getName() + " doesn't seem to be an IDL-generated struct.");
687        }
