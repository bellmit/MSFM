371                    ":" + order.getOrderId().correspondentFirm);
372            // assume not marketable if no valid BBO found.
373            return false;
374        }
375 
376        Price orderPrice = order.getPrice();
377        if (orderPrice.isMarketPrice()) {
378            return true;
379        }
380 
381        Price bboBid = PriceFactory.create(theBBOExchangeMarket.bestBidPrice);
382        Price bboAsk = PriceFactory.create(theBBOExchangeMarket.bestAskPrice);
383 
384        if (orderPrice.isValuedPrice()) {
385            if ( bboAsk.isValuedPrice() &&
386                    order.getSide().isBuySide() &&
387                    orderPrice.greaterThanOrEqual(bboAsk) ) {
388                result = true;
389            } else if ( bboBid.isValuedPrice() &&
390                    order.getSide().isSellSide() &&
391                    orderPrice.lessThanOrEqual(bboBid) ) {
392                result = true;
393            }
394        }

466        {
467            boolean batchProcessingStopped = false;
468 
469            boolean success = false;

487                            asynchronousStartMonitoringForOrder(anEntry.order, anEntry.nbbo);

489                        }
490                    }
491                    catch (Exception e)
492                    {
493                        Log.exception("OrderMarketMonitoring insert failed. Processing will continue with next entry", e);
494                        dequeueEx = e;
495                    }
496                    finally
497                    {
498                        if(!success)
499                        {
500                            Transaction.rollback();
501                        }

503                }
504            }

509            finally
