541    throws DataValidationException
542    {
543        int remainingQuantity = getLeavesQuantity();
544        int quantityCancelled;
545        if ( remainingQuantity < quantity ){
546                Log.alarm(this, "Order " + getOrderIdString() + " got a cancel request for " + quantity + " contracts; it only has " + remainingQuantity + " available.");

548        }
549        else {
550            quantityCancelled = quantity;
551        }
552        setCancelledQuantity(getCancelledQuantity() + quantityCancelled);
553        CancelReportStruct report = new CancelReportStruct();
554        report.cancelReportType = ReportTypes.STRATEGY_LEG_REPORT;
555        report.productKey = getProductKey();
556        report.cancelledQuantity = quantityCancelled;
557        // TLC quantity is limited by number of contracts traded
558        report.tlcQuantity = Math.min(quantity - quantityCancelled, getTradedQuantity());
559        report.totalCancelledQuantity = getCancelledQuantity();
560        report.userAssignedCancelId = userAssignedCancelId;
561        return report;
