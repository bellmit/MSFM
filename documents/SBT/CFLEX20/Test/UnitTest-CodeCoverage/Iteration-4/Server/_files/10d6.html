485     *                   bucket array) is emitted (int), followed  by the
486     *                   <i>size</i> of the IntHashtable (the number of key-value
487     *                   mappings), followed by the key (Object) and value (Object)

489     *                   The key-value mappings are emitted in no particular order.

491    private synchronized void writeObject(java.io.ObjectOutputStream s)
492        throws IOException
493    {
494        // Write out the length, threshold, loadfactor
495        s.defaultWriteObject();
496 
497        // Write out length, count of elements, and then the key/value objects
498        s.writeInt(table.length);
499        s.writeInt(count);
500        for (int index = table.length-1; index >= 0; index--) {
501            Entry entry = table[index];
502 
503            while (entry != null) {
504                s.writeInt(entry.key);
505                s.writeObject(entry.value);
506                entry = entry.next;
507            }
508        }

512     * Reconstitute the IntHashtable from a stream (i.e., deserialize it).
513     */
514    private synchronized void readObject(java.io.ObjectInputStream s)
515         throws IOException, ClassNotFoundException
516    {
517        // Read in the length, threshold, and loadfactor
518        s.defaultReadObject();
519 
520        // Read the original length of the array, number of elements, and the elements themselves
521        int origlength = s.readInt();
522        int elements = s.readInt();
523 
524        // Compute new size with a bit of room 5% to grow but
525        // No larger than the original size.  Make the length
526        // odd if it's large enough, this helps distribute the entries.
527        // Guard against the length ending up zero, that's not valid.
528        int length = (int)(elements * loadFactor) + (elements / 20) + 3;
529        if (length > elements && (length & 1) == 0)
530            length--;
531        if (origlength > 0 && length > origlength)
532            length = origlength;
533 
534        table = new Entry[length];
535        count = 0;
536 
537        // Read the number of elements and then all the key/value objects
538        for (; elements > 0; elements--) {
539            int key = s.readInt();
540            Object value = s.readObject();
541            put(key, value);
542        }
543    }
