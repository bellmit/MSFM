400     * This method will return a new array of Price which are nearest prices on 
401     * tick. 
402     * 
403     * It should be pointed out that all prices here are signed, with negative 
404     * ones representing buy leg prices, and positive ones representing sell
405     * leg prices.
406     */
407    private static void checkTick(
408                    final long[] legPrices,
409                    final CalculatorArguments calArgs)
410    throws SpreadProcessingException
411    {
412        for (int i=0; i<calArgs.spreadInfo.legs.length; i++)
413        {
414            int expectedSign = calArgs.spreadInfo.legs[i].legSide.isBuySide() ? -1 : +1;
415 
416            if (checkPriceLongEpsilon(legPrices[i]))
417            {
418                    legPrices[i] = calArgs.tickScale.getTick(0L, expectedSign);
419            }
420            else
421            {
422                    int sign = legPrices[i]>0 ? +1 : -1;
423                    if (sign != expectedSign)
424                    {
425                            dumpData(legPrices);
426                            throw new SpreadProcessingException("Failed to calculate base prices with the right sign", SpreadProcessingException.LegPricesCanNotBeCalculated);
427                    }
428                    
429                    if (!isOnTick(calArgs.tickScale, legPrices[i]))
430                    {
431                            legPrices[i] = calArgs.tickScale.getNearestPriceOnTickAsLong(legPrices[i]);
432                            if (legPrices[i] == 0L)
433                                    legPrices[i] = calArgs.tickScale.getTick(0L, expectedSign) * expectedSign;
434                    }
435            }
436        }
437    }

472                    final long[] legPricesOnTick,
473                    final CalculatorArguments calArgs)
474    {
475            StrategyLegTradePrice.TradePriceMarketRelation mktRelation = null;
476        boolean allPass = true;

480                boolean allLegMktTickWideCBOE = true;

489                    if (mktRelation.tickWideAndTouchMarket)
490                {
491                        // for tick wide leg market, the leg price is allowed to touch
492                        //         the option leg market or
493                    //         the equity leg market if CBSX is not NBBO
494                        if (calArgs.spreadInfo.legs[legIndex].isEquityLeg &&
495                        ((mktRelation.adjustDirection > 0 && calArgs.isCBOEBidNBBO[legIndex]) ||
496                         (mktRelation.adjustDirection < 0 && calArgs.isCBOEAskNBBO[legIndex])))
497                        {
498                        allPass = false;
499                        }
500                }
501                    else
502                    {
503                            allLegMktTickWideCBOE = false;
504                    }
505            }
506            if (noTouch==0 && !allLegMktTickWideCBOE)
507            {
508                    allPass = false;
509            }
510            calArgs.allLegMktTickWideCBOE.put(sessionName, allLegMktTickWideCBOE);
511        }
512        return allPass;
513    }

550    /**
551     * delta = adjustedSpreadPrice - targetSpreadTradePrice
552     * 
553     * Get the leg price adjustment direction so that when the leg
554     * price is adjusted by one tick, the |delta| is reduced.
555     *
556     * Note: the returned values have the following meaning:
557     *       +1: leg price + one tick
558     *       -1: leg price - one tick
559     *       
560     * The following is the decision table
561     *
562     *      for delta > 0: return +1; 
563     *      for delta < 0: return -1;
564     *      for delta == 0: return 0; -- when adjusting is to avoid touch book violation
565     */
566    private static int getAdjustmentDirection(long delta)
567    {
568        return delta>0 ? +1 : (delta<0 ? -1 : 0);
569    }

571    /**

573     * a buy, and positive indicates a sell. 
574     * 
575     * But in the final trade report, a price can not be signed. So we need to

581    {

583            return PriceFactory.create(-1 * legPrice); //flip the sign

590    /**
591     * Return the Price's relation to the market.
592     * 
593     * IMPORTANT: the legBidMarket, and legAskMarket are "normalized" prices in 
594     * the sense that NoPrice has been replaced by zero prices.
595     */
596    private static StrategyLegTradePrice.TradePriceMarketRelation getPriceMarketRelation(
597                    final long legPriceAsLong,
598                    int legIndex,
599                    final CalculatorArguments calArgs)
600    {
601            Price legPrice = getNormalizedPrice(legPriceAsLong, calArgs.spreadInfo.legs[legIndex]);
602            if (calArgs.legAskMarkets==null || calArgs.isCBOEBidNBBO==null)
603                    return StrategyLegTradePrice.TradePriceMarketRelation.NOT_APPLY;
604            
605            StrategyLegTradePrice.TradePriceMarketRelation relation = null;
606            if (calArgs.legAskMarkets[legIndex].isNoPrice() || checkPriceLongEpsilon(calArgs.legAskMarkets[legIndex].toLong()))
607            {
608                    //TODO: the logic is for without DQ only
609                    if (calArgs.derivedQuote!=null || (calArgs.legBidMarkets[legIndex].isNoPrice() || checkPriceLongEpsilon(calArgs.legBidMarkets[legIndex].toLong())))
610                        return StrategyLegTradePrice.TradePriceMarketRelation.INCOMPLETE_MARKET;
611                            
612                    if (legPrice.lessThan(calArgs.legBidMarkets[legIndex])) {
613                            relation = StrategyLegTradePrice.TradePriceMarketRelation.OUTSIDE_BB; //outside the market
614                    }
615                    else if (legPrice.greaterThan(calArgs.legBidMarkets[legIndex])) {
616                            relation = StrategyLegTradePrice.TradePriceMarketRelation.IN_NO_ASK;
617                }
618                    else if (legPrice.equals(calArgs.legBidMarkets[legIndex])) {
619                            relation = StrategyLegTradePrice.TradePriceMarketRelation.TOUCH_CBOE_BID_NO_ASK;
620                    }
621            }
622            else
623            {
624                    if (calArgs.legBidMarkets[legIndex].greaterThan(calArgs.legAskMarkets[legIndex])) {
625                            relation = StrategyLegTradePrice.TradePriceMarketRelation.MARKET_INVERTED;
626                    }
627                    else if (legPrice.lessThan(calArgs.legBidMarkets[legIndex])) {
628                            relation = StrategyLegTradePrice.TradePriceMarketRelation.OUTSIDE_BB; //outside the market
629                    }
630                    else if (legPrice.greaterThan(calArgs.legAskMarkets[legIndex])) {
631                            relation = StrategyLegTradePrice.TradePriceMarketRelation.OUTSIDE_BO; //outside the market
632                    }
633                    else if (legPrice.greaterThan(calArgs.legBidMarkets[legIndex]) && legPrice.lessThan(calArgs.legAskMarkets[legIndex])){
634                            relation = StrategyLegTradePrice.TradePriceMarketRelation.IN_BETWEEN; //inside the market
635                    }
636                    else {
637                            if (legPrice.equals(calArgs.legBidMarkets[legIndex]) && calArgs.isCBOEBidNBBO[legIndex]) {
638                                    relation = StrategyLegTradePrice.TradePriceMarketRelation.TOUCH_CBOE_BID_WIDER_THAN_TICK; //touches the market.
639                            }
640                            if (legPrice.equals(calArgs.legAskMarkets[legIndex]) && calArgs.isCBOEAskNBBO[legIndex]) {
641                                    if (relation==null)
642                                            relation = StrategyLegTradePrice.TradePriceMarketRelation.TOUCH_CBOE_ASK_WIDER_THAN_TICK;
643                                    else
644                                            relation = StrategyLegTradePrice.TradePriceMarketRelation.TOUCH_CBOE_BOTH;
645                            }
646                            if (relation==null)
647                            {
648                                    if (calArgs.legBidMarkets[legIndex].equals(calArgs.legAskMarkets[legIndex]))
649                                            relation = StrategyLegTradePrice.TradePriceMarketRelation.LOCK_AT_LOCKED_MARKET;
650                                    else
651                                            relation = StrategyLegTradePrice.TradePriceMarketRelation.IN_BETWEEN;
652                            }
653                            if (relation.touchMarket)
654                            {
655                                    if (calArgs.isCBOEBidNBBO[legIndex] && calArgs.isCBOEAskNBBO[legIndex])
656                                    {
657                                            long mktWidth = calArgs.legAskMarkets[legIndex].toLong() - calArgs.legBidMarkets[legIndex].toLong();
658                                            if (mktWidth <= calArgs.tickScale.getTick(legPrice, relation.adjustDirection).toLong()) {
659                                                    switch (relation)
660                                                    {
661                                                    case TOUCH_CBOE_ASK_WIDER_THAN_TICK:
662                                                            relation = StrategyLegTradePrice.TradePriceMarketRelation.TOUCH_CBOE_ASK_TICK_WIDE;
663                                                            break;
664                                                    case TOUCH_CBOE_BID_WIDER_THAN_TICK:
665                                                            relation = StrategyLegTradePrice.TradePriceMarketRelation.TOUCH_CBOE_BID_TICK_WIDE;
666                                                            break;
667                                                    }
668                                            }
669                                    }
670                            }
671                    }
672            }
673        return relation;
674    }
