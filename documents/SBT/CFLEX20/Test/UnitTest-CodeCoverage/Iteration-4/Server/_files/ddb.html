570     * 
571     * A comparator to sort the price details on time basis. This sorting is required because in some cases
572     *  like the HAL auction, the tradable list is not sorted in time.
573     *  
574     */
575    public final Comparator<OrderBookPriceDetail> TIME_PRIORITY = new java.util.Comparator<OrderBookPriceDetail> ()
576    {
577        public int compare(OrderBookPriceDetail firstPriceDetail,OrderBookPriceDetail secondPriceDetail)
578                {
579           long receivedtime1 = 0;
580           long receivedtime2 = 0;
581           Tradable tradable1 = firstPriceDetail.getTradable();
582           Tradable tradable2 = secondPriceDetail.getTradable();
583           receivedtime1 = tradable1.isAuctionResponse()?tradable1.getReceivedTimeForAuctionResponse():getBookedTimeForTradable(tradable1);
584           receivedtime2 = tradable2.isAuctionResponse()?tradable2.getReceivedTimeForAuctionResponse():getBookedTimeForTradable(tradable2);
585                return (receivedtime1 < receivedtime2 ? -1:
586                        (receivedtime1 == receivedtime2) ? 0 : 1);
587                }
588    };

595            }
596            catch(Exception e)
597            {
598                    Log.exception("Exception while sorting the tradables by Time priority.", e);
599            }
