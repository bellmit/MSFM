479    private ArrayList<RelatedTradeReportSummary> organizeTradeReports(TradeReportSummary[] summaries, boolean activeOnly) throws SystemException, TransactionFailedException

483        // Iterate through the given result set, for each summary in the result set, get the parent and the
484        // children.  For each summary then, there could be 3 possibilities:
485        //      1) the summary has neither parent or children
486        //      2) the summary has a parent but no children
487        //      3) the summary has no parent but has children

489        //  Here we have a small number of possibilities:
490        //       1) the parent of a strategy precedes the children
491        //       2) a child or children could precede the parent
492        //  To account for this situation and to mark records that were pulled because of their family relationship an
493        //  index of search results is created and checked when adding elements to a family.

514                    relations.put(currentReport.getTradeId(), rtrs);

517                else if( parent == null  && children != null && children.length > 0 )

520                    if( ! relations.containsKey(currentReport.getTradeId()) )
