655    public long failover(boolean skipMaster, boolean p_useSinglePhase, boolean p_restartKilledProcesses, boolean p_asynchRestart, boolean p_prepareSlave, boolean p_disableMaster) throws Exception{

694        if(p_restartKilledProcesses){
695            startSlaveAfterFailover(p_asynchRestart);
696        }
697        swapMasterAndSlave();
698        return (p2 - p1);
