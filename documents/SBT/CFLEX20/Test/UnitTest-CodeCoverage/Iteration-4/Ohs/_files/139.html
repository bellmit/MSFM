416    private char marketMakerClearing(ManualFillReportContext p_context, TradeReportStruct tradeReport, AtomicTradeStruct p_atomicTrade, Order p_order, char p_side)
417    {
418        String acronym = (p_side == Sides.BUY) ? p_atomicTrade.sellerBroker.acronym : p_atomicTrade.buyerBroker.acronym;
419        boolean mmtnRequired = isMMTNRequired(p_atomicTrade, p_side);
420        
421        //  If the firm is expecting to recieve an MMTN, then we must check to see if they 
422        // are currently logged in.  If they are not, we cannot clear the trade.
423        if (mmtnRequired)
424        {
425            int classKey = getClassKeyTrade(p_order, tradeReport);
426            String cmiUser = getMappedCmiUser(p_atomicTrade, p_order, p_side, classKey);
427            if (cmiUser != null)
428            {
429                if(isUserLoggedIn(cmiUser))
430                {
431                    p_context.setOverridePublishing(acronym, false);
432                    return PAR_TO_MARKET_MAKER_CLEARING;  // clear with the descression of the property.
433                }
434                else
435                {
436                    p_context.setOverridePublishing(acronym, true);
437                    return USER_NOT_LOGGED_IN_CLEARING; // overwrite whats in the property and do not clear the trade.
438                }
439            }             
440            //  else: we can still clear the trades, but the MMTN should end up at the OMT.
441        }
442       
443        // No one is expecting to receive the mmtn, but we would still clear
444        // the trade based off the preferences.
445        p_context.setOverridePublishing(acronym, false);
446        return PAR_TO_MARKET_MAKER_CLEARING;
447    }
