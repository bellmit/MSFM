875     * @param qtyToCancel
876     * 
877     * Generate Cancel Report if there are too many routes

880                                    int requestedCancelQty, int qtyToCancel)

885        try {

887            
888            LegOrderDetail[] legOrderDetails = order.getLegOrderDetails();
889            if(legOrderDetails != null) legCounts = legOrderDetails.length;
890            
891            if (cancelRequest.userAssignedCancelId.equals("Help Desk"))

894                cancelReason = ActivityReasons.SYSTEM;
895            }

899                cancelReason = ActivityReasons.USER;

902            cancelReport.productKey = order.getProductKey();

909            cancelReport.cancelReportType = ReportTypes.CANCEL_ORDER_REQUEST_REJECT;
910            cancelReport.orderId = order.getOrderId();
911            cancelReport.orsId =  order.getOrsId();
912            cancelReport.sessionName = order.getActiveSession();
913            cancelReport.totalCancelledQuantity = 0;
914            cancelReport.transactionSequenceNumber = order.getTransactionSequenceNumber();
915            cancels.add(cancelReport);       
916            for(int index = 0; index < legCounts; index++)
917            {            
918                cancelReport = new CancelReportStruct();

923                cancelReport.tlcQuantity = requestedCancelQty * legRatio;
924                cancelReport.mismatchedQuantity = 0;
925                cancelReport.userAssignedCancelId = cancelId;
926                cancelReport.timeSent = DateWrapper.convertToDateTime(new java.util.Date().getTime());
927                cancelReport.cancelReportType = ReportTypes.STRATEGY_LEG_REPORT;
928                cancelReport.orderId = order.getOrderId();
929                cancelReport.orsId =  order.getOrsId();
930                cancelReport.sessionName = order.getActiveSession();
931                cancelReport.totalCancelledQuantity = 0;
932                cancelReport.transactionSequenceNumber = order.getTransactionSequenceNumber();
933                cancels.add(cancelReport);       
934            }

939            }

941                    orderHistoryCommonInfo);

948                                    cancelRequest.orderId).add(dve));
949
