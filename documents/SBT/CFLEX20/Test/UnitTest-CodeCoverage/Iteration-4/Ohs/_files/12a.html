611    protected void createLegFillRejectOrTimeoutHistories(BaseOHSContext p_context, short activityType, 
612            int[] legTradeQuantities, PriceStruct[] legExecutionPrices,
613            short ohsRoutingReason, String ohsRouteDesc, String p_orderHandlingString, 
614            boolean isFailedRoute, OrderHistoryCommonInfo orderHistoryCommonInfo)
615    {
616        if(!p_context.getOrder().isStrategy())
617            return;
618        
619        short legActivityType = ManualActivityHelper.getManualLegActivityType(activityType);
620        LegOrderDetail[] legs = p_context.getOrder().getLegOrderDetails();
621        Price defaultPrice = PriceFactory.getNoPrice();

636                                    legActivityType == ActivityTypes.MANUAL_FILL_REJECT_STRATEGY_LEG)
