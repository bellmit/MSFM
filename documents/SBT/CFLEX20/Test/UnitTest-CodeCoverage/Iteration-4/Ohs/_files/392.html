421        List listOfHomes = Arrays.asList(homeProperties.getProperty("listOfHomes", "").split(
422                "\\w+,\\w+"));
423        Properties homeImpls = new Properties();
424        Iterator propIter = homeProperties.keySet().iterator();
425        while (propIter.hasNext())
426        {
427            String key = (String) propIter.next();
428            if (key.startsWith("home."))
429            {
430                String homeName = key.substring("home.".length());
431                try
432                {
433                    Class implClass = Class.forName(homeProperties.getProperty(key));
434                    homeImpls.put(homeName, implClass);
435                }
436                catch (ClassNotFoundException ex)
437                {
438                    throw new RuntimeException("Impl for home " + homeName + ": " + ex);
439                }
440            }
441        }
442 
443        // for each home in listOfHomes:
444        for (int i = 0; i < listOfHomes.size(); i++)
445        {
446            String homeName = (String) listOfHomes.get(i);
447            if (homeName.length() == 0)
448            {
449                continue;
450            }
451            if (!homeImpls.containsKey(homeName))
452            {
453                throw new RuntimeException("home." + homeName + " property not found "
454                        + "for home in listOfHomes");
455            }
456            addHome(homeName, (Class) homeImpls.get(homeName), homeProperties, homeName + ".", true);
457        }
458 
459        // for each home NOT in listOfHomes:
460        Iterator nameIter = homeImpls.keySet().iterator();
461        while (nameIter.hasNext())
462        {
463            String homeName = (String) nameIter.next();
464            if (!listOfHomes.contains(homeName))
465            {
466                addHome(homeName, (Class) homeImpls.get(homeName), homeProperties, homeName + ".",
467                        true);
468            }
469        }
470    }
