347            // routed most recently to the specified location. When any single
348            // order is routed to the same location multiple times, there are
349            // multiple routed messages for order for the same location with different
350            // route times.
351            HashSet<Long> includedOrders = new HashSet<Long>();
352            for (RoutedMessage anOrderMsg : results)
353            {
354                // The query result is sorted by last routed time
355                // in descending order. So take the first encounter
356                // of a routed message for an order in the query result
357                // to generate a collection of routed message unique by
358                // subject order based on the most recent route time.
359                long orderDbId = anOrderMsg.getOrderDbId();
360                if (includedOrders.add(orderDbId))
361                {
362                    filteredMsgList.add(anOrderMsg);
363                }                
364            }
365        }
366        catch (PersistenceException pe)
367        {
368            log().alarm( msg().add("Error retrieving order routed messages for location:")
369                    .add(p_location).add(pe, false));
370        }    
371        return filteredMsgList;
372    }
