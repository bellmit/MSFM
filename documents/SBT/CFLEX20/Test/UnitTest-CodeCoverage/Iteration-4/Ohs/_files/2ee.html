476    private void handleWildCardANYCode(IWorkflowEngine<BaseOHSContext> reRouteOrderWorkflow, String[] configuredFunctions, String[] sessions, int g, int d, String exceptionType, String exceptionCodeString, String exceptionTypeTrimmed)
