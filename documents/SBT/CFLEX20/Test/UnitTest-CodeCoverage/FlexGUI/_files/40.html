462    public JasperPrint getFormattedReport()                     // TODO add parameter: String styleName
463    {
464        if (reportParameters == null)
465        {
466            // for backward compatibility, any report that has not been ported to produce a parameter map, will be formatted
467            // using a simple report template which has a single String as its body.
468            reportParameters = new HashMap<String, Object>();
469            reportParameters.put(REPORT_PARAM_TEMPLATE_NAME , REPORT_TEMPLATE_BASIC_NEW);
470            reportParameters.put(REPORT_PARAM_STRING_CONTENT, this.toString());
471            GUILoggerHome.find().debug("PrintReport.getFormattedReport: using backward compatibility template", null);
472        }
473 
474        // TODO caller should get styleName from user and pass in: String templateName = (String) reportStyles.get(styleName);
475        String templateName = (String) reportParameters.get(REPORT_PARAM_TEMPLATE_NAME);
476        GUILoggerHome.find().debug("PrintReport.getFormattedReport: templateName=" + templateName + ", map sipze=" + reportParameters.size(),
477                                   null);
478 
479        try
480        {
481            // note that report is not cached, because it can be large and we don't want to keep one per styleName. caller can cache if needed.
482            JasperReport report = loadReport(templateName);
483 
484            /*
485             * print out what we send to the jasper report, we recurse down the structure so
486             * that sub-template information is also printed. This information is what will
487             * be turned into the print text
488             */
489            if(GUILoggerHome.find().isDebugOn()) {
490                StringBuilder output = new StringBuilder();
491                output.append("PrintReport.getFormattedReport():\n");
492                output.append("\tPre-Jasper Call Stack:\n");
493                for(StackTraceElement str : new Exception().getStackTrace()) {
494                    output.append(String.format("\t\t%s\n", str.toString()));
495                }
496                output.append("\tJasper Parameter Map:\n");
497                debugPrintMapUtil(reportParameters, output,"\t\t");
498                GUILoggerHome.find().debug(output.toString(),null);
499            }
500            
501 
502            return JasperFillManager.fillReport(report, reportParameters, new JREmptyDataSource());
503        }
504        catch (JRException e)
505        {
506            DefaultExceptionHandlerHome.find().process(e, "Report creation error. Template=" + templateName);
507        }
508 
509        return null;
510    }
511 
512    /*
513     * Recurse through the map, printing the values as we go down. This method is used so we can
514     * identify the values that are going to be placed into the jasper report
515     */
516    private void debugPrintMapUtil(Map map, StringBuilder builder, String lineIndent) {
517        // this will cause an exception if concurrently modified. This is a huge big deal
518        // if it happens
519        try {
520            
521            Iterator<Entry<?, ?>> keyIterator = map.entrySet().iterator();
522            while (keyIterator.hasNext()) {
523                Entry<?, ?> entry = keyIterator.next();
524                if (entry.getValue() instanceof Map) {
525                    builder.append(String.format("%s%s=SubMap:\n", lineIndent, entry.getKey()));
526                    debugPrintMapUtil((Map) entry.getValue(), builder, lineIndent + "\t");
527                } else if (entry.getValue() instanceof Collection) {
528                    // contra reports are for some reason in this collection
529                    for (Object obj : (Collection) entry.getValue()) {
530                        if (obj instanceof Map) {
531                            builder.append(String.format("%s%s=<collection>.SubMap:\n", lineIndent, entry.getKey()));
532                            debugPrintMapUtil((Map) obj, builder, lineIndent + "\t");
533                        } else {
534                            builder.append(String.format("%s%s=%s\n", lineIndent, entry.getKey(), entry.getValue()));
535                        }
536                    }
537                } else if (entry.getValue() instanceof JasperReport) {
538                    JasperReport report = (JasperReport)entry.getValue();
539                    builder.append(String.format("%s%s=JasperReport(name=%s)\n", lineIndent, entry.getKey(), report.getName()));
540                } else {
541                    builder.append(String.format("%s%s=%s\n", lineIndent, entry.getKey(), entry.getValue()));
542                }
543 
544            }
545 
546        } catch (ConcurrentModificationException conE) {
547            GUILoggerHome.find().exception("Concurrent Modification related to printing! Map modified", conE);
548        } catch (Exception e) {
549            GUILoggerHome.find().exception(e);
550        }
551

568    public static JasperReport loadReport(String templateName)
569    {
570        JasperReport returnReport = null;
571        if (templateName != null  &&  templateName.length() > 0)
572        {
573            boolean cacheIt = false;
574            if (reportCache.containsKey(templateName))
575            {
576                returnReport = reportCache.get(templateName);
577                cacheIt = true;
578            }
579            if (returnReport == null)
580            {
581                try
582                {
583                    GUILoggerHome.find().debug("PrintReport.loadReport: templateName=" + templateName, null);
584 
585                    URL reportResource = Resources.getReportResource(templateName);
586                    if (reportResource != null)
587                    {
588                        returnReport = (JasperReport)JRLoader.loadObject(reportResource.openStream());
589                        if (cacheIt)
590                        {
591                            reportCache.put(templateName, returnReport);
592                        }
593                    }
594                    else
595                    {
596                        DefaultExceptionHandlerHome.find().process(new RuntimeException("Can't find report template=" + templateName));
597                    }
598                }
599                catch (IOException ioe)
600                {
601                    DefaultExceptionHandlerHome.find().process(ioe, "Can't find, open, or read report template=" + templateName);
602                }
603                catch (JRException e)
604                {
605                    DefaultExceptionHandlerHome.find().process(e, "Report load error. Template=" + templateName);
606                }
607            }
608        }
609        else
610        {
611            DefaultExceptionHandlerHome.find().process(new RuntimeException("No report template name specified to load"));
612        }
613 
614        return returnReport;
615    }
