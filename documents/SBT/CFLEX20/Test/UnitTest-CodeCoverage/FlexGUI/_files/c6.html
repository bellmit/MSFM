537                    public void handleException(Exception exp)
538                        {
539                                String errorMsg = "";
540                                if (exp instanceof DataValidationException)
541                                        errorMsg = ((DataValidationException) exp).details != null ? ((DataValidationException) exp).details.message : "";
542                                else if (exp instanceof CommunicationException)
543                                        errorMsg = ((CommunicationException) exp).details != null ? ((CommunicationException) exp).details.message : "";
544                                else if (exp instanceof AuthorizationException)
545                                        errorMsg = ((AuthorizationException) exp).details != null ? ((AuthorizationException) exp).details.message : "";
546                                else if (exp instanceof SystemException)
547                                        errorMsg = ((SystemException) exp).details != null ? ((SystemException) exp).details.message : "";
548                                else if (exp instanceof NotAcceptedException)
549                                        errorMsg = ((NotAcceptedException) exp).details != null ? ((NotAcceptedException) exp).details.message : "";
550                                else if (exp instanceof TransactionFailedException)
551                                        errorMsg = ((TransactionFailedException) exp).details != null ? ((TransactionFailedException) exp).details.message : "";
552                                else if (exp instanceof NotFoundException)
553                                        errorMsg = ((NotFoundException) exp).details != null ? ((NotFoundException) exp).details.message : "";
554                                else
555                                        errorMsg = exp.getMessage();
556                                
557                                DefaultExceptionHandlerHome.find().process(exp, "Unable to Enter Underlying Price - "+errorMsg);
558                                getStatusBar().setMessageText(SBTStatusBar.ERROR_MESSAGE, new TimeImpl().toString()+space+"Unable to Enter Underlying Price - "+errorMsg);
559                                cleanUpView();
560                        }
561            };
562            return worker;
563    }
