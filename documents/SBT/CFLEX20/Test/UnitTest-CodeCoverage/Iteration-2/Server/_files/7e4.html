574/**
575 * Recreates RFQs on failover.  Pending RFQs wil be lost when the system does
576 * a master/slave swith, so they need to be recreated.
577 */
578protected void recreateRFQs() {
579    Order[] marketOrders = getOrderHome().findActiveMarketOrders();
580    for (int i = 0; i < marketOrders.length; i++) {
581        // find first unprocessed market order
582        if (marketOrders[i] != null) {
583            Integer productKey = marketOrders[i].getProductKey();
584            int rfqQuantity = marketOrders[i].getQuantityAllowed();
585            marketOrders[i] = null;
586            // get total quantity for all market orders for product
587            for (int j = i + 1; j < marketOrders.length; j++) {
588                if(marketOrders[j] != null)
589                {
590                    if (productKey.equals(marketOrders[j].getProductKey())) {
591                        rfqQuantity += marketOrders[j].getQuantityAllowed();
592                        marketOrders[j] = null;
593                    }
594                }
595            }
596            // create RFQ for product if product is in trading state.
597            try {
598                TradingProduct product = getTradingProductHome().findByKey(productKey);
599                if (product.inTradingState()) {
600                    Broker broker = getBroker(productKey, "Unable to get broker to recreate pending RFQ");
601                    broker.createRFQ(productKey, rfqQuantity);
602                }
603            }
604            catch (Exception e) {
605                Log.exception(this, "Unable to recreate RFQ for product = " + productKey, e);
606            }
607        }
608    }

628    /**
629     * Processes notification of a completed state change event.
630     *
631     * @see StateChangeListener#stateChangeEvent
632     */
633    public void stateChangeEvent(TradingClass tradingClass, TradingProduct[] products)
634    {
635        /*
636         * We don't want to start timers for proxy objects, so ignore state change events for proxy objects.
637         */
638        if (tradingClass.isProxy())
639        {
640            return;
641        }
642        
643        TradingProduct product = null;
644        ArrayList preOpenProducts = new ArrayList();
645        ArrayList openRotationProducts = new ArrayList();
646        Broker targetBroker = null;
647        short newState;
648        for(int i=0; i<products.length; i++)
649        {
650            product = products[i];
651            if(product != null)
652            {
653                newState = product.getCurrentStateCode();
654                try
655                {
656                    targetBroker = getBrokerByClass(tradingClass.getProductClassKey(), "Product state change event received for an unknown class");
657                }
658                catch (DataValidationException e)
659                {
660                    Log.alarm(this, "Unable to begin ProductStates."+newState+" processing on products in class = " + tradingClass.getClassSymbol());
661                }
662                if (newState == ProductStates.OPENING_ROTATION && tradingClass.getProductType()!= ProductTypes.STRATEGY)
663                {
664                    openRotationProducts.add(product);
665                }
666                else if (newState == ProductStates.PRE_OPEN && tradingClass.getProductType()!= ProductTypes.STRATEGY)
667                {
668                    preOpenProducts.add(product);
669                }
670                else if (newState == ProductStates.ENDING_HOLD)
671                {
672                    targetBroker.acceptEndingHold(product);
673                }
674            }
675        }
676        if(preOpenProducts.size()>0)
677        {
678            targetBroker.acceptPreOpen(preOpenProducts);
679        }
680        if(openRotationProducts.size()>0)
681        {
682            targetBroker.acceptOpeningRotation(openRotationProducts);
683        }
684    }

878    private void cancelPAndPAOrders()
879    {
880        Order[] orders = getOrderHome().findActivePAOrders();
881 
882        Log.information(this, "Cancelling pending P and PA orders.. found:" + orders.length);
883        
884        int incoming = 0;
885        int outgoing = 0;
886        if(orders != null && orders.length > 0)

888            boolean success = false;

890            Order anOrder;
891            Broker broker;

895            try
