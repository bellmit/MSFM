556private void updateClasses(TemplateClassStruct[] updatedClasses) throws DataValidationException, TransactionFailedException {
557    // remove current classes that are no longer in template
558    TemplateClass[] currentClasses = getClasses();
559    for (int i = 0; i < currentClasses.length; i++) {
560        boolean shouldRemove = true;
561        for (int j = 0; j < updatedClasses.length; j++) {
562            if (currentClasses[i].getClassKey() == updatedClasses[j].classStruct.classKey) {
563                shouldRemove = false;
564                break;
565            }
566        }
567        if (shouldRemove) {
568            removeClass(currentClasses[i].getClassKey());
569        }
570    }
571    // add new or update existing classes
572    for (int i = 0; i < updatedClasses.length; i++) {
573        TemplateClassImpl existingClass = getClassByKey(updatedClasses[i].classStruct.classKey);
574        if (existingClass != null) {
575            existingClass.update(updatedClasses[i]);
576        }
577        else {
578            addClass(updatedClasses[i]);
579        }
