361 * quantity will be added until its accumulated quantity exceeds the
362 * accumulated quantity of the other side.  At this point, the
363 * tradables at a price level from the other side are added.
364 * This continues until either all tradables on one side is depleted
365 * or a price equilibrium is met.  Price equilibrium is said to be
366 * met when the book is no longer crossed.
367 *
368 * Upon completion, the following instance variables will be updated
369 * to reflect the current balance point of the book:
370 *        <code>bestBuyPriceLevel,        openingBuySize,                previousBuySize</code>
371 *        <code>bestSellPriceLevel,        openingSellSize,        previousSellSize</code>
372 *
373 * @param buySideItem         buy side of this order book
374 * @param sellSideItem        sell side of this order book
375 */
376public void evalOpeningCondition() {
377        clear();
378        boolean priceEquilibrium = false;
379    OrderBookPriceItem newPriceLevel;
380    int cumulativeBuySize = 0;
381        int cumulativeSellSize = 0;
382        boolean doBuySide;
383        Enumeration buySideIterator = myOrderBookImpl.getBuySideItem().elements();
384        Enumeration sellSideIterator = myOrderBookImpl.getSellSideItem().elements();
385        if ( buySideIterator.hasMoreElements() ) {
386                doBuySide = true;        // start with buy side
387        } else {
388                doBuySide = false;        // else start with sell side
389        }
390        while (!buySideDepleted && !sellSideDepleted && !priceEquilibrium) {
391                if ( doBuySide ) {
392                        if ( buySideIterator.hasMoreElements() ) {
393                newPriceLevel = (OrderBookPriceItem) buySideIterator.nextElement();
394                priceEquilibrium = doBuySideAccumulation(newPriceLevel,cumulativeBuySize);
395                cumulativeBuySize = newPriceLevel.getCumulativeTotalQuantity();
396                        } else {
397                                buySideDepleted = true;
398                        }
399                        if ( openingBuySize >= openingSellSize ) {
400                                doBuySide = false; // switch to sell side if more quantity on buy side
401                        }
402                } else {
403                        if ( sellSideIterator.hasMoreElements() ) {
404                                newPriceLevel = (OrderBookPriceItem) sellSideIterator.nextElement();
405                                priceEquilibrium = doSellSideAccumulation(newPriceLevel,cumulativeSellSize);
406                cumulativeSellSize = newPriceLevel.getCumulativeTotalQuantity();
407                        } else {
408                                sellSideDepleted = true;
409                        }
410                        if ( openingSellSize >= openingBuySize ) {
411                                doBuySide = true; // switch to buy side if more quantity on sell side
412                        }
413                }
414        }
415 
416    // May have gone one price too far on buy or sell side.  Need to adjust prices so that openingBuyPrice
417    // and openingSellPrice will be range of valid Expected Opening Prices.
418    if ( openingBuySize > openingSellSize ) {
419        if ( previousBuySize >= openingSellSize && previousBuyPriceLevel != null) {
420            bestBuyPriceLevel = previousBuyPriceLevel;
421            openingBuySize = previousBuySize;
422        }
423    }
424    else if ( openingSellSize > openingBuySize ) {
425        if ( previousSellSize >= openingBuySize && previousSellPriceLevel != null) {
426            bestSellPriceLevel = previousSellPriceLevel;
427            openingSellSize = previousSellSize;
428        }
429    }
430 
431        setOpeningIndicators();
432}

434/**
435 * do buy side accumulation
436 * return true if price equilibrium is reached.
437 */
438private boolean doBuySideAccumulation(OrderBookPriceItem newPriceLevel, int preCumulativeBuySize)
439{
440    boolean priceEquilibrium = false;
441    int cumulativeBuySize = newPriceLevel.calculateCumulativeTotalQuantity(preCumulativeBuySize);
442    if ( ( bestSellPriceLevel != null ) &&
443         ( !bestSellPriceLevel.getItemPrice().isMarketPrice() ) &&
444         ( buySide.isFirstBetter( bestSellPriceLevel.getItemPrice(), newPriceLevel.getItemPrice() ) ) )
445    {
446        priceEquilibrium = true;
447    }
448    if ( !priceEquilibrium ) {
449        previousBuyPriceLevel = bestBuyPriceLevel;
450        bestBuyPriceLevel = newPriceLevel;
451        if ( newPriceLevel.getItemPrice().isMarketPrice() ) {
452            previousBuySize = cumulativeBuySize;
453        } else {
454            previousBuySize = openingBuySize;
455        }
456        openingBuySize = cumulativeBuySize;
457    }
458    return priceEquilibrium;
459}

476        bestSellPriceLevel = newPriceLevel;
477        if ( newPriceLevel.getItemPrice().isMarketPrice() ) {

479        } else {
480            previousSellSize = openingSellSize;
481        }
482        openingSellSize = cumulativeSellSize;
483    }
484    return priceEquilibrium;

611    private void checkBOTRRange()
612    {
613        // see if BOTR check is enabled or not
614        TradingProduct myTradingProduct = myOrderBookImpl.getTradingProduct();
615        TradingClass myTradingClass = myTradingProduct.getTradingClass();
616        if (myTradingClass.getTradingPropertyBOTREnabled() && myOrderBookImpl.isOpeningRequirementOn() )
617        {
618            RangeScale[] rangeScales= myTradingClass.getTradingPropertyBotrRangeScales();
619            if  (rangeScales.length == 0)
620            {
621                Log.alarm("Ignore to check BOTR range, because could NOT find BOTRRangeScales tradingProperty for class " + myTradingClass.getClassSymbol() + ", ClassKey:" + myTradingClass.getProductClassKey());
622                return;
623            }
624            double awayRange =  0.0;
625            boolean found = false;
626            for ( int i =0; i < rangeScales.length; i ++)
627            {
628                if (openingQuantity >= rangeScales[i].getLowerRange()
629                    && openingQuantity < rangeScales[i].getUpperRange())
630                {
631                    awayRange = Math.abs(rangeScales[i].getPercentage());
632                    found = true;
633                    break;
634                }
635            }
636            if (!found)
637            {
638                Log.alarm(" Use BOTR range directly, because could NOT find BOTRRange adjument percentage from its BOTRRangeScales value for class " + myTradingClass.getClassSymbol() + ", ClassKey:" + myTradingClass.getProductClassKey() + ", for openingQuantity =" + openingQuantity);
639            }
640            NBBOStruct botr = getBestOfTheRest(myTradingProduct);
641            Price botrBid = PriceFactory.create(botr.bidPrice);
642            Price botrAsk = PriceFactory.create(botr.askPrice);
643            if (botrBid.isValuedPrice() && botrAsk.isValuedPrice()
644                    && botrBid.toLong() > 0 && botrAsk.toLong() >0)
645            {
646                double lowRange = botrBid.toDouble()*(1 - awayRange);
647                double highRange = botrAsk.toDouble()*(1 + awayRange);
648                if (botrAsk.lessThanOrEqual(botrBid))
649                {
650                    lowRange = botrAsk.toDouble()*(1 - awayRange);
651                    highRange = botrBid.toDouble()*(1 + awayRange);
652                }
653                validateAgainstPriceRange(myTradingClass, lowRange, highRange);
654                if (Log.isDebugOn())
655                {
656                    Log.debug("Checking BOTR with Non-Zero Bid, Non-Zero Ask, for productKey " + myOrderBookImpl.getProductKey());
657                }
658 
659            }
660            else if (botrAsk.isValuedPrice() && botrAsk.toLong() >0)
661            {
662//              double lowRange = botrAsk.toDouble()*(1 - awayRange);
663                double lowRange = 0.0;
664                double highRange = botrAsk.toDouble()*(1 + awayRange);
665                validateAgainstPriceRange(myTradingClass, lowRange, highRange);
666                if (Log.isDebugOn())
667                {
668                    Log.debug("Checking BOTR with Non-Zero Ask, for productKey " + myOrderBookImpl.getProductKey());
669                }
670            }
671            else
672            {
673                if (Log.isDebugOn())
674                {
675                    Log.debug("Due to NON-SENSE BOTR, By Pass BOTR opening price validation for productKey " + myOrderBookImpl.getProductKey());
676                }
677            }
678            // else ignore the check, since BOTR has no bid & ask
679        }
680        else
681        {
682            if (Log.isDebugOn())
683            {
684                Log.debug("ByPass BOTR opening price validation for productKey " + myOrderBookImpl.getProductKey());
685            }
686        }
687    }
