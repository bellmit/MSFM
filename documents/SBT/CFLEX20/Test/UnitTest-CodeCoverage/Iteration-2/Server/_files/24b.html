808                if (timeLapFromLast > amiCallCountResetInterval)
809                {
810                    amiTimeoutCheck.reset();                   
811                }
812                amiTimeoutCheck.amiTimeoutCount++;
813                if (amiTimeoutCheck.amiTimeoutCount > amiCallMaxNumWithinInterval)
814                {
815                    //either we are not accepting cross product at all or this is an issue continuing from last check
816                    if (rejectCrossProduct) return;
817                    
818                    Log.alarm(this, caller + ": AMI call timeout exceeds max of "+amiCallMaxNumWithinInterval+" within "+timeLapFromLast+"(threshhold = "+amiCallCountResetInterval+") milliseconds; STOP ACCEPTING CROSS PRODUCT ORDERS for "+resumeCrossProductInterval+" milliseconds!!!");
819                    rejectCrossProduct = true;
820                    if (timerForAMITimeoutHandling == null)
821                    {
822                        timerForAMITimeoutHandling = new Timer();
823                    }
824                    timerForAMITimeoutHandling.schedule(new AMITimeoutTimerTask(caller), resumeCrossProductInterval);
825                }
826            }
827        }        
828    }
