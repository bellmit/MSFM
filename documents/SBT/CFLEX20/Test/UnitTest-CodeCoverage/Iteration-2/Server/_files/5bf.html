851    public CurrentMarketStruct combineManualQuoteMarket(CurrentMarketStruct aMarket, ManualQuoteBook manualBook, short volumeType)
852    {
853        ManualQuote manualQuote;
854        if(manualBook.bidSideManualQuoteExists())
855        {
856            Price marketBidPrice;
857            manualQuote = manualBook.getBidSideManualQuote();
858            Price manualBidPrice = manualQuote.getPrice();
859            //If manual bid price is greater than the market bid price then manual bid price will be used.
860            //If both price are same size will be aggregated.
861            if(aMarket.bidPrice.type != PriceTypes.NO_PRICE)
862            {
863                marketBidPrice = new  ValuedPrice(aMarket.bidPrice);
864 
865                if( manualQuote.getSide().isFirstBetter(manualBidPrice, marketBidPrice))
866                {
867                    //Manual quote is at the top.
868                    aMarket.bidPrice = manualBidPrice.toStruct();
869 
870                    aMarket.bidSizeSequence = manualBook.toBidMarketVolumeSequence(volumeType);
871                }
872                else if(manualQuote.getSide().isFirstBetterOrEqual(manualBidPrice, marketBidPrice) )
873                {
874                    //prices are equal so aggregate them
875                    aMarket.bidSizeSequence = addManualQuoteVolumeToSequence(aMarket.bidSizeSequence, manualQuote, volumeType);
876                }
877                else
878                {
879                    Log.alarm("OrderBookEditor >>> Manual quote is at a price worse than the book. Should never happen. ProdKey/MQPrice:" + 
880                            manualQuote.getProductKey() + "/" + manualQuote.getPrice().getWhole() + "." + manualQuote.getPrice().getFraction() );
881                }
882            }
883            else
884            {
885                //market Price is null
886                aMarket.bidPrice = manualBidPrice.toStruct();
887 
888                aMarket.bidSizeSequence = manualBook.toBidMarketVolumeSequence(volumeType);
889            }
890        }
891 
892        if(manualBook.askSideManualQuoteExists())
893        {
894            Price marketAskPrice;
895            manualQuote = manualBook.getAskSideManualQuote();
896            Price manualAskPrice = manualQuote.getPrice();
897            //If manual bid price is greater than the market bid price then manual bid price will be used.
898            //If both price are same size will be aggregated.
899            if(aMarket.askPrice.type != PriceTypes.NO_PRICE)
900            {
901                marketAskPrice = new  ValuedPrice(aMarket.askPrice);
902 
903                if( manualQuote.getSide().isFirstBetter(manualAskPrice, marketAskPrice))
904                {
905                    //use the manual book bid price
906                    aMarket.askPrice = manualAskPrice.toStruct();
907 
908                    aMarket.askSizeSequence = manualBook.toAskMarketVolumeSequence(volumeType);
909                }
910                else if(manualQuote.getSide().isFirstBetterOrEqual(manualAskPrice, marketAskPrice))
911                {
912                    //prices are equal so aggregate them
913                    aMarket.askSizeSequence = addManualQuoteVolumeToSequence(aMarket.askSizeSequence, manualQuote, volumeType);
914                }
915                else
916                {
917                    Log.alarm("OrderBookEditor >>> Manual quote is at a price worse than the book. Should never happen. ProdKey/MQPrice:" + 
918                            manualQuote.getProductKey() + "/" + manualQuote.getPrice().getWhole() + "." + manualQuote.getPrice().getFraction() );
919                }
920            }
921            else
922            {
923                //market Price is null
924                aMarket.askPrice = manualAskPrice.toStruct();
925 
926                aMarket.askSizeSequence = manualBook.toAskMarketVolumeSequence(volumeType);
927            }
928        }
929        
930        return aMarket;
931    }
