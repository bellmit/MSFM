304        public void run()
305        {
306        int count = 0;
307        for (int i = 0; i < itsTradingClasses.length; i++) {
308            if (i % itsNumberBuildThreads == itsThreadNumber) {
309                if(count%10 == 0){
310                    Log.information(this.getName() + ">>Building market data in progress... Number of classes completed: " + count);
311                }

313                count = count + 1;
314            }
315        }
316        Log.information(this.getName() + ">>Building market data completed. Total number of classes processed:" + count);

320         * Creates market data entries for all a given class's products.
321         */
322        private void createMarketDataForClass(TradingClass currentClass)
323        {
324                Transaction.startTransaction();
325        boolean success = false;
326            int createdCount = 0;
327            int alreadyExistsCount= 0;
328 
329        try {
330 
331            createMarketDataForUnderlyingProductOfClass(currentClass);
332 
333            if (Log.isDebugOn())
334            {
335                    Log.debug(MarketDataServiceHomeImpl.this, "Creating Market Data for " + currentClass.getClassSymbol());
336            }
337 
338            TradingProduct[] products = currentClass.getProducts(true /* enabled products only */);
339                    for (int j = 0; j < products.length; j++)
340                    {
341                            try
342                            {
343 
344                    Price theSettlementPrice = getSettlementPrice(products[j]);
345                    if (theSettlementPrice == null) {
346                                        itsMarketDataHome.create(itsSessionName, products[j].getProductKey(), currentClass.getProductClassKey());
347                    }
348                    else {
349                        itsMarketDataHome.create(itsSessionName, products[j].getProductKey(), currentClass.getProductClassKey(), theSettlementPrice);
350                    }
351                                    createdCount++;
352                            }
353                catch (AlreadyExistsException e)
354                            {
355                                    alreadyExistsCount++;
356                            }
357                catch (TransactionFailedException e)
358                            {
359                    Log.exception("Failed to create market data for product " +  products[j].getProductKey(), e);
360                            }
361                    }
362 
363                    success = Transaction.commit();
364        }
365        finally {
366            if (success) {
367                    if (Log.isDebugOn())
368                    {
369                        Log.debug(MarketDataServiceHomeImpl.this, "Created market data for " + currentClass.getClassSymbol() + ": created " + createdCount + " entries and " + alreadyExistsCount + " already existed");
370                }
371            }
372            else {
373                Transaction.rollback();
374                Log.alarm(MarketDataServiceHomeImpl.this, "Unable to create market data for " + currentClass.getClassSymbol());
375            }
376        }
377        }

496/**
497 *  Creates market data for all configured classes of the open session.
498 */
499protected void createMarketData()
500{
501    // the synchronization is done to partially solve a race condition during start up.  The product state
502    // changes generated putting products into a session may be incorrectly processed if they are handled
503    // while the market data entries are being created.  Some product state updates will fail because the
504    // state change will arrive before the market data for the product is created.
505    synchronized (getInternalService()){
506        String openSessionName = TradingSessionNameHelper.getOpenConfiguredSessionName();
507        // skip if no session is open
508        if (openSessionName == null) {
509            return;
510        }
511        TradingClass[] classes = getTradingClassHome().findAllConfiguredForSession(openSessionName);
512            if (classes == null || classes.length <= 0){
513                    Log.alarm(this, "Market Data service not created, no classes assigned to service ");
514            return;
515        }
516 
517        IntHashMap settlementPrices = getSettlementPrices( openSessionName, classes );
518 
519        int numberBuildThreads = numbOfBuildThreads;
520        MarketDataBuildThread[] threads = new MarketDataBuildThread[numberBuildThreads];
521        Log.information(this, "Creating market data for session " + openSessionName);
522        for (int threadNumber = 0; threadNumber < numberBuildThreads; threadNumber++) {
523            threads[threadNumber] = new MarketDataBuildThread(openSessionName, threadNumber, numberBuildThreads, classes, getMarketDataHome(), settlementPrices);
524            threads[threadNumber].start();
525        }
526            if (Log.isDebugOn())
527            {
528            Log.debug(this, "" + numberBuildThreads + " threads created to build market data, waiting for completion");
529        }
530        for (int i = 0; i < numberBuildThreads; i++) {
531            try {
532                threads[i].join();
533            }
534            catch (InterruptedException e) {
535                Log.alarm(this, "Interrupted while waiting for market data threads to complete - assuming creation completed");
536                return;
537            }
538        }
539        Log.information(this, "Market data creation complete for session " + openSessionName);
540     }
541}

543    /**
544     * Get settlement prices for all classes in session.
545     *
546     * @return map containing prices by product.  Map will be empty if session is not an internal session.
547     */
548    protected IntHashMap getSettlementPrices( String openSessionName, TradingClass[] classes ) {
549        IntHashMap settlementPrices = new IntHashMap();
550        // don't get prices for external sessions
551        if (TradingSessionNameHelper.isExternalSession(openSessionName)) {
552            return settlementPrices;
553        }
554        // get the settlement price for all products.
555        int classKeys[] = new int[classes.length];
556        for (int i = 0; i < classes.length; i++) {
557            classKeys[i] = classes[i].getProductClassKey();
558        }
559        try {
560            ClassSettlementStruct[] classSettlementStructs = getProductMaintenanceService().getSettlementForClasses(classKeys);
561            // calculate the size of the HashMap
562            int theSize = 0;
563            for (int i = 0; i < classSettlementStructs.length; i++) {
564                theSize += classSettlementStructs[i].settlements.length;
565            }
566            theSize = MathExtensions.nextPrime(theSize);
567            settlementPrices = new IntHashMap(theSize, (float) 0.5);
568            for (int i = 0; i < classSettlementStructs.length; i++) {
569                for (int j = 0; j < classSettlementStructs[i].settlements.length; j++) {
570                    settlementPrices.put(classSettlementStructs[i].settlements[j].productKeys.productKey,
571                        PriceFactory.create(classSettlementStructs[i].settlements[j].settlementPrice));
572                }
573            }
574        } catch (Exception e) {
575            Log.alarm(this, "Exception caught when getting the settlement price: " + e);
576            settlementPrices = new IntHashMap();
577        }
578        return settlementPrices;
579    }
