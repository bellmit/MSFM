480private boolean changeState(String pOaName, POAManager p_pm, String p_toState) throws AdapterInactive
481    {
482        if(p_toState.equalsIgnoreCase("ACTIVE")){
483          if(p_pm.get_state().value() == State._ACTIVE ){
484            Log.information("The " + pOaName + " POA is already ACTIVE");
485            return true;   
486          }
487          Log.information(this,"Changing the " + pOaName + " POA state to ACTIVE");
488          p_pm.activate();
489        }
490        else if(p_toState.equalsIgnoreCase("HOLDING")){
491            if(p_pm.get_state().value() == State._HOLDING ){
492                Log.information(this, "The " + pOaName + " POA is already HOLDING");
493                return true;   
494              }
495            Log.information(this, "Changing the " + pOaName + " POA state to HOLDING");
496              p_pm.hold_requests(false);
497            }
498        else if(p_toState.equalsIgnoreCase("DISCARDING")){
499            if(p_pm.get_state().value() == State._DISCARDING ){
500                Log.information(this, "The " + pOaName + " POA is already DISCARDING");
501                return true;   
502              }
503            Log.information(this, "Changing the " + pOaName + " POA state to DISCARDING");
504              p_pm.discard_requests(false);
505            }
506        else{
507            Log.information(this, "Cannot change " + pOaName + " POA state to " + p_toState + ". This state is invalid");
508          return false; //  wrong state   
509        }
510        return true;
511    }
