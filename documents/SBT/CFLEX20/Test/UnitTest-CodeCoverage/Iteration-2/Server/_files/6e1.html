663        //1-2: prefix, 3:firm, 4:branch, 5:branchSequenceNumber

675        }

678        if (firmNumber.length() == 0)
679        {
680            String msg = "Empty firmNumber in firmOrder.OptionalData=" + firmOrderOptionalData;

689         * firmOrderOptionalData; Log.alarm(msg); throw
690         * ExceptionBuilder.dataValidationException(msg, DataValidationCodes.INVALID_ORDER_ID); }
691*/
692        String branchNum = (String)fieldVect.elementAt(4);
693 
694        orderId.executingOrGiveUpFirm.firmNumber = firmNumber;
695        //orderId.correspondentFirm = correspondentFirm;

808    /**
809     * Parse a string. It allows for empty tokens. Note than StringTokenizer does not allow empty
810     * token "A:RE:AAA" ==> [A, RE, AAA] "" ==> [] "AAA :BBB:CCC:DD" ==>[AAA, BBB, CCC, DD]
811     * " A:B::D" ==> [A, B, , D] ":B:C:D" ==> [, B, C, D] "A:B:C:D:" ==> [A, B, C, D] "A:B:C:D: "
812     * ==> [A, B, C, D, ] "A:B:C:::D:" ==> [A, B, C, , , D]
813     * 
814     * @param optionalData
815     * @param delimeter
816     * @return
817     */
818    private Vector tokenizeOptionalData(String optionalData, String delimeter)
819    {
820        Vector tokens = new Vector();
821        int dw = delimeter.length();
822        while (optionalData.length() > 0 && tokens.size() < FIRM_OPTIONAL_DATA_TOKEN_COUNT)
823        {
824            int pos = optionalData.indexOf(delimeter);
825            String token = "";
826            if (pos > 0)
827            {
828                token = optionalData.substring(0, pos);
829                optionalData = optionalData.substring(pos+dw, optionalData.length());
830            }
831            else if (pos == 0)
832            {
833                optionalData = optionalData.substring(pos + dw, optionalData.length());
834            }
835            else
836            {
837                token = optionalData;
838                optionalData = "";
839            }
840 
841            tokens.addElement(token.trim());
842        }
843        return tokens;
844    }
