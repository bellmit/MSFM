367    public void updateEvent(Object key, DistributedCacheValue<TeSynchPoint> updatedValue)
368    {
369        if (!(key instanceof Integer))
370        {
371            Log.alarm("DISHIUPDATEEVENT - Update event called with non-Integer key, returning without processing");
372            return;
373        }
374        Integer intKey = (Integer) key;
375        // ignore updates for objects other than the SynchPoint and all events in master mode
376        if (intKey.intValue() != DataIntegrityServiceTeImpl.DISTRIBUTEDCACHE_SYNCHPOINT_KEY_INTEGER .intValue() || this.isMasterMode)
377        {
378            Log.information("DISHIUPDATEEVENT - Update event called in mode[" + this.isMasterMode
379                    + "] and key[" + key + "], returning");
380            return;
381        }
382        Map<String, String> results = new HashMap<String, String>();
383        TeSynchPoint dcSynchPoint = updatedValue.getValue();
384        this.setLastSynchPointTimeStamp(dcSynchPoint.getLastCountAddedTime());
385        try
386        {
387            // read current count values on slave side
388            this.readCounts();
389            if (this.localSynchPoint.isInitializingSynchPoint() || dcSynchPoint.isInitializingSynchPoint())
390            {
391                // set offsets before comparison
392                Log.information("DISHIUPDATEEVENT - Initializing offsets with dc synch point number[" + dcSynchPoint.getSynchPointCounter() + "]");
393                // use master synchpoint to set offsets for slave side event counts
394                this.setEventCountOffsets(dcSynchPoint);
395                Log.information("DISHIUPDATEEVENT - slave side SynchPoint[" + this.localSynchPoint.toString() + "]");
396                Log.information("DISHIUPDATEEVENT - master side SynchPoint[" + dcSynchPoint.toString() + "]");
397            }
398            // increment slave side synchPoint counter
399            increment();
400            this.localSynchPoint.setCounter(synchPointSequenceNumber);
401        }
402        catch (DataValidationException e)
403        {
404            String msg = "DISHI - Caught " + e.getClass().getName() + " while trying to retrieve counts for local server, unable to compare to primary server counts";
405            Log.exception(this, msg, e);
406            Log.alarm(this, msg);
407        }
408        // Compare distributed cache SynchPoint to local SynchPoint
409        try
410        {
411            results = compareSynchPoints(dcSynchPoint, this.localSynchPoint);
412            if (results.size() == 0)
413            {
414                // every thing matched
415                Log.information(this, BaseSynchPoint.AGREGATECOMPARISONRESULT_MATCH_MSG);
416            }
417            else
418            {
419                this.alarm(results);
420            }
421        }
422        catch (DataValidationException e)
423        {
424            Log.alarm("DISHI - Caught DataValidationException[" + e.getMessage()
425                    + "] while trying to compare synch points");
426        }
427        catch (Exception e)
428        {
429            Log.alarm("DISHI - Caught Exception[" + e.getMessage()
430                    + "] while trying to compare synch points");
431        }
432    }

434    protected void alarm(Map<String, String> results)
435    {
436        // there were mismatches
437        Log.information(this, BaseSynchPoint.AGREGATECOMPARISONRESULT_DISCREPANCY_MSG);
438        // log the details
439        Iterator<Entry<String, String>> countComparisonResults = results.entrySet().iterator();
440        Entry<String, String> msgEntry;
441        String msgKey, msg;
442        HashMap<String, Integer> tempMap = new HashMap<String, Integer>();
443        while (countComparisonResults.hasNext())
444        {
445            msgEntry = countComparisonResults.next();
446            msgKey = msgEntry.getKey();
447            msg = msgEntry.getValue();
448            
449            Integer count = outOfSyncCounters.get(msgKey);
450            if (count == null)
451            {
452                    count= 1;
453            }
454            else
455            {
456                    count ++;
457            }
458            
459            if (count >= alarmTriggerCount)
460            {
461                    Log.alarm(this, msg);
462                    count = 0;
463            }
464            else
465            {
466                    Log.information(this, msg);
467                    
468            }
469            tempMap.put(msgKey,count);
470 
471        }
472        outOfSyncCounters=tempMap;
473    }

482    public void initiateSynchPoint() throws DataValidationException
483    {
484        Log.debug(this, "DISHISPINIT - Executing inititiateSynchPoint on the master side");
485        // Slave mode processing
486        // Do a get against dc and compare the last master side counts with current slave side
487        // counts.
488        if (!isMasterMode)
489        {
490            Log.debug("DISHIINIT - Reading TeSynchPoint from DC");
491            DistributedCacheValue<TeSynchPoint> dcv = this.dcOrderCache
492                    .get(DataIntegrityServiceTeImpl.DISTRIBUTEDCACHE_SYNCHPOINT_KEY_INTEGER);
493            TeSynchPoint masterSynchPoint = null;
494            if (dcv != null)
495            {
496                try
497                {
498                    masterSynchPoint = dcv.getValue();
499                }
500                catch (Exception e)
501                {
502                    Log.exception(this,
503                            "Unable to retrieve OhsSynchPoint data from distributed cache", e);
504                }
505            }
506            if (masterSynchPoint != null)
507            {
508                this.readCounts();
509                Log
510                        .debug("DISHIINIT - Comparing local counts with master side TeSynchPoint taken at "
511                                + new Date(masterSynchPoint.getCountCreateTime())
512                                + " to slave side TeSynchPoint read at "
513                                + new Date(this.localSynchPoint.getLastCountAddedTime()));
514                this.compareSynchPoints(masterSynchPoint, this.localSynchPoint);
515            }
516        }
517        else
518        // Master mode processing
519        {
520            Log.debug("DISHIINIT - Reading counts");
521            this.readCounts();
522            Log.debug("DISHIINIT - Setting sequence number on synch point["
523                    + this.synchPointSequenceNumber + "]");
524            this.localSynchPoint.setCounter(this.synchPointSequenceNumber);
525            Log.debug("DISHIINIT - Set sequence number on synch point["
526                    + this.synchPointSequenceNumber + "]");
527            if (this.isInitialSynchPoint())
528            {
529                Log.information("DISHIINIT - Pushing initializing TeSynchPoint number["
530                        + this.localSynchPoint.getSynchPointCounter() + "] to DC");
531            }
532            else
533            {
534                Log.debug("DISHIINIT - Pushing TeSynchPoint number["
535                        + this.localSynchPoint.getSynchPointCounter() + "] to DC");
536            }
537            this.genericDataSyncHome.put(
538                    DataIntegrityServiceTeImpl.DISTRIBUTEDCACHE_SYNCHPOINT_KEY_INTEGER,
539                    this.localSynchPoint);
540            Log.debug("DISHIINIT - Incrementing synchpoint sequence number["
541                    + this.synchPointSequenceNumber + "]");
542            this.increment();
543            Log.debug("DISHIINIT - Incremented synchpoint sequence number["
544                    + this.synchPointSequenceNumber + "]");
545        }
546    }
