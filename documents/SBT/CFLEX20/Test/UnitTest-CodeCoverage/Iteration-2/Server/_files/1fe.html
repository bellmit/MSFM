632            }
633            catch (FatalFoundationFrameworkException fffe)
634            {
635                Log.exception(
636                        new StringBuilder(fffe.getMessage()).append(" Error while reading GroupService property '")
637                                .append(MAX_ELEMENT_NAME_LENGTH)
638                                .append("' :").toString(), fffe);
639            }
640            finally
641            {
642                if (maxNameLength == 0)
643                {
644                    maxNameLength = 64; // setting default value in case of any errors
645                }
646            }
647        }
