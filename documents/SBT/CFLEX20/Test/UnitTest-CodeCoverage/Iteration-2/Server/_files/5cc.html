345    public int getBestCustomerOrderQuantity(Side p_side)
346    {
347            /* The PUBLIC_AT_TOP_CACHE has the best customer market.
348             * The NONCONTINGENT_CACHE (returned by getBestLimitPrice(p_side)) 
349              has the best current market which includes the customer. 
350              So, the two steps:
351              1. Get the best customer market
352              2. Get the best limit price item( best current market)
353              3. If best customer price is equal or better than the best current market, return the customer qty
354                              else customer quantity = 0*/
355            int bestCustomerOrderQty = 0;
356        try
357        {
358                // Step 1 : Get the best customer market.
359            CurrentMarketStruct[] cms = StrategyTradeServiceHelper.getCurrentMarketCacheHome()
360                    .find().getCurrentMarketStructs(this.getProductKey());
361 
362            CurrentMarketStruct bestPublicAtTop = cms[CurrentMarketCache.PUBLIC_AT_TOP_CACHE];
363 
364            Price price = (bestPublicAtTop == null) ? PriceFactory.getNoPrice() : PriceFactory.create(p_side.isBuySide()
365                    ? bestPublicAtTop.bidPrice
366                    : bestPublicAtTop.askPrice);
367            if(!price.isNoPrice())
368            {
369                     MarketVolumeStruct[] sequence = p_side.isBuySide()
370                 ? bestPublicAtTop.bidSizeSequence
371                 : bestPublicAtTop.askSizeSequence;
372 
373                 for (int i = 0; i < sequence.length; i++)
374                 {
375                     if (sequence[i].volumeType == VolumeTypes.CUSTOMER_ORDER)
376                     {
377                             bestCustomerOrderQty += sequence[i].quantity;
378                     }
379                 }
380                 // Step 2 & 3 : Get the best limit price item. If the customer price not better than or equal to best limit price item,
381                 // then, customer is not TOB.So, return 0;
382                 if( (p_side.isBuySide() &&  price.lessThan(getBestLimitPrice(p_side))) ||
383                                 (p_side.isSellSide() && price.greaterThan(getBestLimitPrice(p_side))) )
384                 {
385                         bestCustomerOrderQty = 0;
386                 }
387             }
388         }
389        catch (SystemException e)
390        {
391            Log.exception(e);
392        }
393        return bestCustomerOrderQty;
394    }
