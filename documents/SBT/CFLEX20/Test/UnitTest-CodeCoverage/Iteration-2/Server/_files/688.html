698    /**
699     * determine if a price is inside the derived quote
700     *
701     * Note: DerivedQuote has two sides: AsDefined and Opposite sides. The prices for
702     * both sides are represented by signed number. Negative means a debt. And positive
703     * means a credit. Let b the price for AsDefined side, and a the price for Opposite
704     * side. It is assumed that: b + a > 0. (If b + a <= 0, then the derived quote is crossed).
705     *
706     * 1. A price p for AsDefined side is said to be inside the derived quote if
707     *    -a < p < b.
708     * 2. A price p for Opposite side is said to be inside the derived quote if
709     *    -b < p < a
710     */
711    public boolean isInsideDerivedQuote(Price aPrice, Side aSide){
712        long p = aPrice.toLong();
713        boolean result = false;
714        Price price = getBid().getPrice();
715        long bid = 0;
716        if (!price.isNoPrice())
717        {
718            bid=price.toLong();
719        }
720        price = getAsk().getPrice();
721        long ask = 0;
722        if (!price.isNoPrice())
723        {
724            ask = price.toLong();
725        }
726        if (aSide.isAsDefinedSide()) {
727            result = p < bid && -p < ask;
728        }
729        else if ( aSide.isOppositeSide() ){
730            result = p < ask && -p < bid;
731        }
732        else {
733            Log.alarm("DerivedQuoteHelper >> isInsideDerivedQuote: Side " + aSide.toSideValue() + " is unexpected");
734        }
735        return result;
736    }
737 
738    /**
739     * determine if a price is at the derived quote
740     *
741     * Note: DerivedQuote has two sides: AsDefined and Opposite sides. The prices for
742     * both sides are represented by signed number. Negative means a debt. And positive
743     * means a credit. Let b the price for AsDefined side, and a the price for Opposite
744     * side. It is assumed that: b + a > 0. (If b + a <= 0, then the derived quote is crossed).
745     *
746     * 1. A price p for AsDefined side is said to be at the derived quote if
747     *    (p = b || p = -a).
748     * 2. A price p for Opposite side is said to be at the derived quote if
749     *    (p = a || p = -b).
750     */
751    public boolean isAtDerivedQuote(Price aPrice, Side aSide) {
752        boolean result = false;
753        long price = aPrice.toLong();
754            Price bid = getBid().getPrice();
755            Price ask = getAsk().getPrice();
756        if (aSide.isAsDefinedSide()) {
757                long askPrice = 0;
758            if (!ask.isNoPrice()) {
759                askPrice = ask.toLong();
760            }
761                result = aPrice.equals(bid) || (price == -askPrice);
