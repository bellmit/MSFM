636     * Sets latest expiration date for selected options.
637     */
638    private void setLatestExpirationDate(DateStruct newValue) throws DataValidationException {
639        long numberMonthsOut;
640        if (allProductsSelected()) {
641            // will set to zero and near term month will always be returned.
642            numberMonthsOut = 0;
643        }
644        else {
645            DateStruct nearTerm = ExpirationDateFactory.createNearTermDate(getExpirationStyle()).toStruct();
646            DateStruct latestDate = ExpirationDateFactory.createStandardDate(newValue,getExpirationStyle()).toStruct();
647            // Calculate number of months between the two expiration dates.
648            numberMonthsOut = 12 * (latestDate.year - nearTerm.year) + (latestDate.month - nearTerm.month);
649            if (numberMonthsOut < 0) {
650                throw ExceptionBuilder.dataValidationException("Latest expiration date can't be prior to near term expiration", DataValidationCodes.INVALID_TIME);
651            }
652        }
653        editor.set(_latestExpirationDate, numberMonthsOut, latestExpirationDate);
654    }
