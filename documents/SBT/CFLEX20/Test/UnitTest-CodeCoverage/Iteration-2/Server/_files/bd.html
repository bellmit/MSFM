619                                entry = nextEntry.get(i);
620                                entry.execute();
621                            }
622                        }
623                        catch(Exception e) {

625                            dequeueEx = e;
626                        }
627                        finally {
628                            commitTransaction();
629                        }
630                    }
631                }
632            }
633            catch (Exception e) {
634                dequeueEx = e;
635            }
636            finally {

638                    notifyAll();
639                }
