392    private Tag findTag(String[] pathEls, String currPathEl, int idx, Tag[] curr, String xpath, StringBuffer pathFoundSoFar)
393    {
394        if (verboseFind) { System.err.println("XXX - findTag - currPathEl="+currPathEl); }
395        
396        String el=currPathEl, lbl=el;
397        String attN="name", attV=null;
398        
399        if (el.length() == 0)
400        {
401            throw new FormatException("Invalid property query '" + xpath + "': element #"+(idx+1)+" is empty!");
402        }
403        
404        // Separate the XML tag name (labal) from the attribute name/value to match 
405        //
406        if(el.endsWith(")"))
407        {
408            final int opIdx = el.lastIndexOf('(');
409            if (opIdx < 0)
410            {
411                throw new FormatException("Invalid property query '" + xpath + "': element '"+el+"' is missing '(' to match ')'!");
412            }
413            lbl = el.substring(0, opIdx);
414            attV = el.substring(opIdx+1, el.length()-1);
415            final int eqIdx = attV.lastIndexOf('=');
416            if (eqIdx >= 0)
417            {
418                attN = attV.substring(0, eqIdx);
419                attV = attV.substring(eqIdx+1);
420            }
421        }
422        else
423        {
424            final int usIdx = el.lastIndexOf('_');
425            if (usIdx>0)
426            {
427                lbl = el.substring(0, usIdx);
428                attV = el.substring(usIdx+1);
429            }
430        }
431        lbl = lbl.trim();
432        attN = attN.trim();
433        attV = (attV==null) ? null : attV.trim();
434        if (attV!=null && attV.equals("0") && attN.equals("name"))
435        {
436            // xxx_0 or xxx(0) means 'first match', which is the same as 
437            // having no attribute specification at all.
438            attV = null; 
439        }
440        
441        if (lbl.length()==0 || attN.length()==0 || (attV!=null && attV.length()==0))
442        {
443            throw new FormatException("Invalid property query '" + xpath + "': element "+(idx+1)+" is not a legal form of {xxx|xxx_aname|xxx(aname)|xxx(att=val)} - at least of these values is empty!");
444        }
445        
446        int tagIdx = findTagIndex(curr, lbl, attN, attV);
447        if (verboseFind) { System.err.println("XXX - looking for lbl,N=V - " + lbl+","+attN+"="+attV + " : " + tagIdx); }
448        if (tagIdx < 0)
449        {
450            if (verboseFind) { System.err.println("XXX - not found in tags: "); dumpTags(curr); }
451            return null;  // oh well.
452        }
453        else // found currPathEl
454        {
455            pathFoundSoFar.append(currPathEl).append('.');
456            if (idx+1 == pathEls.length)
457            {
458                return curr[tagIdx];  // got it (the final tag)!
459            }
460            
461            // Recurse:
462            //
463            final int oldBufLen = pathFoundSoFar.length();
464            if (verboseFind) System.err.println("RECURSE Properties: using actual " + pathEls[idx+1] + " at " + (idx+1) + " in " + xpath + "(pathSoFar=" + pathFoundSoFar+")");
465            Tag result = findTag(pathEls, pathEls[idx+1], idx+1, curr[tagIdx].tags, xpath, pathFoundSoFar);
466            
467            // This is a weird bit.  An SMA query for "Xxx.yyy" does not specify the implicit tag
468            // "Xxx.XxxProperties.yyy" for ManagedProperties.  So if we fail to find the former, we'll
469            // attempt to interject a fake value of XXXProperties to produce the latter.
470            //
471            if (result == null && !lbl.endsWith("Properties"))
472            {
473                pathFoundSoFar.setLength(oldBufLen); // 'erase' anything found previously.
474                String fakeCurrPathEl = lbl + "Properties";
475                if (verboseFind) System.err.println("RECURSE Properties: using fake " + fakeCurrPathEl + " at " + idx + " in " + xpath + "(pathSoFar=" + pathFoundSoFar+")");
476                result = findTag(pathEls, fakeCurrPathEl, idx, curr[tagIdx].tags, xpath, pathFoundSoFar);
477            }
478            return result;
479        }
480    }

498    private static Tag nextTag(PushbackReader reader, Position pos) throws IOException
499    {
500        final Tag tag = new Tag();
501        StringBuffer buf = new StringBuffer();
502        TokenType tokType = nextToken(reader, buf, pos);
503        if (tokType == TokenType.NULL_TOKEN)
504        {
505            return null; // done!
506        }
507        if (tokType != TokenType.OPEN_TAG)
508        {
509            throw new FormatException("Expected next tag tag or EOF - got a " + tokType + " at " + pos);
510        }
511        buf = new StringBuffer(32);
512        tokType = nextToken(reader, buf, pos);
513        if (tokType != TokenType.IDENT)
514        {
515            throw new FormatException("Expected STRING tag after OPEN_TAG - got a " + tokType + " at " + pos);
516        }
517        tag.label = buf.toString();
518        
519        tokType = TokenType.NULL_TOKEN; // just so long as it isn't IDENT anymore.
520        
521        // Parse attributes:
522        //
523        List<Attribute> attList = null;
524        while (true)
525        {
526            if (tokType != TokenType.IDENT)
527            {
528                buf.setLength(0);
529                tokType = nextToken(reader, buf, pos);
530                if (tokType == TokenType.CLOSE_EMPTY_TAG)
531                {
532                    break;
533                }
534                if (tokType == TokenType.CLOSE_TAG)
535                {
536                    break;
537                }
538                if (tokType != TokenType.IDENT)
539                {
540                    throw new FormatException("Expected IDENT, CLOSE_TAG, or CLOSE_EMPTY_TAG - got a " + tokType + " at " + pos);
541                }
542            }
543            
544            // Parse an attribute:
545            //
546            if (attList==null)
547            {
548                attList = new ArrayList<Attribute>(4);
549            }
550            Attribute att = new Attribute();
551            att.label = buf.toString();
552            buf.setLength(0);
553            tokType = nextToken(reader, buf, pos);
554            if (tokType == TokenType.CLOSE_EMPTY_TAG || tokType == TokenType.CLOSE_TAG)
555            {
556                att.value = "";
557                attList.add(att);
558                break;
559            }
560            if (tokType == TokenType.IDENT) // an empty attribute was followed by another one, or an end tag.
561            {
562                att.value = "";
563                attList.add(att);
564                continue; // because tokType is now IDENT, we'll skip the nextToken parse at the beginning of the loop.
565            }
566            if (tokType != TokenType.ASSIGN_OP)
567            {
568                throw new FormatException("Expected IDENT, CLOSE_TAG, or ASSIGN_OP - got a " + tokType + " at " + pos);
569            }
570            buf.setLength(0);
571            tokType = nextToken(reader, buf, pos);
572            if (tokType != TokenType.QUOTED_STRING && tokType != TokenType.IDENT)
573            {
574                throw new FormatException("Expected QUOTED_STRING or IDENT - got a " + tokType + " at " + pos);
575            }
576            att.value = buf.toString();
577            if (att.label.equalsIgnoreCase("name"))
578            {
579                tag.name = att.value;
580            }
581            attList.add(att);
582        }
583        if (attList != null)
584        {
585            tag.attributes = (Attribute[]) attList.toArray(new Attribute[attList.size()]);
586        }
587        if (tokType == TokenType.CLOSE_EMPTY_TAG)
588        {
589            return tag;
590        }
591        
592        // This is a bit tricky.  It's hard to know if we're going to find a subtag, a comment, or
593        // the text body of the current tag (tag).
594        //
595        buf = new StringBuffer(128);
596        List<Tag> subTags = null;
597        while (true)
598        {
599            final int frRow=pos.row;
600            int ch = readUntil(reader, buf, pos, "<", Integer.MAX_VALUE);
601            if (ch != '<')
602            {
603                throw new FormatException("Unterminated tag starting at line " + frRow);
604            }
605            reader.read(); // dispose of the '<'
606            // maybe a subtag, maybe a comment: is followed by a '!'?
607            int ch2 = reader.read(); // get the next char
608            if (ch2 < 0)
609            {
610                throw new FormatException("Invalid open tag at EOF at " + pos);
611            }
612            if (ch2 == '!') // read "<!", the beginning of a directive or a comment.  Skip it.
613            {
614                final int commentRow = pos.row;
615                pos.move(ch);
616                pos.move(ch2);
617                if (skipIgnorable(reader, pos) < 0)
618                {
619                    throw new FormatException("Unterminated comment starting on line " + commentRow);
620                }
621                continue;
622            }
623            if (ch2 == '/') // read "</", the beginning of an "end tag"
624            {
625                pos.move(ch);
626                pos.move(ch2);
627                StringBuffer endTagBuf = new StringBuffer(32);
628                tokType = nextToken(reader, endTagBuf, pos);
629                if (tokType != TokenType.IDENT)
630                {
631                    throw new FormatException("Contents of end tag not an XML identifier: '" + endTagBuf + " at " + pos);
632                }
633                if (!endTagBuf.toString().equalsIgnoreCase(tag.label))
634                {
635                    throw new FormatException("End tag '"+endTagBuf+"' != Begin tag '" + tag.label + "' at " + pos);
636                }
637                final int atLine = pos.row;
638                tokType = nextToken(reader, endTagBuf, pos);
639                
640                if (tokType != TokenType.CLOSE_TAG)
641                {
642                    throw new FormatException("End tag not not terminated on line " + atLine);
643                }
644                
645                break; // done parsing this tag!
646            }
647            
648            // Otherwise, this should be a subtag.  Parse it:
649            //
650            reader.unread(ch2);
651            reader.unread(ch);
652            if (subTags==null)
653            {
654                subTags = new ArrayList<Tag>(32);
655            }
656            Tag subtag = nextTag(reader, pos);
657            subtag.parentTag = tag;
658            subTags.add(subtag);  // Recurse!
659        }
660        if (subTags != null)
661        {
662            tag.tags = (Tag[]) subTags.toArray(new Tag[subTags.size()]);
663        }
664        tag.body = buf.toString();
665        return tag;
666    }

668    /**
669     * Return the next token.  This will be the next angle brace, close tag, empty closing tag,
670     * equals sign, identifier, or string (with any wrapped ticks removed) encountered in the stream.
671     * whitespace and comments are skipped.
672     * 
673     * @param reader
674     * @return
675     * @throws IOException
676     */
677    private static TokenType nextToken(PushbackReader reader, StringBuffer buf, Position pos) throws IOException
678    {
679        int lastCh = 0;
680        int ch = 0;
681        for (;;lastCh = ch)
682        {
683            ch = reader.read();
684            if (ch < 0)
685            {
686                return TokenType.NULL_TOKEN;
687            }
688            pos.move(ch);
689            if (Character.isWhitespace((char)ch))
690            {
691                continue;
692            }
693            if (lastCh == '<')
694            {
695                if (ch == '!' || ch == '?')
696                {
697                    if (skipIgnorable(reader, pos)<0)
698                    {
699                        return TokenType.NULL_TOKEN;
700                    }
701                    ch = 0;
702                    continue;
703                }
704                else
705                {
706                    pos.unmove(ch);
707                    reader.unread(ch);
708                    buf.append((char)ch);
709                    return TokenType.OPEN_TAG;  // parsed an "open tag".
710                }
711            }
712            if (ch == '\'' || ch == '\"')
713            {
714                pos.unmove(ch);
715                reader.unread(ch);
716                if (parseQuotedString(reader, buf, pos) < 0)
717                {
718                    return TokenType.NULL_TOKEN;
719                }
720                return TokenType.QUOTED_STRING;
721            }
722            if (IDENT_MEMBERS.indexOf(ch)>=0 || Character.isLetterOrDigit((char)ch))
723            {
724                pos.unmove(ch);
725                reader.unread(ch);
726                if (parseIdent(reader, buf, pos) < 0)
727                {
728                    return TokenType.NULL_TOKEN;
729                }
730                return TokenType.IDENT;
731            }
732            if (ch == '>')
733            {
734                buf.append((char)ch);
735                return (lastCh=='/' ? TokenType.CLOSE_EMPTY_TAG : TokenType.CLOSE_TAG);
736            }
737            if (ch == '=')
738            {
739                buf.append((char)ch);
740                return TokenType.ASSIGN_OP;
741            }
742        }
743    }

746    {

750        {

753            {
754                return ch;
755            }
756            if (ch == openQuote)
757            {
758                return ch;

761            {

764                if (ch < 0)
765                {
766                    return ch;
767                }
768            }
769            else
770            {
771                pos.move(ch);
772            }
773            buf.append((char)ch);
774        }
775    }

777    private static int parseSpecialChar(PushbackReader reader, Position pos) throws IOException
778    {
779        final int ch = reader.read();
780        if (ch != '&')
781        {
782            if (ch >= 0)
783            {
784                pos.move(ch);
785            }
786            return ch;
787        }
788        final StringBuffer buf = new StringBuffer(4);
789        if (readUntil(reader, buf, pos, ";", 10)<0 || reader.read() < 0) // (also read the semicolon)
790        {
791            return -1;
792        }
793        final String str = buf.toString();
794        if (str.equalsIgnoreCase("lt"))
795        {
796            return '<';
797        }
798        if (str.equalsIgnoreCase("gt"))
799        {
800            return '>';
801        }
802        if (str.equalsIgnoreCase("amp"))
803        {
804            return '&';
805        }
806        if (str.equalsIgnoreCase("quot"))
807        {
808            return '\"';
809        }
810        int intVal;
811        try
812        {
813            intVal = Integer.parseInt(str);
814        }
815        catch (NumberFormatException ex)
816        {
817            throw new FormatException("'&" + str + ";' is an invalid or unsupported special tag at " + pos);
818        }
819        reader.read();
820        pos.move(ch);
821        return intVal;
822    }

827    private static int readUntil(PushbackReader reader, StringBuffer buf, Position pos, String readUntilAnyOf, int max) throws IOException

844    private static int parseIdent(PushbackReader reader, StringBuffer buf, Position pos) throws IOException
845    {
846        while (true)
847        {
848            int ch = reader.read();
849            if (ch < 0)
850            {
851                return ch;
852            }
853            if (IDENT_MEMBERS.indexOf(ch)>=0 || Character.isLetterOrDigit((char)ch))
854            {
855                pos.move(ch);

857            }
858            else if (ch == '&')
859            {
860                int specialCh = parseSpecialChar(reader, pos);
861                if (specialCh < 0)
862                {

877     */
878    private static int skipIgnorable(PushbackReader reader, Position pos) throws IOException

895            {
896                return ch;
897            }
898            ch3=ch2;
899            ch2=ch;
900        }

903    private Map getSMATypeMap()
904    {
