393    {
394        String casName = registration.getName();
395        int heartBeatBucketForDelete = registration.getBucket();
396        Log.debug(this, "deleteRegistration casName:" + casName + " bucket:" + heartBeatBucketForDelete);
397

399        heartBeatRegistrations.remove((Object)casName);

402        Iterator delIter = processingBuckets[heartBeatBucketForDelete].iterator();
403        String bucketCasName = null;
404        while ( delIter.hasNext() )

412            }
413            bucketCasName = delReg.getName();
414            Log.debug(this, "deleteRegistration: bucket:" + heartBeatBucketForDelete + " name:" + bucketCasName);
415            if (bucketCasName.equals(casName))
416            {
417                try {
418                    delIter.remove();
419                } catch (Exception e) {
420                    Log.debug(this, "deleteRegistration: unexpected in remove:" + casName + " bucket:" + heartBeatBucketForDelete + " - returning early");
421                    Log.exception(this,e);
422                    return;
423                }
424                break;
425            }
426        }
427        Log.debug(this, "deleteRegistration: removed:" + casName + " bucket:" + heartBeatBucketForDelete);
428        // keep everything in step
429        housekeeping();
430    }

433    private synchronized void housekeeping()
434    {
435        int lowCount = Integer.MAX_VALUE;
436 
437        for (int i=0; i < numBuckets; i++)
438        {
