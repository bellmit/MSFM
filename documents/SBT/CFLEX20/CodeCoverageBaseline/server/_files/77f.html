339        // this is really a false quote update, in the situation, we don't want to check firm quote
340        // it should be treated as a new quote
341        // the workaround is to identify true update from new quote, in order to do so,
342        // we have to make an assumption, if current quote is valid price but zero quantity, 
343        // we will consider as a new quote and bypass firm quote check
344        // we have to check both sides quantity being zero, because incoming quote update got traded
345        // could also resolve in non zero price and quantity zero 
346        double askQuotePrice = currentQuote.getAsk().getPrice().isValuedPrice() ? currentQuote.getAsk().getPrice().toDouble() : 0d;
347        double bidQuotePrice = currentQuote.getBid().getPrice().isValuedPrice() ? currentQuote.getBid().getPrice().toDouble() : 0d;
348        if (askQuotePrice != 0 && currentQuote.getAsk().getQuantity() == 0 &&
349                        bidQuotePrice != 0 && currentQuote.getBid().getQuantity() == 0) {
350                return null;
351        }
352        
353        // check firm quote requirement for nbbo_reject
354        // if price to be updated or the aggregated quantity at auction starting time 
355        // at opposite side CBOE best price is less than auctioning quantity, expire auction
356        // before update
357        int updateQuantity = 0;
358        int currentQuantity = 0;
359        Price updatePrice;
360        Side auctionSide = auction.getSide();
361        Price currentPrice;
362        if (auctionSide.isBuySide())
363        {
364                updateQuantity = updateQuote.quoteV3.quote.askQuantity;
365                currentQuantity = currentQuote.getAsk().getQuantityAllowed();
366                updatePrice = PriceFactory.create(updateQuote.quoteV3.quote.askPrice);
367                currentPrice = currentQuote.getAsk().getPrice();
368        }
369        else
370        {
371                updateQuantity = updateQuote.quoteV3.quote.bidQuantity;
372                currentQuantity = currentQuote.getBid().getQuantityAllowed();
373                updatePrice = PriceFactory.create(updateQuote.quoteV3.quote.bidPrice);
374                currentPrice = currentQuote.getBid().getPrice();
375        }
376        
377        if (toPrematureAuctionForFirmQuoteRule(auction, currentPrice, updatePrice, currentQuantity, updateQuantity)) {
378                Log.information(className + " Expiring " + HALHelper.AUCTION_TYPE_STRING.getAuctionTypeString(auction.getAuctionType()) +
379                                " Auction for quote update as quote price is changed or not enough firm quote quantity at price: " 
380                     + auction.getOppSideCboePrice() + " " + auction.toString());
381                processAuctionExpire(auction, AuctionTerminateReasons.QUOTE_QUANTITY_DECREASE, null);
382                return null;
383        }
384        return null;

387    private boolean toPrematureAuctionForFirmQuoteRule(AuctionInternal auction, Price currentPrice, Price updatePrice, int cancelQuantity, int updateQuantity)
388    {
389            // if price is going to be updated to be worse, set update quantity to zero
390            if (updatePrice != null && 
391                            updateQuantity != 0 &&
392                            auction.getSide().getOtherSide().isFirstBetter(auction.getOppSideCboePrice(), updatePrice)) {
393                    updateQuantity = 0;
394            }
395            if (auction.getSide().getOtherSide().isFirstBetterOrEqual(currentPrice, auction.getOppSideCboePrice()) &&
396                        !hasEnoughQQuantity(auction, cancelQuantity, updateQuantity)) {
397                    return true;
398            }

402    /**
403     * Check if any update would cause firm quote not met to premature auction
404     * if total quote like quantity is less than auction quantity, premature auction
405     * AuctionProcessorHAL2Impl
406     * @param auction
407     * @param orderBook
408     * @param currentQuantity
409     * @param updateQuantity
410     * @return
411     * boolean
412     */
413    private boolean hasEnoughQQuantity(AuctionInternal auction, int cancelQuantity, int updateQuantity)
414    {
415            // get firm quote aggregated quantity which is lesser of auction order remaining quantity
416            // and opposite side BBO quantity at auction starting time
417            int auctionRemainingQuantity = auction.getAuctionRemainingQuantity();
418            int obligatedQuantity = auction.getOppSideCboeQuantity() > auctionRemainingQuantity ? auctionRemainingQuantity : auction.getOppSideCboeQuantity();
419            
420            // get firm quote aggregated quantity in the book
421                // it is the current booked q quantity plus/minus changed quantity
422                int bookedQQuantity = getQQuantityAhead(auction);
423                
424                // the quantity after the update should be enough to satisfy obligated quantity
425                if ((bookedQQuantity - cancelQuantity + updateQuantity) < obligatedQuantity) {
426                        return false;
427                }
428            return true;
429    }
