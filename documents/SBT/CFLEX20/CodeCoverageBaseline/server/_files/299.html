286    /**
287     * Figure out if our book was traded through. Generate satisfaction alert.
288     * @param productInfo
289     * @param ticker
290     * @param sessionName
291     */
292    private void processSatisfactionAlert(TradingProductBBBOInfo productInfo, InternalTickerStruct ticker, String sessionName){
293 
294 
295        ValuedPrice bookAskPrice = null;
296        ValuedPrice bookBidPrice = null;
297        ValuedPrice tickerPrice = null;
298        int bookAskCustomerQuantity = 0;
299        int bookBidCustomerQuantity = 0;
300        int bookAskQuantity = 0;
301        int bookBidQuantity = 0;
302        ArrayList orderIds;
303 
304        orderIds = new ArrayList();
305        tickerPrice = PriceFactory.createValuedPrice (ticker.ticker.lastSalePrice);
306 
307        try{
308            if(productInfo.getBestAsk() != null){
309                bookAskPrice = PriceFactory.createValuedPrice (productInfo.getBestAsk());
310                bookAskCustomerQuantity = productInfo.getBestCustomerAskVolume();
311                bookAskQuantity = productInfo.getBestTotalAskVolume();
312            }
313            else { if(Log.isDebugOn()) Log.debug(this, "No ask info");}
314            if(productInfo.getBestBid() != null){
315                bookBidPrice = PriceFactory.createValuedPrice (productInfo.getBestBid());
316                bookBidCustomerQuantity = productInfo.getBestCustomerBidVolume();
317                bookBidQuantity = productInfo.getBestTotalBidVolume();
318            }
319            else {if(Log.isDebugOn()) Log.debug(this, "No bid info");}
320 
321 
322            if((bookAskPrice != null) && (!bookAskPrice.isNoPrice()) && (bookAskQuantity > 0)
323                    && (bookAskPrice.lessThan(tickerPrice))) { // tradethrough on the ask side
324                if(Log.isDebugOn()) Log.debug(this, "Ask tradethrough");
325                if(Log.isDebugOn()) Log.debug(this, "book ask price: " + bookAskPrice.toString() + "ticker price: " + tickerPrice.toString());
326                if(bookAskCustomerQuantity <= 0){  /* We haven't gotten the customer quantity for this top of the book yet*/
327                    bookAskCustomerQuantity = getCustomerQuantity(ticker.ticker.productKeys, Sides.ASK, bookAskPrice, orderIds); // only customer orders are eligible for satisfaction alerts
328                    updateCustomerQuantity(productInfo, Sides.ASK, bookAskCustomerQuantity, orderIds);
329                }
330                int quantity = bookAskCustomerQuantity <= ticker.ticker.lastSaleVolume ?  bookAskCustomerQuantity : ticker.ticker.lastSaleVolume;
331 
332                if(quantity > 0){
333                    if(Log.isDebugOn()) Log.debug(this, "Ask customer quantity: " + quantity);
334                    SatisfactionAlertStruct alert = new SatisfactionAlertStruct();
335                    orderIds = productInfo.getOrderIds();
336                    populateSatisfactionAlert(alert, ticker, Sides.ASK, bookAskPrice.toStruct(), quantity, sessionName, orderIds);
337                    getAlertService().createSatisfactionAlert(alert);
338                    alertCounter++;
339                }
340                else{if(Log.isDebugOn()) Log.debug(this, "Ask customer quantity: " + quantity);}
341            }
342            if((bookBidPrice != null) && (!bookBidPrice.isNoPrice()) && (bookBidQuantity > 0)
343                    && (tickerPrice.lessThan(bookBidPrice))){  // tradethrough on the bid side
344                if(Log.isDebugOn()) Log.debug(this, "Bid tradethrough");
345                if(bookBidCustomerQuantity <= 0){  /* We haven't gotten the customer quantity for this top of the book yet*/
346                    bookBidCustomerQuantity = getCustomerQuantity(ticker.ticker.productKeys, Sides.BID, bookBidPrice, orderIds); // only customer orders are eligible for satisfaction alerts
347                    updateCustomerQuantity(productInfo, Sides.BID, bookBidCustomerQuantity, orderIds);
348                }
349                int quantity = bookBidCustomerQuantity <= ticker.ticker.lastSaleVolume ?  bookBidCustomerQuantity : ticker.ticker.lastSaleVolume;
350                if(quantity > 0){
351                    if(Log.isDebugOn()) Log.debug(this, "Bid customer quantity: " + quantity);
352                    SatisfactionAlertStruct alert = new SatisfactionAlertStruct();
353                    orderIds = productInfo.getOrderIds();
354                    populateSatisfactionAlert(alert, ticker, Sides.BID, bookBidPrice.toStruct(), quantity, sessionName, orderIds);
355                    getAlertService().createSatisfactionAlert(alert);
356                    alertCounter++;
357                }else {if(Log.isDebugOn()) Log.debug(this, "Bid customer quantity: " + quantity);}
358            }
359        } catch(Exception e)
360        {Log.exception(e);}
361    }

394    private int getCustomerQuantity(ProductKeysStruct productKeys, char side, Price price, ArrayList orderIds) throws SystemException {
