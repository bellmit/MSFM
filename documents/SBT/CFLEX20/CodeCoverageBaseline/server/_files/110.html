608        * Creates a name for a strategy.
609        *
610        * @param legs legs of the strategy
611        * @param legProducts the product of the corresponding leg
612        * @return strategy name
613        */
614        private String createStrategyName(StrategyLegStruct[] legs, Product[] legProducts, int strategyType) {
615            String name;
616            boolean containsEquityLeg = false;
617            for (Product legProduct : legProducts)
618            {
619                if (legProduct.isProductType(ProductTypes.EQUITY))
620                {
621                    containsEquityLeg = true;
622                }
623            }
624        Log.information("containsEquityLeg = " + containsEquityLeg);
625        if (legs.length == 2) {
626            name = (String) commonTypeNames.get( strategyType );
627        }
628        else if (legs.length > 2 && containsEquityLeg){
629            name = MULTIPLE_LEG_EQUITY_SPREAD;
630        }
631        else if (legs.length == 3) {
632            name = THREE_LEG_SPREAD;
633        }
634        else if (legs.length == 4) {
635            name = FOUR_LEG_SPREAD;
636        }
637        else {
638            name = MULTIPLE_LEG_SPREAD;
639        }
640        //skip this for equity legged spreads
641        if ( !containsEquityLeg )
642        {
643        // find Option leg to use to name strategy
644        for ( int legToUse = 0; legToUse < legProducts.length; legToUse++ )
645        {
646            if ( legProducts[ legToUse ].isProductType( ProductTypes.OPTION ) )
647            {
648                // add date, price of the first option to name
649                OptionEditor option = (OptionEditor) legProducts[ legToUse ].getProductEditor();
650                name = name + " " + option.getExpirationDate().toString();
651                name = name + " " + option.getExercisePrice().toString();
652                break;
653            }
654            else if ( legProducts[ legToUse ].isProductType( ProductTypes.FUTURE ) )
655            {
656                // add date of first future to name
657                FutureEditor future = (FutureEditor) legProducts[ legToUse ].getProductEditor();
658                name = name + " " + future.getExpirationDate().toString();
659                break;
660            }
661          }
662 
663        }
664 
665        // add time stamp for uniqueness
666        name = name + ":" + System.currentTimeMillis();
667        return name;
668        }
