346    public void updateEvent(Object key, DistributedCacheValue<TeSynchPoint> updatedValue)
347    {
348        if (!(key instanceof Integer))
349        {
350            Log.alarm("DISHIUPDATEEVENT - Update event called with non-Integer key, returning without processing");
351            return;
352        }
353        Integer intKey = (Integer) key;
354        // ignore updates for objects other than the SynchPoint and all events in master mode
355        if (intKey.intValue() != DataIntegrityServiceTeImpl.DISTRIBUTEDCACHE_SYNCHPOINT_KEY_INTEGER .intValue() || this.isMasterMode)
356        {
357            Log.information("DISHIUPDATEEVENT - Update event called in mode[" + this.isMasterMode
358                    + "] and key[" + key + "], returning");
359            return;
360        }
361        Map<String, String> results = new HashMap<String, String>();
362        TeSynchPoint dcSynchPoint = updatedValue.getValue();
363        this.setLastSynchPointTimeStamp(dcSynchPoint.getLastCountAddedTime());
364        try
365        {
366            // read current count values on slave side
367            this.readCounts();
368            if (this.localSynchPoint.isInitializingSynchPoint() || dcSynchPoint.isInitializingSynchPoint())
369            {
370                // set offsets before comparison
371                Log.information("DISHIUPDATEEVENT - Initializing offsets with dc synch point number[" + dcSynchPoint.getSynchPointCounter() + "]");
372                // use master synchpoint to set offsets for slave side event counts
373                this.setEventCountOffsets(dcSynchPoint);
374                Log.information("DISHIUPDATEEVENT - slave side SynchPoint[" + this.localSynchPoint.toString() + "]");
375                Log.information("DISHIUPDATEEVENT - master side SynchPoint[" + dcSynchPoint.toString() + "]");
376            }
377            // increment slave side synchPoint counter
378            increment();
379            this.localSynchPoint.setCounter(synchPointSequenceNumber);
380        }
381        catch (DataValidationException e)
382        {
383            String msg = "DISHI - Caught " + e.getClass().getName() + " while trying to retrieve counts for local server, unable to compare to primary server counts";
384            Log.exception(this, msg, e);
385            Log.alarm(this, msg);
386        }
387        // Compare distributed cache SynchPoint to local SynchPoint
388        try
389        {
390            results = compareSynchPoints(dcSynchPoint, this.localSynchPoint);
391            if (results.size() == 0)
392            {
393                // every thing matched
394                Log.information(this, BaseSynchPoint.AGREGATECOMPARISONRESULT_MATCH_MSG);
395            }
396            else
397            {
398                this.alarm(results);
399            }
400        }
401        catch (DataValidationException e)
402        {
403            Log.alarm("DISHI - Caught DataValidationException[" + e.getMessage()
404                    + "] while trying to compare synch points");
405        }
406        catch (Exception e)
407        {
408            Log.alarm("DISHI - Caught Exception[" + e.getMessage()
409                    + "] while trying to compare synch points");
410        }
411    }

413    protected void alarm(Map<String, String> results)
414    {
415        // there were mismatches
416        Log.alarm(this, BaseSynchPoint.AGREGATECOMPARISONRESULT_DISCREPANCY_MSG);
417        // log the details
418        Iterator<Entry<String, String>> countComparisonResults = results.entrySet().iterator();
419        Entry<String, String> msgEntry;
420        String msgKey, msg;
421        while (countComparisonResults.hasNext())
422        {
423            msgEntry = countComparisonResults.next();
424            msgKey = msgEntry.getKey();
425            msg = msgEntry.getValue();
426            Log.alarm(this, msg);
427        }

437    public void initiateSynchPoint() throws DataValidationException
438    {
439        Log.debug(this, "DISHISPINIT - Executing inititiateSynchPoint on the master side");
440        // Slave mode processing
441        // Do a get against dc and compare the last master side counts with current slave side
442        // counts.
443        if (!isMasterMode)
444        {
445            Log.debug("DISHIINIT - Reading TeSynchPoint from DC");
446            DistributedCacheValue<TeSynchPoint> dcv = this.dcOrderCache
447                    .get(DataIntegrityServiceTeImpl.DISTRIBUTEDCACHE_SYNCHPOINT_KEY_INTEGER);
448            TeSynchPoint masterSynchPoint = null;
449            if (dcv != null)
450            {
451                try
452                {
453                    masterSynchPoint = dcv.getValue();
454                }
455                catch (Exception e)
456                {
457                    Log.exception(this,
458                            "Unable to retrieve OhsSynchPoint data from distributed cache", e);
459                }
460            }
461            if (masterSynchPoint != null)
462            {
463                this.readCounts();
464                Log
465                        .debug("DISHIINIT - Comparing local counts with master side TeSynchPoint taken at "
466                                + new Date(masterSynchPoint.getCountCreateTime())
467                                + " to slave side TeSynchPoint read at "
468                                + new Date(this.localSynchPoint.getLastCountAddedTime()));
469                this.compareSynchPoints(masterSynchPoint, this.localSynchPoint);
470            }
471        }
472        else
473        // Master mode processing
474        {
475            Log.debug("DISHIINIT - Reading counts");
476            this.readCounts();
477            Log.debug("DISHIINIT - Setting sequence number on synch point["
478                    + this.synchPointSequenceNumber + "]");
479            this.localSynchPoint.setCounter(this.synchPointSequenceNumber);
480            Log.debug("DISHIINIT - Set sequence number on synch point["
481                    + this.synchPointSequenceNumber + "]");
482            if (this.isInitialSynchPoint())
483            {
484                Log.information("DISHIINIT - Pushing initializing TeSynchPoint number["
485                        + this.localSynchPoint.getSynchPointCounter() + "] to DC");
486            }
487            else
488            {
489                Log.debug("DISHIINIT - Pushing TeSynchPoint number["
490                        + this.localSynchPoint.getSynchPointCounter() + "] to DC");
491            }
492            this.genericDataSyncHome.put(
493                    DataIntegrityServiceTeImpl.DISTRIBUTEDCACHE_SYNCHPOINT_KEY_INTEGER,
494                    this.localSynchPoint);
495            Log.debug("DISHIINIT - Incrementing synchpoint sequence number["
496                    + this.synchPointSequenceNumber + "]");
497            this.increment();
498            Log.debug("DISHIINIT - Incremented synchpoint sequence number["
499                    + this.synchPointSequenceNumber + "]");
500        }
501    }
