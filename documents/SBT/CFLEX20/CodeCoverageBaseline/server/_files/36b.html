357    /** Turn a RuntimeException into the proper kind of CBOEdirect exception,
358     * and throw that exception.
359     * @param rte Exception to translate.
360     * @throws com.cboe.exceptions.CommunicationException in case of CORBA communications failure.
361     * @throws com.cboe.exceptions.AuthorizationException if user not allowed to call this operation.
362     * @throws com.cboe.exceptions.SystemException in case of other exceptions.
363     */
364    private void translateRuntimeException(RuntimeException rte)
365            throws CommunicationException, AuthorizationException, SystemException
366    {
367        if (rte instanceof org.omg.CORBA.COMM_FAILURE)
368        {
369            org.omg.CORBA.COMM_FAILURE e = (org.omg.CORBA.COMM_FAILURE) rte;
370            Log.exception(this, e);
371            throw ExceptionBuilder.communicationException(e.toString(), com.cboe.idl.cmiErrorCodes.CommunicationFailureCodes.TRANSPORT_FAILURE);
372        }
373        if (rte instanceof org.omg.CORBA.NO_PERMISSION)
374        {
375            org.omg.CORBA.NO_PERMISSION e = (org.omg.CORBA.NO_PERMISSION) rte;
376            Log.exception(this, e);
377            throw ExceptionBuilder.authorizationException(e.toString(), 0);
378        }
379        // Some other kind of RuntimeException
380        Log.exception(this, rte);
381        throw ExceptionBuilder.systemException(rte.toString(), 0);
382    }
