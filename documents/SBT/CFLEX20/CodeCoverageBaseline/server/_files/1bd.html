345        /**
346         * Main processing loop for acknowledgement processing thread.
347         */
348        public void run() {
349            int maxBatchSize = getMaxBatchSize();
350            Log.information(UserReportManager.this, "Ack. Processing thread started. Batch Size for ack processing = " + maxBatchSize);
351            int currentBatchSize = 0;
352            Exception dequeueEx = null;
353            if (acknowledgementQueue != null) {
354                acknowledgementQueue.reportDequeueThreadRunning();
355            }
356            try  {
357                while (!acknowledgementProcessingStopped) {
358                    int waitTime = currentBatchSize == 0 ? Queue.INFINITE_TIMEOUT : 50;
359                    Object nextAck = null;
360                    try {
361                        nextAck = acknowledgementQueue.dequeue(waitTime);
362                    }
363                    catch (QueueException e) {
364                        Log.alarm(UserReportManager.this, "Acknowledgement processing interrupt ignored");
365                    }
366                    if (nextAck != null) {
367                        if (currentBatchSize == 0) {
368                            Transaction.startTransaction();
369                        }
370                        if (reportType == UserReportHome.ORDER_REPORT_TYPE) {
371                            processOrderAck((OrderAcknowledgeStructV3) nextAck);
372                        }
373                        else {
374                            processQuoteAck((QuoteAcknowledgeStructV3) nextAck);
375                        }
376                        currentBatchSize++;
377                        if (currentBatchSize >= maxBatchSize) {
378                            if (!Transaction.commit()) {
379                                Log.alarm(UserReportManager.this, "Unable to commit batch of " + currentBatchSize + " acknowledgements");
380                                Transaction.rollback();
381                            }
382                            currentBatchSize = 0;
383                        }
384                    }
385                    else  {
386                        // Null ack returned.
387                        if(currentBatchSize > 0) {
388                            // No acks have been received in the timeout period
389                            // so we will persist what we have now and then
390                            // loop back around.  When attempting to dequeue we
391                            // will have no acks waiting so we'll have an infinite
392                            // timeout (unless acks have been received and then
393                            // we'll start processing another batch).
394                            if(!Transaction.commit())  {
395                                Log.alarm(UserReportManager.this, "Unable to commit batch of " + currentBatchSize + " acknowledgements");
396                                Transaction.rollback();
397                            }
398                            currentBatchSize = 0;
399                        }
400                    }
401                }
402            }
403            catch(Exception e)  {
404                dequeueEx = e;
405            }
406            finally { // if anything goes wrong be sure to commit if needed
407                if(currentBatchSize > 0) {
408                    Log.alarm(UserReportManager.this, "ACKNOWLEDGEMENT PROCESSING THREAD ABORTED. No more acks will be saved.");
409                    if(!Transaction.commit()) {
410                        Log.alarm(UserReportManager.this, "Unable to commit batch of " + currentBatchSize + " acknowledgements");
411                        Transaction.rollback();
412                    }
413                }
414 
415                // report to instumentor
416                if (acknowledgementQueue != null) {
417                    acknowledgementQueue.reportDequeueThreadExited(dequeueEx);
418                }
419            }
420        }
421    }   // End of inner class AckProcessingThread
