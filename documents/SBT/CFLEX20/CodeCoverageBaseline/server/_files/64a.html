398     * pre-calculate the remainder of the opposite incoming tradable after the auction trade
399     * @param tradingClass

401     * @param incomingTradable - the opposite incoming tradable only
402     * @return
403     */
404    private static int getRemainingQuantity(TradingClass tradingClass, AuctionInternal auction, Tradable incomingTradable)
405    {
406        int result = 0;
407        int incomingQty = incomingTradable.getQuantityAllowed();
408        if (incomingQty <=0)
409        {
410            return result;
411        }
412 
413        short matchType = MatchTypes.UNSPECIFIED;
414        if (auction.getAuctionType() == AuctionTypes.AUCTION_INTERNALIZATION )
415        {
416            if (auction.getFirmMatchOrder().getPrice().isMarketPrice())
417            {
418               matchType =MatchTypes.AUTO_MATCH;
419            }
420        }
421        if (matchType != MatchTypes.AUTO_MATCH
422                || !(incomingTradable.treatedLikeQuote() || incomingTradable.getContingencyType() == ContingencyTypes.AUCTION_RESPONSE) )
423        {   // not aucto match type, or not a matchable inbound tradable
424            result =  incomingQty - auction.getAuctionedOrder().getRemainingQuantity();
425        }
426        else
427        {   // for auto-match, we need to find out the match percentage for one group other than match order
428            InternalizationPercentageStruct[] percentageScale =  tradingClass.getTradingPropertyInternalizationGuaranteedPercentage();
429            double rate = 0.0d;
430            int otherGroupNumber = 1;
431            for (int i=0; i< percentageScale.length; i++)
432            {
433                if (otherGroupNumber >= percentageScale[i].lowerRange &&
434                    otherGroupNumber < percentageScale[i].upperRange &&
435                    auction.getAuctionType()== percentageScale[i].auctionType)
436                {
437                    rate = percentageScale[i].percentage;
438                    break;
439                }
440            }
441            // calculate the remainder
442            int auctionedQty = auction.getAuctionedOrder().getRemainingQuantity();
443            result = incomingQty -  Math.round((float)((1.0- rate)* auctionedQty));
444        }
445        return result;
446    }

449     * get mid-point between the best auction price of the oppoiste side and the rest market (NBBO) price of the auction side
450     * the two input prices should always never cross with each other, though they can be equal
451     * @param bestAuctionPrice - the current best valid price point of auction participants,
452     * which include auction reponses, quotes and orders on the oppoiste side of the auciton
453     * @param bestNBBOPrice - the best NBBO price of the same side of the auction
454     * @param tradingClass - a TradingClass
455     * @return
456     */
457    private static Price getMidPointAuctionPrice(Price bestAuctionPrice, Price bestNBBOPrice, TradingClass tradingClass)
458    {
459        Price midPoint = PriceFactory.create((bestAuctionPrice.toDouble() + bestNBBOPrice.toDouble())/2.0D);
460 
461        Price midPointFloor = tradingClass.floorPremiumToAuctionTick(midPoint, AuctionTypes.AUCTION_INTERNALIZATION, bestAuctionPrice);
462        Price midPointCeil = tradingClass.ceilPremiumToAuctionTick(midPoint, AuctionTypes.AUCTION_INTERNALIZATION, bestAuctionPrice);
463        Price newPrice = midPointFloor;
464        if (!midPointFloor.equals(midPointCeil))
465        {
466            if (bestAuctionPrice.toDouble() > bestNBBOPrice.toDouble())
467            { // auction(book) side is buy side, to favor auction, price goes down.
468                  newPrice = midPointFloor ;
469            }
470            else
471            { // auction(book) side is sell side, to favor auction, price goes up.
472                  newPrice = midPointCeil;
473            }
474        }
475        if (Log.isDebugOn())
476        {
477            Log.debug(  "Get the midPoint price between the bestAuctionPrice ("
478                    + bestAuctionPrice + ") and bestNBBOPrice (" + bestNBBOPrice + ") = " + newPrice);
479        }
480        return newPrice;

888     * @param priorityTradableSets -the input priority tradableSets
889     * @param tradableSide -the side of the priority tradable sets
890     * @param aProduct
891     * @param quantityToAllocate
892     * @param outputParticipantsLists
893     * @param outputTradePrices
894     * @param outputTradeQuantityList
895     * @return the total allocated quantity to the input priorityTradableSets
896     */
897    private static int createParticipantLists(Iterator priorityTradableSets,
898                                    Side tradableSide,
899                                    TradingProduct aProduct,
900                                    int quantityToAllocate,
901                                    ArrayList outputParticipantsLists,
902                                    ArrayList outputTradePrices,
903                                    ArrayList outputTradeQuantityList)
904    {
905        int allocatedQuantity = 0;
906        Price currentPrice = new NoPrice();
907        int allocatedAtCurrentPrice = 0;
908        ParticipantList currentOpposite = null;
909        while (priorityTradableSets.hasNext()&& allocatedQuantity < quantityToAllocate)
910        {
911            PriorityTradableSet nextSet = (PriorityTradableSet) priorityTradableSets.next();
912            if (nextSet.getPrice() != currentPrice ) {
913                // The first time through, currentPrice will be a NoPrice.
914                // If this is not the first time through, we have hit a new price that we
915                // need to allocate quantity for, so we save the last price information.
916                if (!currentPrice.isNoPrice()) {
917                    outputTradePrices.add(determineTradePrice(tradableSide, currentPrice, aProduct));
918                    outputTradeQuantityList.add(new Integer(allocatedAtCurrentPrice));
919                    outputParticipantsLists.add(currentOpposite);
920                }
921                // get ready to allocate quantity at a new price.
922                currentOpposite = new ParticipantListImpl();
923                currentPrice = nextSet.getPrice();
924                allocatedAtCurrentPrice = 0;
925            }
926 
927            int nextAllocatedQuantity = nextSet.allocate(currentOpposite, quantityToAllocate - allocatedQuantity);
928            allocatedQuantity += nextAllocatedQuantity;
929            allocatedAtCurrentPrice += nextAllocatedQuantity;
930        }
931        if (allocatedQuantity > 0) {
932            // If we were able to find opposite side quantity, we now need to create corresponding
933            // participant lists for the same side tradables (the new one and any we are trading ahead)
934            outputTradePrices.add(determineTradePrice(tradableSide, currentPrice, aProduct));
935            outputTradeQuantityList.add(new Integer(allocatedAtCurrentPrice));
936            outputParticipantsLists.add(currentOpposite);
937        }
938        return allocatedQuantity;
939    }
