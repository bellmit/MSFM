396protected BrokerStrategy findOrderStrategy(Order order, boolean useTargetState) throws StrategyNotFoundException {
397 
398    short contingencyType;
399        if (order == null) {
400                throw new StrategyNotFoundException("Order is null");
401        }
402    else

409 
410    protected BrokerStrategy findOrderStrategy(Order order, boolean useTargetState, short contingencyType)
411            throws StrategyNotFoundException
412    {
413        short state = getStateCode(order.getProductKey(), useTargetState);
414 
415    boolean nbboProtection = order.needsNBBOProtection();
416    boolean useMarketOpenStrategy = useMarketOpenStrategy(order, useTargetState);
417 
418    if(order.isLinkageOrder())
419    {
420        return findLinkageOrderStrategy(state, order.getOrderOriginType());
421    }
422    
423    if(order.getOrderOriginType() == InternalOrderOriginTypes.CROSS_PRODUCT_LEG_SYNTHETIC_ORDER)
424    {
425        return this.findCrossProductLegOrderStrategy(state, nbboProtection);
426    }
427    
428    BrokerStrategy targetStrategy;
429        if (contingencyType == ContingencyTypes.NONE ||
430            contingencyType == ContingencyTypes.RESERVE || 
431            contingencyType == ContingencyTypes.NOTHELD || contingencyType == ContingencyTypes.DO_NOT_ROUTE) {
432                if (order.getPrice().isMarketPrice()) {
433                        targetStrategy = findMarketOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
434                }
435                else {
436                        targetStrategy = findLimitOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
437                }
438        }
439        else if (contingencyType == ContingencyTypes.IOC || contingencyType == ContingencyTypes.FOK) {
440                targetStrategy = findTimedOrderStrategy(state, nbboProtection, order.generatesNBBOFlash());
441        }
442    else if (contingencyType == ContingencyTypes.AON) {
443        // AON orders are treated the same as a regular limit order
444        targetStrategy = findLimitOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
445    }
446    else if (contingencyType == ContingencyTypes.OPG) {
447        targetStrategy = findOPGOrderStrategy(state, useMarketOpenStrategy);
448    }
449    else if (order.isCrossOrder()) {
450        targetStrategy = findCrossOrderStrategy(state);
451    }
452    else if (contingencyType == ContingencyTypes.STP || contingencyType == ContingencyTypes.STP_LOSS) {
453        // Inactive stop orders need to have stop price met to be made active, active stop orders
454        // have already had the contingency met and can be processed as if they don't have a
455        // contingency.
456        if (order.isWaiting()) {
457            targetStrategy = findStopOrderStrategy(state, true /* market order */, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
458        }
459        else {
460                        targetStrategy = findMarketOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
461                }
462    }
463    else if (contingencyType == ContingencyTypes.STP_LIMIT) {
464        // Inactive stop orders need to have stop price met to be made active, active stop orders
465        // have already had the contingency met and can be processed as if they don't have a
466        // contingency.
467        if (order.isWaiting()) {
468            targetStrategy = findStopOrderStrategy(state, false /* not a market order */, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
469        }
470                else {
471                        targetStrategy = findLimitOrderStrategy(state, nbboProtection, useMarketOpenStrategy, order.generatesNBBOFlash());
472                }
473    }
474    else if (contingencyType == ContingencyTypes.INTERMARKET_SWEEP) {
475        targetStrategy = findSweepOrderStrategy(state);
476    }
477    else
478    {
479        throw new StrategyNotFoundException("Order strategy not found, unsupported contingency code of " + contingencyType);
480    }
481        return targetStrategy;
482}
