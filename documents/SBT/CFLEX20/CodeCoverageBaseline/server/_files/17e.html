379    protected boolean isDeltaNeutral(OrderStruct order)
380    {
381        boolean stockFlag = false;
382        int legPriceCount = 0;
383        LegOrderDetailStruct[] legs = order.legOrderDetails;
384        try
385        {
386            if(legs != null)
387            {
388                // Atleast one leg should be an Equity
389                for(int i =0; i< legs.length; i++)
390                {
391        
392                    ProductKeysStruct legProd = getProductQueryService().getProductKeys(legs[i].productKey);
393                    if(legProd.productType == ProductTypes.EQUITY || legProd.productType == ProductTypes.INDEX) 
394                    {
395                        stockFlag = true;
396                    }
397                    legPriceCount = (legs[i].mustUsePrice.type == PriceTypes.VALUED)? (legPriceCount+1): legPriceCount;
398        
399                }
400        
401                boolean noOrderPrice = (order.price.type == PriceTypes.NO_PRICE)? true: false;
402        
403                return stockFlag && noOrderPrice && (legPriceCount == legs.length);
404            }
405        }
406        catch (DataValidationException e)
407        {
408            Log.exception("PendingAdjustedmentOrdersLogger: Can't resolve Delta Neutral", e);
409        }
410        catch (SystemException e)
411        {
412            Log.exception("PendingAdjustedmentOrdersLogger: Can't resolve Delta Neutral", e);
413        }
414        catch (NotFoundException e)
415        {
416            Log.exception("PendingAdjustedmentOrdersLogger: Can't resolve Delta Neutral", e);
417        }
418        catch (CommunicationException e)
419        {
420            Log.exception("PendingAdjustedmentOrdersLogger: Can't resolve Delta Neutral", e);
421        }
422        catch (AuthorizationException e)
423        {
424            Log.exception("PendingAdjustedmentOrdersLogger: Can't resolve Delta Neutral", e);
425        }
426        return stockFlag;
