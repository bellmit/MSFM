542/**
543 * Process a side of a current market message.  This expects that it is called in a transactional context.
544 *
545 * <pre>
546 *  Handle these cases (consider ">" means "better than" and "<" means "worse than"):
547 *
548 *   1) CM'.price > NBBO.price
549 *        - NBBO.price = CM'.price
550 *        - NBBO.vols = CM'.vol
551 *
552 *   2) CM'.price == NBBO.price
553 *        - if NBBO does not include sessExch (ie, CM):
554 *          - NBBO.vols = CM'.vol
555 *        - else
556 *          - NBBO.vols[CM] = CM'.vol
557 *
558 *   3) CM'.price < NBBO.price
559 *        3a) CM'.price > BOTR.price
560 *            - NBBO.price = CM'.price
561 *            - NBBO.vols = exchVol
562 *
563 *        3b) CM'.price = BOTR.price
564 *            - NBBO.vols = merge(CM'.vol, BOTR.vols)
565 *
566 *        3c) CM'.price < BOTR.price
567 *           - if NBBO includes sessExch (ie, CM):
568 *            - NBBO.price = BOTR.price
569 *            - NBBO.vols = BOTR.vol
570 * </pre>
571 *
572 * @param side - Sides.BID or Sides.ASK
573 * @param currMktStruct IN/OUT - the noncontingent market to be checking.  It's isAskInNBBO / isBidInNBBO indicators may be modified
574 * @param mktData - The MarketData object for this guy.  May be modified transactionally.
575 * @return boolean - true if NBBO was modified by this method call.
576 */
577protected boolean handleMarketBestSide(String session, char side, CurrentMarketStruct currMktStruct, MarketData mktData)
578{
579    final BestQuote nbbo = mktData.getNBBO();
580    final Price nbboPrice = helper.getPrice(nbbo, side);
581    final PriceStruct priceStruct = (side==Sides.BID) ? currMktStruct.bidPrice : currMktStruct.askPrice;
582    final Price cmPrice = PriceFactory.create(priceStruct);
583    final boolean tie = nbboPrice!=null && cmPrice.equals(nbboPrice);
584    final boolean better = !tie && (nbboPrice==null || helper.priceIsZero(nbboPrice) || helper.isBetter(side, cmPrice, nbboPrice));
585 
586    boolean modifiedNBBO = false;
587 
588    if (better)
589    {
590        if (unitTesting) coverageSet.add("CM.1");
591        helper.setNbboToCurrMkt(session, side, cmPrice, currMktStruct, nbbo);
592        helper.setCurrMktInNBBO(side, mktData, true);
593        modifiedNBBO = true;
594    }
595    else if (tie)
596    {
597        if (unitTesting) coverageSet.add("CM.2");
598        helper.addSessExchToNBBO(session, side, nbbo, helper.calcVolume(side, currMktStruct));
599        helper.setCurrMktInNBBO(side, mktData, true);
600        modifiedNBBO = true;
601    }
602    else //(worse)
603    {
604        if (unitTesting) coverageSet.add("CM.3");
605        final BestQuote botr = mktData.getBOTR();
606        final Price botrPrice = (botr==null) ? null : helper.getPrice(botr, side);
607        final boolean botrIsBetter = helper.isBetter(side, botrPrice, cmPrice);
608        final boolean tieBotr = (botrPrice!=null && botrPrice.equals(cmPrice));
609 
610        modifiedNBBO = true;
611 
612        if (!botrIsBetter && !tieBotr) // (better)
613        {
614            helper.setNbboToCurrMkt(session, side, cmPrice, currMktStruct, nbbo);
615            helper.setCurrMktInNBBO(side, mktData, true);
616            if (unitTesting) coverageSet.add("CM.3a");
617        }
618        else if (tieBotr)
619        {
620            final String sessExch = helper.getSessionExchange(session);
621            final int newVol = helper.calcVolume(side, currMktStruct);
622            final ExchangeVolumeStruct[] botrExchVol = helper.getVolumes(botr,side);
623 
624            helper.setNBBO(side, nbbo, cmPrice, helper.addToVolumes(sessExch, newVol, botrExchVol));
625            helper.setCurrMktInNBBO(side, mktData, true);
626            if (unitTesting) coverageSet.add("CM.3b");
627        }
628        else //(worse: botr is better)
629        {
630            helper.setNBBO(side, nbbo, helper.getPrice(botr,side), helper.getVolumes(botr,side));
631            helper.setCurrMktInNBBO(side, mktData, false);
632            if (unitTesting) coverageSet.add("CM.3c");
633        }
634    }
635    return modifiedNBBO;
636}
