276            threads[i] = new SettlementQueryThread(i, fromIdx, toIdx, p_settlements);
277            threads[i].start();
278        }
279        int numNotFound = 0;
280        int numIgnored = 0;
281        for (int i = 0; i < threads.length; i++)
282        {
283            try
284            {
285                threads[i].join();
286            }
287            catch (InterruptedException e)
288            {
289                Log.alarm(this, "Join of settlement product query thread ["+i+"] interrupted.");
290            }
291            p_ignoredWriter.write(threads[i].ignoredWriter.toString());
292            p_errWriter.write(threads[i].errWriter.toString());
293            numNotFound += threads[i].getNumNotFound();
294            numIgnored += threads[i].getNumIgnored();
295        }
296        Log.information(this, "All "+numThreads+" SettlementQueryThreads have completed.");
297        
298        // Build a list of blocks to update:
299        //
300        // (List of List of SettlementValues)
301        //
302        final List settlementBlocks = new ArrayList(1 + len/p_commitBlockSize);
303        List currBlock = null; // (List of SettlementValues)
304        for (int i = 0; i < threads.length; i++)
305        {
306            final List settlements = threads[i].getSettlementValues();
307            for (int j = 0; j < settlements.size(); j++)
308            {
309                if (currBlock==null)
310                {
311                    currBlock = new ArrayList(p_commitBlockSize);
312                }
313                currBlock.add(settlements.get(j));
314                if (currBlock.size() == p_commitBlockSize)
315                {
316                    settlementBlocks.add(currBlock);
317                    currBlock = null;
318                }
319            }
320        }
321        if (currBlock != null) // don't forget the leftovers!
322        {
323            settlementBlocks.add(currBlock);
324        }   
325        Log.information(this, "Created "+settlementBlocks.size()+" blocks of settlement prices to update.");
326            
327        final int numUpdated = actuallyUpdateSettlements(p_errWriter, p_infoWriter, settlementBlocks);
328        return new int[]
329           {
330                numUpdated,
331                numIgnored,
332                numNotFound + (totalBlocks(settlementBlocks) - numUpdated),
333           };
334    }

748        private String translate(PositionAwareInputStream p_in, int p_num, String[][] p_exchange_map)
