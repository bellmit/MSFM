478            final ServerQueue targetQueue = find(Object.class, targetQueueName);
479

502    /**
503     * For admin requests, return a list of ServerQueues for a ServerQueue name. A comma-delimited list of

506     * order.

508    protected List<String> queueNames(String p_queueNameSpec)
509    {
510        Map<String, ServerQueue> queueMap = getCollectionMapping(false);
511        ArrayList<String> matchedNames = new ArrayList<String>();
512        ArrayList<String> allNames = new ArrayList<String>(queueMap.keySet());
513        String[] names = p_queueNameSpec.split(",");
514        for (int i = 0; i < names.length; i++)
515        {
516            // For backwards compatibility, replace "%" with ".*" to create a regex.
517            //
518            final String nameRegex = names[i].replaceAll("%", ".*");
519            for (Iterator iter = allNames.iterator(); iter.hasNext();)
520            {

522                if (queueName.matches(nameRegex))
523                {
524                    matchedNames.add(queueName);
525                }
526            }
527        }
528        Collections.sort(matchedNames);
529        return matchedNames;
530    }
