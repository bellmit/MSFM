338            return findLinkageOrderStrategy(state, nbboProtection);
339        }

342            {
343                return findHybridComplexOrderStrategy(order, useTargetState, contingencyType); 
344            }
345

348                return findInternalizedStrategy(order, haveOrderHandlingInstruction, theRemainderHandlingMode, useTargetState, state, contingencyType, nbboProtection);

363            }
