399        
400        if (p_type.equals("rebuildClassList"))

415            }

426                }
427            }
428        }
429        else if (p_type.equals("currentlyMonitoring"))
430        {
431            if (parseBoolean(p_value, buf))
432            {
433                for (final UserInputMonitorImpl impl : this.implPerSession)
434                {
435                    final boolean boolVal = p_value.endsWith("true");

438                    " from ").append(wasMon).append(" to ").append(boolVal);

441                    {
442                        final boolean started = impl.startMonitoring();
443                        buf.append("\n(started monitoring)").append(started?"":" [FAILED TO START UIM]");
444                    }

446                    {
447                        impl.stopMonitoring();
448                        buf.append("\n(stopped monitoring)");
449                    }
450                }
451            }
452        }

490                            + "ms due to poll:quote timeout ratio.");

499            }
500            final int val = Integer.parseInt(p_value);
