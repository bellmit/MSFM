310    setCurrentEndOfSessionEvent((short) -1);
311    setLastCompletedEndOfSessionEvent((short) -1);
312    final DateStruct currentDay = DateWrapper.convertToDate(getBusinessDay());
313    final TradingSessionElement[] elements = new TradingSessionElement[templates.length];
314    
315        TradingSessionThreadPoolHome threadPoolHome;
316        try {
317                 threadPoolHome = (TradingSessionThreadPoolHome) HomeFactory.getInstance().findHome(TradingSessionThreadPoolHome.HOME_NAME);
318        } catch (CBOELoggableException e1) {
319                threadPoolHome = new TradingSessionThreadPoolHomeImpl();
320        }
321 
322        if (threadPoolHome==null)
323        {
324                threadPoolHome = new TradingSessionThreadPoolHomeImpl();
325        }
326 
327        ThreadPoolExecutor threadPool = threadPoolHome.findSessionTemplateThreadPool();
328            
329    
330    for ( int idx = 0; idx < templates.length; idx++) {
331            final int i=idx;
332            Runnable worker = new Runnable()
333            {
334                    public void run()
335                    {
336                            
337                            try 
338                            {
339                                     Transaction.startTransaction();
340                                      elements[i] = getTradingSessionElementHome().getCurrentForTemplate(templates[i].getTemplateName());
341 
342                                    if (elements[i] == null) 
343                                    {
344                                            elements[i] = getTradingSessionElementHome().createFromTemplate(templates[i], currentDay);
345                                    }
346                            Log.information("Create session for:"+ elements[i].getElementName()+" started");
347                                    elements[i].updateFromTemplate();
348                                    Transaction.commit();
349                            }
350                            catch (TransactionFailedException e) 
351                            {
352                                    Log.exception(TradingSessionImpl.this, "Unable to create session element for template " + templates[i].getTemplateName() + ". Due to error : " + e.details.message, e);
353                            }
354                            catch (Exception e) 
355                            {
356                                    Log.exception(TradingSessionImpl.this, "Unable to create session element for template " + templates[i].getTemplateName(), e);
357                            }
358                            Log.information("Create session for:"+ elements[i].getElementName()+" finished");
359                    }
360            };
361            threadPool.execute(worker);
362    }
363 
364    Log.information("Waiting for all the session template worker for "+sessionName+" to be finished");
365    threadPool.shutdown();
366    try {
367                if (!threadPool.awaitTermination(3600, TimeUnit.SECONDS))
368                {
369                        Log.alarm("createSession "+getSessionName()+" is not completed after 1 HOUR");
370                }
371        } catch (InterruptedException e) 
372        {
373                Log.alarm("createSession"+e);
374        }
375    Log.information("All session template workers finished for "+sessionName);
376 
377    return elements;
378}
