443            typedValue = value.split(",");
444        }
445        else if (propType.isArray() && propType.getComponentType() == Integer.TYPE)
446        {
447            String[] values = value.split(",");
448            int[] ints = new int[values.length];
449            for (int i = 0; i < values.length; i++)
450            {
451                ints[i] = Integer.parseInt(values[i]);
452            } 
453            typedValue = ints;
454        }
455        else if (propType == Collection.class || propType == List.class || propType == ArrayList.class)
456        {
457            ArrayList<String> values = new ArrayList<String>();
458            for (String str : value.split(","))
459            {
460                values.add(str);
461            }
462            typedValue = values;
463        }
464        else
465        {
466            throw new FatalFoundationFrameworkException(errMsg.add("Method arg").add("type",propType)
467                    .add("is not supported by this annotation.").end());
468        }
469        return typedValue;
470    }

472    private void setServiceReference(Object p_object, Method p_method, FFServiceDependency p_svcDep) 
473        throws FatalFoundationFrameworkException
474    {
475        Class<?>[] args = p_method.getParameterTypes();
476        MsgBuilder errMsg = msg().add("Cannot set service reference for")
477            .add("home",getName()).add(": ").add("in object ").addShortName(p_object.getClass())
478            .add("method",p_method.getName()).add(": ");
479        if (args.length != 1)
480        {
481            throw new FatalFoundationFrameworkException(errMsg.add("argCount", args.length).add("!= 1").end());
482        }
483        Class<?> serviceType = args[0];
484        
485        String findMethodName = p_svcDep.method();
486        BOHome home = null;
487        if (p_svcDep.homeName().length()>0)
488        {
489            if (p_svcDep.homeName().startsWith("#"))
490            {
491                throw new FatalFoundationFrameworkException(
492                        errMsg.add("@FFServiceDependency.homeName does not support '#' prefixes.").end()); 
493            }
494            try
495            {
496                home = HomeFactory.getInstance().findHome(p_svcDep.homeName());
497                if (home == null)
498                {
499                    throw new FatalFoundationFrameworkException(errMsg.add("Home",p_svcDep.homeName())
500                            .add("not found!").end());
501                }
502            }
503            catch (CBOELoggableException e)
504            {
505                throw new FatalFoundationFrameworkException(e, errMsg.add("Home",p_svcDep.homeName())
506                        .add("not found").end());
507            }
508        }
509 
510        if (home != null)
511        {
512            if (!setServiceReference(p_object, p_method, errMsg, serviceType, home, findMethodName))
513            {
514                throw new FatalFoundationFrameworkException(errMsg.add("Home",home.getName())
515                        .add("does not provide an appropriate ").add(findMethodName).add(" method")
516                        .end());
517            }
518        }
519        else
520        {
521            Enumeration homesEnum = HomeFactory.getInstance().getHomes();
522            boolean referenceIsSet = false;
523            while (homesEnum.hasMoreElements())
524            {
525                BOHome aHome = (BOHome)homesEnum.nextElement();
526                if (setServiceReference(p_object, p_method, errMsg, serviceType, aHome, findMethodName))
527                {
528                    if (referenceIsSet)
529                    {
530                        throw new FatalFoundationFrameworkException(
531                                errMsg.add("More than one home provides a \"")
532                                .add(findMethodName).add("():").addShortName(serviceType)
533                                .add("\" method: this is an ambiguous reference in this process ")
534                                .add("(use the homeName=\"...\" attribute in FFServiceDependency)").end());
535                    }
536                    referenceIsSet = true;
537                    home = aHome;
538                }
539            }
540            if (!referenceIsSet)
541            {
542                throw new FatalFoundationFrameworkException(errMsg.add("No home provides a \"")
543                        .add(findMethodName).add("():").addShortName(serviceType).add("\" method.").end());
544            }
545        }
546        log().info(msg().add("Injected ").add(home.getName()).add('.').add(findMethodName)
547                .add("() to ").addShortName(p_object.getClass()).add('.').add(p_method.getName()));
548    }

551    private void setFactoryReference(Object p_object, Method p_method, FFFactoryDependency p_factoryDep) 
552        throws FatalFoundationFrameworkException
553    {
554        // Verify that p_method is 'sane':
555        //
556        Class<?>[] args = p_method.getParameterTypes();
557        MsgBuilder errMsg = msg().add("Cannot set factory reference for")
558            .add("home",getName()).add(": ").add("method",p_method.getName()).add(": ");
559        if (args.length != 1)
560        {
561            throw new FatalFoundationFrameworkException(errMsg.add("argCount", args.length).add("!= 1").end());
562        }
563        Class<?> serviceType = args[0];
564        
565        // Get the factory instance (may be null - then we'll hope for a static getter method)
566        //
567        Class<?> factoryType = p_factoryDep.type();
568        String instanceMethodName = p_factoryDep.factoryInstanceMethod();
569        Method instanceMethod = null;
570        Object factoryInstance = null;
571        if (instanceMethodName.equals(""))
572        {
573            // Search for "public static {factoryType} xxxXxx()"
574            for (Method method : factoryType.getMethods())
575            {
576                if (Modifier.isPublic(method.getModifiers()) && Modifier.isStatic(method.getModifiers())
577                        && method.getReturnType().isAssignableFrom(factoryType) && method.getReturnType() != Object.class)
578                {
579                    instanceMethod = method;
580                    break;
581                }
582            }
583        }
584        else
585        {
586            try
587            {
588                instanceMethod = factoryType.getMethod(instanceMethodName, new Class[0]);
589            }
590            catch (NoSuchMethodException ex)
591            {
592                throw new FatalFoundationFrameworkException(ex, errMsg.add("instanceMethod",instanceMethodName)
593                  .add("does not exist").end());      
594            }
595        }
596        if (instanceMethod != null)
597        {
598            try
599            {
600                factoryInstance = instanceMethod.invoke(null, new Object[0]);
601            }
602            catch (Exception e)
603            {
604                throw new FatalFoundationFrameworkException(e, errMsg.add("Failed to invoke ")
605                        .addShortName(factoryType).add('.').add(instanceMethodName)
606                        .add("as a static method").end());
607            }
608        }
609        
610        // Get the method to invoke:
611        //
612        String methodName = p_factoryDep.method();
613        Method factoryMethod;
614        try
615        {
616            factoryMethod = factoryType.getMethod(methodName, new Class[0]);
617        }
618        catch (NoSuchMethodException e)
619        {
620            throw new FatalFoundationFrameworkException(e, errMsg.add("Method ")
621                    .addShortName(factoryType).add('.').add(methodName)
622                    .add("not found.").end());
623        }
624        if (factoryInstance==null && !Modifier.isStatic(factoryMethod.getModifiers()))
625        {
626            throw new FatalFoundationFrameworkException(errMsg.add("Method ")
627                    .addShortName(factoryType).add('.').add(methodName)
628                    .add(" is not static but no factory instance method was found.").end());
629        }
630        if (!serviceType.isAssignableFrom(factoryMethod.getReturnType()))
631        {
632            throw new FatalFoundationFrameworkException(errMsg.add("Method ")
633                    .addShortName(factoryType).add('.').add(methodName)
634                    .add("returnType",factoryMethod.getReturnType())
635                    .add("is not a valid param for the annotated method.").end());
636        }
637        
638        // Invoke the p_object method to pass the factory result object:
639        //
640        Object value;
641        try
642        {
643            value = factoryMethod.invoke(factoryInstance, new Object[0]);
644        }
645        catch (Exception e)
646        {
647            throw new FatalFoundationFrameworkException(e, errMsg.add("Method ")
648                    .addShortName(factoryType).add('.').add(methodName)
649                    .add("is not static but no factory instance method was found.").end());
650        }
651        
652        // Finally, invoke annotated p_method with the value returned by the factory:
653        //
654        try
655        {
656            p_method.invoke(p_object, value);
657        }
658        catch (Exception e)
659        {
660            throw new FatalFoundationFrameworkException(e, errMsg.add("Failed to invoke method with ")
661                    .addShortName(serviceType).end());
662        }
663        
664        log().info(msg().add("Injected ").addShortName(factoryType).add('.').add(methodName)
665                .add("() to ").addShortName(p_object.getClass()).add('.').add(p_method.getName()));
666    }
