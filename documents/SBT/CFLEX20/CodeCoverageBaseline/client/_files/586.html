443    private int getMDUpdateDelay()
444    {

447        {

458                updateDelay = Integer.parseInt(value);
459                updateDelay = Math.max(updateDelay, MIN_UPDATE_DELAY);
460                updateDelay = Math.min(updateDelay, MAX_UPDATE_DELAY);
461            }
462            catch(NumberFormatException e)

467            }
468        }
469        else
470        {
471            updateDelay = DEFAULT_UPDATE_DELAY;
472        }
473 
474        if (GUILoggerHome.find().isDebugOn() && GUILoggerHome.find().isPropertyOn(GUILoggerBusinessProperty.COMMON))
475        {
476            GUILoggerHome.find().debug(Category + ".getMDUpdateDelay",
477                                       GUILoggerBusinessProperty.COMMON, "updateDelay = " + updateDelay);
478        }
