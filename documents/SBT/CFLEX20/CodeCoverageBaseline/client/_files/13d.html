533    private void removeMarketDataSubscriptions(Object source, Object clientListener, Map subscriptionsByGroupKey, Map subscriptionsBySource)
534    {
535        Set listenerSubscriptionSet = (Set) marketDataSubscriptionsByListener.remove(clientListener);
536        Map groupKeysMap = (Map) subscriptionsBySource.get(source);
537 
538        RemoteMarketDataSubscriptionInfoContainer subscriptionInfo = null;

540        Iterator subscriptionIterator = listenerSubscriptionSet.iterator();
541        while ( subscriptionIterator.hasNext() ) { 
542            subscriptionInfo = (RemoteMarketDataSubscriptionInfoContainer) subscriptionIterator.next();
543            int[] groupKeys = (int[]) groupKeysMap.remove(subscriptionInfo);
544            for (int i=0; i < groupKeys.length ; i++) {
545                Integer key = groupKeys[i];
546                Map subscriptionInfoMap = (Map) subscriptionsByGroupKey.get(key);
547                subscriptionInfoMap.remove(subscriptionInfo);
548                if (subscriptionInfoMap.isEmpty()) {
549                    subscriptionsByGroupKey.remove(key);
550                }
551            }
552        }
553        if (groupKeysMap.isEmpty()) { 
554            subscriptionsBySource.remove(source); 
555        }
