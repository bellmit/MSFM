518        /**
519         * Opens connection to FIX engine and registers as event listener
520         * @throws Exception is thrown if fails to start engine
521         */
522    public void startup() throws Exception {
523            // Get global properties
524            setAppiaIniFile(System.getProperty(FIXINIFILE));
525            if (getAppiaIniFile() == null) {
526                    throw ExceptionBuilder.communicationException(
527                                        "Failed to connect to FIX engine; missing FIXINIFile property",
528                                        CommunicationFailureCodes.TRANSPORT_FAILURE);
529            }
530            setConnection(System.getProperty(FIXCONNECTION));
531            if (getConnection() == null) {
532                    throw ExceptionBuilder.communicationException(
533                                        "Failed to connect to FIX engine; missing FIXConnection property",
534                                        CommunicationFailureCodes.TRANSPORT_FAILURE);
535            }
536 
537                // Now we construct the appia argument list.
538                String[] appiaArgs = new String[1];
539                appiaArgs[0] = getAppiaIniFile();
540                IGUILogger logger = GUILoggerHome.find();
541                if (logger.isDebugOn()) {
542                        logger.debug(
543                                        "About to call MiddlewareInterfaceFactory.getInProcMiddlewareInterface with "
544                                        + appiaArgs[0] + " and connection "
545                                        + getConnection(),
546                                        GUILoggerBusinessProperty.USER_SESSION);
547                }
548 
549        middleware = MiddlewareInterfaceFactory.getInProcMiddlewareInterface(
550                                appiaArgs, getConnection(), 0, MiddlewareInterface.AUTO_ACKNOWLEDGE);
551 
552        // Make sure that middleware handles received session messages
553        MiddlewareConfig config = middleware.getMiddlewareConfig();
554        if (logger.isDebugOn()) {
555                        logger.debug("MiddlewareConfig.SessionMessages = " +
556                                        (config.SessionMessages ? "ON" : "OFF"),
557                                        GUILoggerBusinessProperty.USER_SESSION);
558        }
559        config.SessionMessages = true;
560        middleware.setMiddlewareConfig(config);
561 
562        // wait for Appia to initialize before proceeding
563                if (logger.isDebugOn()) {
564                        logger.debug("About to call MiddlewareInterface.waitForServer",
565                                        GUILoggerBusinessProperty.USER_SESSION);
566                }
567        middleware.waitForServer();
568 
569                // Add the listner, i.e, register the event listener (the callback object that
570                // will receive execution reports).
571                middleware.setMiddlewareEventListener(this);
572 
573                // Start middleware event delivery
574                if (logger.isDebugOn()) {
575                        logger.debug("About to call MiddlewareInterface.start",
576                                        GUILoggerBusinessProperty.USER_SESSION);
577                }
578                middleware.start();
579    }
