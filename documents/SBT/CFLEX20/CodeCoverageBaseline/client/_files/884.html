852    private int appendInt(int offset, int number)
853    {
854        if (number == Integer.MIN_VALUE)
855        {
856            writeInto(offset, INTEGER_MIN_STRING.toCharArray());
857            return INTEGER_MIN_STRING.length();
858        }
859 
860        boolean minus = number < 0;
861        int oldOffset = offset;
862 
863        if (minus) // if negative, just add a minus sign
864        {
865           number = -number;
866           offset = writeInto(offset, '-');
867        }
868 
869        // see if the number is already pre-defined in the lookup table
870        if (number < StringHelper.numbersChars.length)
871        {
872            offset = writeInto(offset, StringHelper.numbersChars[number]);
873        }
874        else if (number < StringHelper.NUMBER_1_000_000) // 123,456
875        {
876            offset = writeInto(offset, StringHelper.trimNumberCharsThousand_x_xxx_000_xxx(number), StringHelper.zeroNumberCharsThousand_x_xxx_xxx_000(number));
877        }
878        else if (number < StringHelper.NUMBER_1_000_000_000) // 123,456,789
879        {
880            offset = writeInto(offset, StringHelper.trimNumberCharsThousand_x_000_xxx_xxx(number), StringHelper.zeroNumberCharsThousand_x_xxx_000_xxx(number), StringHelper.zeroNumberCharsThousand_x_xxx_xxx_000(number));
881        }
882        else // 1,234,567,890
883        {
884            offset = writeInto(offset, StringHelper.trimNumberCharsThousand_0_xxx_xxx_xxx(number), StringHelper.zeroNumberCharsThousand_x_000_xxx_xxx(number), StringHelper.zeroNumberCharsThousand_x_xxx_000_xxx(number), StringHelper.zeroNumberCharsThousand_x_xxx_xxx_000(number));
885        }
886 
887        return offset - oldOffset;
888    }
