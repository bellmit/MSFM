614    private void remove(OrderStruct orderStruct)
615    {
616        CompleteOrderIdStructContainer orderIndex = createCompleteOrderIdStructContainer(orderStruct.orderId);
617 
618        synchronized (getOrderLock(orderIndex))
619        {
620        if (finishedOrderMap.containsKey(orderIndex))
621        {
622                // We're now processing message from 2nd FE.
623            // Or we're processing a Cancel after a Fill, and we'll get those
624                // messages again and we'll need the transactionSequenceNumber;
625                    // So, replacing the previous orderStruct with the latest.
626                finishedOrderMap.put(orderIndex, orderStruct);
627            return;
628        }
629            // This order is no longer active.
630            // Remove from active order cache and other secondary caches;
631            // But first, add it to finished order cache to handle late cancel requests as well as
632            // status messages from second FE. The orders in the finished cache will be
633            // cleared after a predetermined interval.
634            // Now, the order/orderIndex be there only in finishedOrderMap, finishedOrderQueue
635            // and in the userOrderCrossReferenceMap.
636        
637            finishedOrderMap.put(orderIndex, orderStruct);
638            finishedOrderQueue.add(new FinishedOrder(orderStruct));
639            if (userOrderMap.remove(orderIndex)== null)
640            {
641                // This order was not in the order map before.
642                // Seems like a fill or cancel came before NEW.
643                // Any how, we need it in the cross reference map until
644                // removed from the finished order map.
645 
646                    // Add the CMi and Cboe IDs to the x-ref map
647                addToCrossReferenceMaps(orderStruct, orderIndex);
648                
649                // The order could be there in the orderEntryCache. If so, market it as CACHED.
650                orderEntryComplete(orderStruct.orderId);
651            }
652        }
653        if(Log.isDebugOn())
654        {
655            Log.debug(logPrefix + " : moved from active to finished Order Cache " + orderIndex );
656        }
657    }
