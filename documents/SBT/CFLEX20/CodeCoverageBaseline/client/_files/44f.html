886            for ( int i = 0; i < numProps; i++ ) {
887                String prop = propsTok.nextToken();
888                StringTokenizer tok = new StringTokenizer(prop, "=" );
889                String name = null;
890                String value = null;
891                if ( tok.hasMoreTokens() )
892                    name = tok.nextToken();
893                if ( tok.hasMoreTokens() )
894                    value = tok.nextToken();
895                
896                if ( name != null ) {
897                    org.omg.CORBA.Any tempAny = cboeOrb.create_any();
898                    tempAny.insert_string(value);
899                    cosProps[i] = new org.omg.CosTrading.Property( name, tempAny );
900                }
901            }
902        }
903        
904        return cosProps;
