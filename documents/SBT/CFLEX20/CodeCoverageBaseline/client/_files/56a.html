552    private int getOrderQueryDelay()
553    {
554        if (queryDelay < 0)
555        {
556            if (AppPropertiesFileFactory.isAppPropertiesAvailable())
557            {
558                String value = AppPropertiesFileFactory.find()
559                        .getValue(PROPERTIES_SECTION_NAME, ORDER_QUERY_LEVEL_KEY_NAME);
560 
561                if (GUILoggerHome.find().isDebugOn() &&
562                        GUILoggerHome.find().isPropertyOn(GUILoggerBusinessProperty.OMT))
563                {
564                    GUILoggerHome.find().debug(getClass().getName() + ".getOrderQueryDelay",
565                            GUILoggerBusinessProperty.OMT,
566                            ORDER_QUERY_LEVEL_KEY_NAME + '=' + value);
567                }
568 
569                try
570                {
571                    queryDelay = Integer.parseInt(value);
572                    queryDelay = Math.max(queryDelay, MIN_QUERY_DELAY);
573                    queryDelay = Math.min(queryDelay, MAX_QUERY_DELAY);
574                }
575                catch (NumberFormatException e)
576                {
577                    GUILoggerHome.find().exception(getClass().getName() + ".getOrderQueryDelay",
578                            "Error parsing " + ORDER_QUERY_LEVEL_KEY_NAME +
579                                    ", value =" + value, e);
580                    queryDelay = DEFAULT_QUERY_DELAY;
581                }
582            }
583            else
584            {
585                queryDelay = DEFAULT_QUERY_DELAY;
586            }
587 
588            if (GUILoggerHome.find().isDebugOn() &&
589                    GUILoggerHome.find().isPropertyOn(GUILoggerBusinessProperty.OMT))
590            {
591                GUILoggerHome.find().debug(getClass().getName() + ".getOrderQueryDelay",
592                        GUILoggerBusinessProperty.OMT,
593                        "queryDelay = " + queryDelay);
594            }
595            seconds = queryDelay / 1000;
596        }
597        return queryDelay;
598    }

669    private void handleTransactionException(String apiMethodName, Exception e)

677            {
678                SystemException se = (SystemException)e;
679                buffer.append(se.details.message);
680                GUILoggerHome.find().audit(buffer.toString());

682            }
683            else if (e instanceof CommunicationException)
684            {
685                CommunicationException ce = (CommunicationException)e;
686                buffer.append(ce.details.message);
687                GUILoggerHome.find().audit(buffer.toString());

689            }
690            else if (e instanceof DataValidationException)
691            {
692                DataValidationException dve = (DataValidationException)e;
693                buffer.append(dve.details.message);
694                GUILoggerHome.find().audit(buffer.toString());
695                throw dve;
696            }
697            else if (e instanceof TransactionFailedException)
698            {
699                TransactionFailedException tfe = (TransactionFailedException)e;
700                buffer.append(tfe.details.message);
701                GUILoggerHome.find().audit(buffer.toString());
702                throw tfe;
703            }
704            else if (e instanceof NotAcceptedException)
705            {
706                NotAcceptedException nae = (NotAcceptedException)e;
707                buffer.append(nae.details.message);
708                GUILoggerHome.find().audit(buffer.toString());
709                throw nae;
710            }
711            else if (e instanceof AuthorizationException)
712            {
713                AuthorizationException ae = (AuthorizationException)e;
714                buffer.append(ae.details.message);
715                GUILoggerHome.find().audit(buffer.toString());
716                throw ae;
717            }
718            else if (e instanceof NotFoundException)
719            {
720                NotFoundException nfe = (NotFoundException)e;
721                buffer.append(nfe.details.message);
722                GUILoggerHome.find().audit(buffer.toString());

728                throw (UserException)e;
729            }
730        }
731        else
