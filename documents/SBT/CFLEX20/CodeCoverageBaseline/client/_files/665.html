581    protected void registerEnvironment(EnvironmentProperties env)
582    {
583        PropertiesFile propsFile = AppPropertiesFileFactory.find();
584        String envList = propsFile.getValue(ENVIRONMENT_LIST_SECTION_HEADING, "AllEnvironments");
585        // we need to add this to the list of environments, otherwise it will never
586        // be read in when we initialize an EnvironmentManager
587        String envName = env.getName().toUpperCase();
588        envList += VALUE_LIST_DELIMITER + envName;
589        propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, APP_ALL_ENV_PROPERTY_KEY, envList);
590 
591        // now add the environment section itself
592        String iorKey = ENV_BASE + envName + "." + EnvironmentProperties.INIT_REF_IOR;
593        String prefixKey = ENV_BASE + envName + "." + EnvironmentProperties.SBT_PREFIX;
594        String defaultFlagKey = ENV_BASE + envName + ".default";
595        String monSvcChannelPrefixKey = ENV_BASE + envName + "." + EnvironmentProperties.MONITOR_PREFIX;
596        String extentMapChannelPrefixKey = ENV_BASE + envName + "." + EnvironmentProperties.EXTENT_MAP_PREFIX;
597        String processListKey = ENV_BASE + envName + "." + EnvironmentProperties.PROCESS_LIST_REF;
598        String alertErrorPrefixKey = ENV_BASE + envName + "." + EnvironmentProperties.ALERT_ERROR_PREFIX;
599        String alertWarningPrefixKey = ENV_BASE + envName + "." + EnvironmentProperties.ALERT_WARNING_PREFIX;
600 
601        propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, iorKey, env.getInitIORRef());
602        propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, prefixKey, env.getSBTPrefix());
603        propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, defaultFlagKey, "false");
604        propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, monSvcChannelPrefixKey, env.getMonitorPrefix());
605        propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, extentMapChannelPrefixKey, env.getExtentMapPrefix());
606        propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, processListKey, env.getProcessListRef().toString());
607        propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, alertErrorPrefixKey,
608                           Long.toString(env.getAlertErrorThreshold()));
609        propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, alertWarningPrefixKey,
610                           Long.toString(env.getAlertWarningThreshold()));
611 
612        Map svcProps = env.getServiceProperties();
613        Iterator spIter = svcProps.keySet().iterator();
614        while (spIter.hasNext())
615        {
616            String key = (String) spIter.next();
617            String val = (String) svcProps.get(key);
618            propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, ENV_BASE + envName + ".serviceName." + key, val);
619        }

622    protected boolean saveEnvironments()
623    {
624        boolean rv = true;
625        // find the current default env flag
626        PropertiesFile propsFile = AppPropertiesFileFactory.find();
627        Map environmentList = propsFile.getSection(ENVIRONMENT_LIST_SECTION_HEADING);
628        String allEnvironmentsValue = (String) environmentList.get(APP_ALL_ENV_PROPERTY_KEY);
629 
630        if (allEnvironmentsValue != null)
631        {
632            String[] environmentNames = allEnvironmentsValue.split(VALUE_LIST_DELIMITER);
633            // for each environment that we read in on initialization (or later added) ...
634            for (int m = 0; m < environmentNames.length; ++m)
635            {
636                // first, replace the actual sbt_prefix String (e.g. "Prod") with the
637                // token "%sbtPrefix%, so that when we read them in again, the token
638                // substitution works
639                EnvironmentProperties ep = environments.get(environmentNames[m]);
640 
641                Map svcProps = ep.getServiceProperties();
642                Iterator propKeys = svcProps.keySet().iterator();
643 
644                while (propKeys.hasNext())
645                {
646                    String key = (String) propKeys.next();
647                    String val = (String) svcProps.get(key);
648                    String newVal = translateOut(val);
649//                                        val.replaceFirst(ep.getSBTPrefix(),APP_PREFIX_TOKEN);
650                    propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING,
651                                       ENV_BASE + environmentNames[m] + ".serviceName." + key, newVal);
652                }
653                // if it purports to be the default environment
654                if (defaultEnvironment != null && defaultEnvironment.equals(environments.get(environmentNames[m])))
655                {
656                    propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, ENV_BASE + environmentNames[m] + ".default",
657                                       "true");
658                }
659                else
660                {
661                    // reset it to default=false
662                    propsFile.addValue(ENVIRONMENT_LIST_SECTION_HEADING, ENV_BASE + environmentNames[m] + ".default",
663                                       "false");
664                }
665            }
666        }
667 
668        // now that all is in order, save
669        try
670        {
671            propsFile.save();
672        }
673        catch (Exception e)
674        {
675            GUILoggerHome.find().exception("Error while saving environments to properties file: " + e.getMessage(), e);
676            rv = false;
677        }
678 
679        return rv;
680    }
