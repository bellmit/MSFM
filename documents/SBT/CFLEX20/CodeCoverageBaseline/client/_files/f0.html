686            '0', '1', '2', '3', '4', '5', '6', '7',

688    /**
689     * Convert a byte buffer into a hexadecimal string.
690     * @param buffer Input to convert
691     * @return Hexadecimal character equivalent, or empty string.
692     */
693    private String convertToHex(byte[] buffer)
694    {
695        if (buffer == null || buffer.length == 0)
696        {
697            return "";
698        }
699 
700        StringBuilder result = new StringBuilder(2*buffer.length);
701        for (byte b : buffer)
702        {
703            int high = (b >> 4) & 0xF;
704            int low = b & 0xF;
705            result.append(hexDigits[high]).append(hexDigits[low]);
706        }
707 
708        return result.toString();
709    }
