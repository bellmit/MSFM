365    /**
366     * Returns true if (based on applicatin properties) the MDX data should be translated
367     * and republished for the given exchange and session.
368     */
369    protected boolean publishForExchange(String exchange, String tradingSession, int classKey)
370    {
371        boolean retVal = false;
372        if(isMDXSupportedSession(tradingSession))
373        {
374            String[] exchanges = getDefaultExchanges(tradingSession);
375            if(exchanges != null)
376            {
377                // if wildcard "*", then publish for all exchanges
378                if(exchanges.length == 1 && "*".equals(exchanges[0]))
379                {
380                    retVal = true;
381                }
382                // if "PRIMARY", then publish for only the ProductClass' primary exchange
383                else if(exchanges.length == 1 && "PRIMARY".equals(exchanges[0]))
384                {
385                    ProductClass productClass = ProductHelper.getProductClass(classKey);
386                    if(productClass != null)
387                    {
388                        // check if exchange matches the ProductClass' primary exchg
389                        if(primaryExchangeEqualsMDXExchange(productClass.getPrimaryExchange(), exchange))
390                        {
391                            retVal = true;
392                        }
393                    }
394                }
395                // else, publish only for exchanges that are exact matches for the designated default exchanges for the session
396                else
397                {
398                    for(String tmpExchange : exchanges)
399                    {
400                        if(exchange.equals(tmpExchange))
401                        {
402                            retVal = true;
403                            break;
404                        }
405                    }
406                }
407            }
408            else
409            {
410                // if there were no exchanges designated as defaults in the properties file for an MDX-supported
411                // trading session, then it may be a miss in the properties file
412                retVal = true;
413            }
414        }
415        return retVal;
416    }
