458         * @return the value as a string, or empty string if it is not found

460        private static String extractTextValue(String text, String key) {
461                String value = "";
462                int start = text.indexOf(key+"=") + key.length() + 1;
463                if (start >= 0) {
464                        int end = text.indexOf(',', start);
465                        if (end < 0) {
466                                value = text.substring(start);
467                        } else {
468                                value = text.substring(start, end);
469                        }
470                }
471                return value;
472        }
