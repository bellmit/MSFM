451            //First check all legs are of same side, if "YES", then get side of the first leg and process
452            if(isAllLegSidesSame(orderDetails))
453            {
454                    LegOrderDetail legOrderDetail = orderDetails[0];
455                    
456                    //If all legs side are 'B' i.e. Buy
457                    if(Character.toUpperCase(legOrderDetail.getSide()) == 'B')
458                    {
459                            if(orderPrice.greaterThanOrEqual(zeroPrice))
460                            {
461                                    indicator = MarketabilityIndicator.X_NOT_MARKETABLE;
462                                    return indicator;
463                            }
464                    }
465                    //If all legs side are 'S' i.e. Sell
466                    else if(Character.toUpperCase(legOrderDetail.getSide()) == 'S')
467                    {
468                            if(orderPrice.lessThanOrEqual(zeroPrice))
469                            {
470                                    indicator = MarketabilityIndicator.X_NOT_MARKETABLE;
471                                    return indicator;
472                            }
473                    }
474            }
475        }
476        //Added until here to address PITS#62227        - NAS
477 
478        if(strategyImpliedMarketWrapper == null)
479        {
480            indicator = MarketabilityIndicator.UNKOWN;
481        }
482        else
483        {
484            orderPrice = element.getPrice();
485            double same = strategyImpliedMarketWrapper.getImpliedSame();
486            Price samePrice = DisplayPriceFactory.create(same);
487            double opposite = strategyImpliedMarketWrapper.getImpliedOpposite();
488            Price oppPrice = DisplayPriceFactory.create(opposite);
489            zeroPrice = DisplayPriceFactory.create(zero);
490            
491            //if same and opposite are both CR then it isn't marketable
492            if(samePrice.greaterThanOrEqual(zeroPrice) && oppPrice.greaterThanOrEqual(zeroPrice))
493            {
494                indicator = MarketabilityIndicator.NOT_MARKETABLE;
495            }
496            //if same and opposite are both DB then it's always marketable
497            else if(samePrice.lessThan(zeroPrice) && oppPrice.lessThan(zeroPrice))
498            {
499                indicator = MarketabilityIndicator.MARKETABLE;
500            }
501            //if same and opposite are CR and DB or DB CR then there is a check to do that the same doesn't touch opposite in order to be marketable.
502            else if(Character.toUpperCase(element.getSide()) == 'O' ||
503                    Character.toUpperCase(element.getSide()) == 'D')
504            {
505                //CR DB
506                if(samePrice.greaterThanOrEqual(zeroPrice))
507                {
508                    Price positiveOppPrice = DisplayPriceFactory.create(-1 * oppPrice.toDouble());
509                    indicator = orderPrice.lessThanOrEqual(positiveOppPrice) ?
510                                MarketabilityIndicator.MARKETABLE :
511                                MarketabilityIndicator.NOT_MARKETABLE;
512                }
513                //DB CR
514                else if(oppPrice.greaterThanOrEqual(zeroPrice))
515                {
516                    Price negativeOppPrice = DisplayPriceFactory.create(-1 * oppPrice.toDouble());
517                    indicator = orderPrice.lessThanOrEqual(negativeOppPrice) ?
518                                MarketabilityIndicator.MARKETABLE :
519                                MarketabilityIndicator.NOT_MARKETABLE;
520                }
521                else { //should never pass in this case.
522                    throw new IllegalStateException("Illegal state for a price.");
523                }
524            }
525            else
526            {
527                throw new IllegalArgumentException(
528                        "Unsupported case -- OrderMessageElement found was:" + element.toString());
529            }
530        }
531        return indicator;
532    }

535     * Assess the marketability indicator for a simple order.
536     * @param element order to assess marketability.
537     * @param currentMarketStruct for the order being assessed.

540    private static MarketabilityIndicator getMarketabilityForSimpleOrder(OrderMessageElement element,
541                                                                    CurrentMarketStruct currentMarketStruct)
542    {
543        MarketabilityIndicator indicator = MarketabilityIndicator.UNKOWN;
544        Price orderPrice = element.getPrice();
545        if(Character.toUpperCase(element.getSide()) == 'B')
546        {
547            Price askPrice = PriceFactory.create(currentMarketStruct.askPrice);
548            if(askPrice != null && (askPrice.isValuedPrice() || askPrice.isMarketPrice()))
549            {
550                if(orderPrice.greaterThanOrEqual(askPrice))
551                {
552                    indicator = MarketabilityIndicator.MARKETABLE;
553                }
554                else
555                {
556                    indicator = MarketabilityIndicator.NOT_MARKETABLE;
557                }
558            }
559        }
560        else if(Character.toUpperCase(element.getSide()) == 'S')
561        {
562            Price bidPrice = PriceFactory.create(currentMarketStruct.bidPrice);
563            if(bidPrice != null && (bidPrice.isValuedPrice() || bidPrice.isMarketPrice()))
564            {
565                if(orderPrice.lessThanOrEqual(bidPrice))
566                {
567                    indicator = MarketabilityIndicator.MARKETABLE;
568                }
569                else
570                {
571                    indicator = MarketabilityIndicator.NOT_MARKETABLE;
572                }
573            }
574        }
575        else {
576            throw new IllegalArgumentException("Unexpected side for a simple order. Side found was:" + element.getSide());
577        }
578        return indicator;
579    }

648                                        }
649                                        else

652                                        }
653                                }

655                                {

657                                }//If any leg side is other than previous leg side, then return false
658                                else if(sideOfFirstLeg != null && sideOfFirstLeg != strategyLegSide)
659                                {
660                                        return false;
661                                }
662                                else if(sideOfFirstLeg != null && sideOfFirstLeg == strategyLegSide)
663                                {
