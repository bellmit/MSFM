428                {
429                    for( int r = 0;r < subscribers.length;++r )
430                    {
431                        if( subscribers[r].getName().equals( publishers[z].getName() ) )
432                        {
433                            continue; // do not worry about self-link
434                        }
435                        DirectedChannelLink link = new DirectedChannelLink( publishers[z], subscribers[r], topic, true );
436                        linkModel.add( link );
437                    }
438                }
439            }
440        }
441        if (GUILoggerHome.find().isDebugOn())
442        {
443            GUILoggerHome.find().debug("ChannelLinkConstructor.analyze() - completed analyzing global links",GUILoggerMMBusinessProperty.CHANNELS);
444        }
445 
446        // before completing, do some re-processing.   Look through entire set of
447        // links where link == link.reverse().  In this case, remove both, add back
448        // in a Bidirectional Link (???? think about this some more ... )
449        // can't use iterator here because of double remove
450        if (GUILoggerHome.find().isDebugOn())
451        {
452            GUILoggerHome.find().debug("ChannelLinkConstructor.analyze() - searching for bidirectional links",GUILoggerMMBusinessProperty.CHANNELS);
453        }
454        Set doubleups = new HashSet();
455        for( int m = 0;m < linkModel.size();++m )
456        {
457            DirectedChannelLink l = (DirectedChannelLink)linkModel.get( m );
458            for( int j = 0;j < linkModel.size();++j )
459            {
460                DirectedChannelLink other = (DirectedChannelLink)linkModel.get( j );
461                if( l.equals( other ) )
462                {
463                    continue;
464                }
465                else
466                {
467                    if( l.equals( other.reverse() ) )
468                    {
469                        // Create birectional link?
470                        // remove linkModel(m) and linkModel(j);
471                        linkModel.remove( l );
472                        linkModel.remove( other );
473                        ChannelLink bidirectional = new ChannelLink( l.getPublisher(), l.getSubscriber(), new Topic( l.getChannelName(), l.getExtent(), false ), l.isGlobal() );
474                        doubleups.add( bidirectional );
475                    }
476                }
477            }
478        }
479        linkModel.addAll( doubleups );
480        if (GUILoggerHome.find().isDebugOn())
481        {
482            GUILoggerHome.find().debug( "Done analyzing channel links! - model contains " + linkModel.size() + " links." ,GUILoggerMMBusinessProperty.CHANNELS);
483        }
