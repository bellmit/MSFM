684    public Object getEventChannelData()
685            throws CommunicationException, SystemException, DataValidationException, AlreadyExistsException,
686                   NotAcceptedException, NotFoundException, TransactionFailedException, TimedOutException
687    {
688        Object eventChannelData = null;
689 
690        setStartTime(System.currentTimeMillis());
691 
692        synchronized(this)
693        {
694            while(!isChannelUpdated() && !isTimedOut() && !isExceptionThrown())
695            {
696                try
697                {
698                    wait(getTimeout());
699                }
700                catch(InterruptedException e)
701                {
702                    //Either timed out or interrupted for data input. Do nothing.
703                }
704                calculateIsTimedOut();
705            }
706        }
707 
708        unsubscribeEventChannel();
709 
710        if(isChannelUpdated())
711        {
712            eventChannelData = getEventData();
713        }
714        else if(isExceptionThrown())
715        {
716            processException();
717        }
718        else if(isTimedOut())
719        {
720            processTimeOut();
721        }
722 
723        return eventChannelData;
724    }

726    public synchronized void channelUpdate(ChannelEvent event)
727    {
728        ChannelKey key = (ChannelKey) event.getChannel();
729        int channelType = key.channelType;
730        ExceptionDetails eDetails;
731        switch(channelType)
732        {
733            case ChannelType.IC_ACCEPT_ALREADY_EXISTS_EXCEPTION:
734                eDetails = (ExceptionDetails) event.getEventData();
735                exception = new AlreadyExistsException(getExceptionMessage(), eDetails);
736                isChannelUpdated = false;
737                isExceptionThrown = true;
738                break;
739            case ChannelType.IC_ACCEPT_DATA_VALIDATION_EXCEPTION:
740                eDetails = (ExceptionDetails) event.getEventData();
741                exception = new DataValidationException(getExceptionMessage(), eDetails);
742                isChannelUpdated = false;
743                isExceptionThrown = true;
744                break;
745            case ChannelType.IC_ACCEPT_NOT_ACCEPTED_EXCEPTION:
746                eDetails = (ExceptionDetails) event.getEventData();
747                exception = new NotAcceptedException(getExceptionMessage(), eDetails);
748                isChannelUpdated = false;
749                isExceptionThrown = true;
750                break;
751            case ChannelType.IC_ACCEPT_NOT_FOUND_EXCEPTION:
752                eDetails = (ExceptionDetails) event.getEventData();
753                exception = new NotFoundException(getExceptionMessage(), eDetails);
754                isChannelUpdated = false;
755                isExceptionThrown = true;
756                break;
757            case ChannelType.IC_ACCEPT_SYSTEM_EXCEPTION:
758                eDetails = (ExceptionDetails) event.getEventData();
759                exception = new SystemException(getExceptionMessage(), eDetails);
760                isChannelUpdated = false;
761                isExceptionThrown = true;
762                break;
763            case ChannelType.IC_ACCEPT_TRANSACTION_FAILED_EXCEPTION:
764                eDetails = (ExceptionDetails) event.getEventData();
765                exception = new TransactionFailedException(getExceptionMessage(), eDetails);
766                isChannelUpdated = false;
767                isExceptionThrown = true;
768                break;
769            default:
770                this.eventChannelData = event.getEventData();
771                isChannelUpdated = true;
772                isExceptionThrown = false;
773                break;
774        }
