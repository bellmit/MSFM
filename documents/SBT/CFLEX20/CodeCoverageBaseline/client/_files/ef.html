690     * @return duration in seconds, or 0 on error.
691     */
692    private int parseDuration(String duration)
693    {
694        int hours = 0, minutes = 0, seconds = 0;
695        String[] parts = duration.split("[HMS:hms]");
696 
697        if (colonPattern.matcher(duration).matches())
698        {
699            int nextPart = 0;
700            if (parts.length >= 3)  // can be 3, should never be greater
701            {
702                hours = Integer.parseInt(parts[nextPart++]);
703            }
704            if (parts.length >= 2)
705            {
706                minutes = Integer.parseInt(parts[nextPart++]);
707            }
708            seconds = Integer.parseInt(parts[nextPart]);
709        }
710        else if (hmsPattern.matcher(duration).matches())
711        {
712            int nextPart = 0;
713            if (duration.contains("h") || duration.contains("H"))
714            {
715                hours = Integer.parseInt(parts[nextPart++]);
716            }
717            if (duration.contains("m") || duration.contains("M"))
718            {
719                minutes = Integer.parseInt(parts[nextPart++]);
720            }
721            if (duration.contains("s") || duration.contains("S"))
722            {
723                seconds = Integer.parseInt(parts[nextPart]);
724            }
725        }
726        else
727        {
728            // Badly formatted input, leave
729            return 0;
730        }
731 
732        return (hours*60 + minutes)*60 + seconds;
