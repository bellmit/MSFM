450    public int linkageCancel(Order p_order, int productKey, String userAssignedCancelId, 
451            short cancelReason, short p_reportType, OrderHistoryCommonInfo orderHistoryCommonInfo)
452    {
453        int cancelledQuantity = p_order.getRemainingQuantity();
454        cancelWithoutReport(p_order, cancelledQuantity, false,orderHistoryCommonInfo);
455 
456        // Pull the user assigned cancel ID, which in here is an extensions field.  It cannot be
457        // an extensions field for the CMi, so we need to remove server assumptions and put the
458        // real user assigned cancel id, if one exists, where it belongs for the CMi (in that field on the
459        // CancelReportStruct).  Additional fields will be dealt with as needed.
460        ExtensionsHelper helper = p_order.getExtensionsHelper(userAssignedCancelId, LINKAGE_CANCEL_REPORT);
461 
462        // This is the actual expected cancel ID that will go into the CMi struct for the report.
463        String expectedCancelId = null;
464 
465        // Pull the exec ID out and put it in this object's extensions field.
466        // This is the best way to get the exec ID to the FIXCAS.  (ExecId is a required tag).
467        // There can be at most one OLA Execution Report message (OLA FIX message type) that
468        // meets the following criteria: (reportType = NEW_ORDER_REJECT || reportType = REGULAR_REPORT)
469        // -- hence there is only one possible FIX ExecId that can apply to any portion of the order
470        // for cancel report or new order reject cases.  (For fills the exec id is coded into the
471        // FilledReportStruct.extensions field.
472        String execId = helper.getValue(ExtensionFields.AWAY_EXCHANGE_EXEC_ID);
473        if (execId != null && execId.length() > 0)
474        {
475            ExtensionsHelper orderExtensionsHelper = p_order.getExtensionsHelper(p_order.getExtensions(), 
476                    LINKAGE_CANCEL_REPORT);
477            addValueToExtensionsHelper(orderExtensionsHelper, ExtensionFields.USER_ASSIGNED_CANCEL_ID, expectedCancelId);
478            addValueToExtensionsHelper(orderExtensionsHelper, ExtensionFields.AWAY_CANCEL_REPORT_EXEC_ID, execId);
479            expectedCancelId = orderExtensionsHelper.toString();
480            p_order.setExtensions(expectedCancelId);
481        }
482 
483        
484        String executingExchange = p_order.getExecutingExchange();
485        String exchangeExecIdString = executingExchange + "/" + execId;
486 
487        int tlcQuantity = 0; // tlcQty > 0 implies _user_ cancel request, currently not supported.
488 
489        CancelReportStruct report = buildCancelReportStructForLinkageCancel(p_order, cancelReason, cancelledQuantity, 
490                                                     tlcQuantity, expectedCancelId, p_reportType);
491        
492        if (cancelledQuantity > 0 && p_order.getRemainingQuantity() == 0)
493        {
494            p_order.setState(OrderStates.CANCEL);
495        }
496 
497        if (Log.isDebugOn())
498        {
499            Log.debug("Processed cancel report for order (" + p_order.getOrderIdString() +
500                    "); exchange/execId=(" + exchangeExecIdString + ") - all remaining quantity canceled.");
501        }
502 
503        CancelReportStruct[] result;
504        result = new CancelReportStruct[1];
505        result[0] = report;
506 
507        p_order.publishCancelReports(result, cancelReason, 0);
508                    
509            p_order.getOrderHistoryHome().createHistoryCancelOrder(p_order, cancelReason,
510                                                                   p_order.getRemainingQuantity(), cancelledQuantity ,
511                                                                   report.tlcQuantity, userAssignedCancelId,
512                                                                   orderHistoryCommonInfo);
513        return cancelledQuantity;
514    }

555    private CancelReportStruct[] doOrphanedCancel(Order order, short cancelReason, int quantity, String userAssignedCancelId, 
556            boolean createHistory,OrderHistoryCommonInfo orderHistoryCommonInfo)
557    {
558        cancelReason = InternalActivityReasons.getMappedExternalReason(cancelReason);
559 
560        int cancelledQuantity = cancelWithoutReport(order, quantity, false,orderHistoryCommonInfo);
561        CancelReportStruct report = new CancelReportStruct();
562        populateCommonValuesForReport(order, report);
563        report.cancelReason = cancelReason;
564        report.cancelReportType = order.getProductType() == ProductTypes.STRATEGY ? ReportTypes.STRATEGY_REPORT : ReportTypes.REGULAR_REPORT;
565        report.productKey = order.getPrimitiveProductKey();
566        report.cancelledQuantity = cancelledQuantity;
567        report.totalCancelledQuantity = order.getCancelledQuantity();
568        report.userAssignedCancelId = userAssignedCancelId;
569 
570        if (cancelledQuantity > 0 && order.getRemainingQuantity() == 0) 
571        {
572            order.setState(OrderStates.CANCEL);
573            order.markCancelReason((short)0);
574        }
575 
576        if (createHistory)
577        {
578            order.getOrderHistoryHome().createHistoryCancelOrphanedOrder(order, cancelReason, order.getRemainingQuantity() , cancelledQuantity ,
579                    0, userAssignedCancelId, orderHistoryCommonInfo);
580        }
581 
582        CancelReportStruct[] result;
583 
584        if (order.getProductType() == ProductTypes.STRATEGY)
585        {
586            result = order.cancelLegsOnOrphaned(report, quantity, createHistory);
587        }
588        else
589        {
590            result = new CancelReportStruct[1];
591            result[0] = report;
592        }
593        return result;
594    }
