482    private boolean shouldCancelForRestrictedSeries(Order order) throws SystemException
483    {

485        boolean isOpenPosition = false;

487        if(order.isStrategy()) {

490            LegOrderDetailStruct[] orderLegs =  order.getStruct().legOrderDetails;

492                boolean legRSS = getProductData(legs.get(i).getProductKey()).isRestrictedIndicator();
493                isOpenPosition = orderLegs[i].positionEffect == PositionEffects.OPEN;
494                if(legRSS && isOpenPosition) {
495                    isProductRestricted = true; 
496                    break;
497                }
498            }
499        }
500        else {
501            isOpenPosition = order.getPositionEffect() == PositionEffects.OPEN;
502            isProductRestricted = prodData.isRestrictedIndicator();
503        }
504        boolean isRSSEligible = isRestrictedSeriesEligible(order);
505        return isProductRestricted && isRSSEligible && isOpenPosition;
506    }

913    {

915        int[] orphanedOrderClassKeys = getOrderClassKeys(orders);
916        if (orphanedOrderClassKeys.length == 0)

919        }

921        ProductClassStructV4[] prodClassStructs = getProductClassesByKeys(orphanedOrderClassKeys);
922        List<Order> ordersToBeCancelled = new ArrayList<Order>();
923        for (Order anOrder : orders)
924        {

926            for (ProductClassStructV4 classStruct : prodClassStructs )

934                    }
935                    else

938                        for (ProductStruct theStruct : prodStructs)
939                        {
940                            if (theStruct.productKeys.productKey == anOrder.getPrimitiveProductKey())
941                            {

943                                break;
944                            }
945                        }
946                    }
947                    anOrder.setActiveClass(listingState == 1 ? true : false);

949                    break;
950                }
951            }
952        }
