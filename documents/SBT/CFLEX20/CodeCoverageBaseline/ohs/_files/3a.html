396            boolean isOriginalOrderISO = (originalOrder.getRealContingencyType() == ContingencyTypes.INTERMARKET_SWEEP);

404            boolean isOriginalOrderISO_BOOK = (originalOrder.getRealContingencyType() == ContingencyTypes.INTERMARKET_SWEEP_BOOK);

409                        DataValidationCodes.INVALID_CONTINGENCY_TYPE);
410            }
411            
412            // Check for Do Not Route Orders //
413            boolean isOriginalOrderDoNotRoute = (originalOrder.getRealContingencyType() == ContingencyTypes.DO_NOT_ROUTE);
414            boolean isReplacedOrderDoNotRoute = (replacedOrderStruct.contingency.type == ContingencyTypes.DO_NOT_ROUTE);
415            
416            if ( (isOriginalOrderDoNotRoute == false) && (isReplacedOrderDoNotRoute == true) )

419                        DataValidationCodes.INVALID_CONTINGENCY_TYPE);
420            }
421        }
422        
423        if (!SessionNameHelper.isEquitySession(originalOrder.getActiveSession()))
424        {
425            // For W_STOCK allow RESERVE contingency replacements   
426        // Check for Reserve Orders //
