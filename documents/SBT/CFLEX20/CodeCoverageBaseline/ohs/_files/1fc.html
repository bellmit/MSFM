635    private void incrementParPendingCancelQty(BaseOHSContext p_context, Order p_order, CancelRequestStruct p_cancelRequestStruct) throws TransactionFailedException 
636    {        
637        int cancelQuantity = 0;
638        if (p_cancelRequestStruct.cancelType == OrderCancelTypes.CANCEL_ALL_QUANTITY)  {
639            cancelQuantity = p_order.getRemainingQuantity();
640        }
641        else if (p_cancelRequestStruct.cancelType == OrderCancelTypes.DESIRED_CANCEL_QUANTITY)  {
642            cancelQuantity = p_cancelRequestStruct.quantity;
643        }
644        else if (p_cancelRequestStruct.cancelType == OrderCancelTypes.DESIRED_REMAINING_QUANTITY) {
645            cancelQuantity = p_order.getRemainingQuantity() - p_cancelRequestStruct.quantity;
