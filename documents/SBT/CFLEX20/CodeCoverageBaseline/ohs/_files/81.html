479     * @param orderOriginType
480     * @param timeInMillis
481     * @return
482     */
483    @FFCommand(description = "Change Order Expiration Time ", paramDescriptions = { "originTyp", "valueInMillis" })
484    public String adminModifyExpirationtimer(String orderOriginType, String timeInMillis)
485    {
486        java.io.StringWriter stringWriter = new java.io.StringWriter();
487        try
488        {
489            String sessionName = "W_MAIN";
490            StringBuffer resultSet = new StringBuffer();
491            if(!(System.getProperty("EventService.Admin.AdminPrefix").startsWith("Prod")))
492            {
493 
494                char timerOriginType = orderOriginType.charAt(0);
495                long newExpirationTime = Long.parseLong(timeInMillis);
496                switch(timerOriginType)
497                {
498                    case OrderOrigins.PRINCIPAL :
499                        getLOHSConfigPropertiesForSession(sessionName).setPOrderTimerExpirationInMillis(newExpirationTime);
500                        service.setPOrderExpirationTimeInMillis(newExpirationTime);
501                        break;
502                    case OrderOrigins.PRINCIPAL_ACTING_AS_AGENT :
503                        getLOHSConfigPropertiesForSession(sessionName).setPAOrderTimerExpirationInMillis(newExpirationTime);
504                        service.setPAOrderExpirationTimeInMillis(newExpirationTime);
505                        break;
506                    case OrderOrigins.SATISFACTION :
507                        getLOHSConfigPropertiesForSession(sessionName).setSOrderTimerExpirationInMillis(newExpirationTime);
508                        service.setSOrderExpirationTimeInMillis(newExpirationTime);
509                        break;
510                    default :
511                        stringWriter.write("Invalid Order originType for this Command. Valid values are P, A or S .\n");
512                }
513            }
514            else
515            {
516                stringWriter.write("Modifcation Expiration Timer Admin Request is only available in DEVELOPMENT Environment.\n");
517            }
518            stringWriter.write(resultSet.toString());
519        }
520        catch (Throwable ex)
521        {
522            java.io.PrintWriter printWriter = new java.io.PrintWriter(stringWriter);
523            stringWriter.write("Error Changing the Expiration Timer: " + ex + "\n");
524            ex.printStackTrace(printWriter);
525            printWriter.flush();
526        }
527        stringWriter.flush();
528        return stringWriter.toString();
529    }
